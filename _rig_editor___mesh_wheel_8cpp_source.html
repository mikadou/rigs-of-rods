65 void MeshWheel::GenerateMeshWheelGeometry(

70  Ogre::Vector3* rigidity_point_ptr,

75  LandVehicleWheel::Type wheel_type,
76  Ogre::AxisAlignedBox & out_aabb
77  )
78 {
79  // GENERATE VERTICES
80 
81  LandVehicleWheel::BuildWheelNodes(out_vertices, num_rays, axis_point_inner, axis_point_outer, reference_arm_node, rigidity_point_ptr, wheel_radius, out_aabb);
82 
83  // GENERATE EDGES
84 
85  // Prepare buffers
86  out_edges.clear();
87  out_edges.reserve((rigidity_point_ptr != nullptr) ? num_rays * 8 : num_rays * 9);
88  // Determine edge types
89  Edge::Type bounded_type = Edge::TYPE_INVALID;
90  Edge::Type reinforcement_type = Edge::TYPE_INVALID;
91  Edge::Type rigidity_type = Edge::TYPE_INVALID;
92  switch (wheel_type)
93  {
94  case LandVehicleWheel::TYPE_MESHWHEEL:
95  bounded_type = Edge::TYPE_MESHWHEEL_BOUNDED;
96  reinforcement_type = Edge::TYPE_MESHWHEEL_REINFORCEMENT;
97  rigidity_type = Edge::TYPE_MESHWHEEL_RIGIDITY;
98  break;
99  case LandVehicleWheel::TYPE_MESHWHEEL_2:
100  bounded_type = Edge::TYPE_MESHWHEEL2_BOUNDED;
101  reinforcement_type = Edge::TYPE_MESHWHEEL2_REINFORCEMENT;
102  rigidity_type = Edge::TYPE_MESHWHEEL2_RIGIDITY;
103  break;
104  default:
105  break;
106  }
107  // Generate!
108  const unsigned axis_inner_index = 0;
109  const unsigned axis_outer_index = 1;
110  const unsigned rigidity_index = 2;
111  const unsigned base_index = 3;
112  for (unsigned int i = 0; i < num_rays; ++i)
113  {
114  // Bounded
115  unsigned int inner_point_index = base_index + (i * 2);
116  unsigned int outer_point_index = inner_point_index + 1;
117 
118  out_edges.push_back(Edge(axis_inner_index, inner_point_index, bounded_type));
119  out_edges.push_back(Edge(axis_outer_index, outer_point_index, bounded_type));
120  out_edges.push_back(Edge(axis_outer_index, inner_point_index, bounded_type));
121  out_edges.push_back(Edge(axis_inner_index, outer_point_index, bounded_type));
122 
123  // Reinforcement
124  unsigned int next_inner_index = base_index + ((i + 1) % num_rays) * 2;
125  unsigned int next_outer_index = next_inner_index + 1;
126 
127  out_edges.push_back(Edge(inner_point_index, outer_point_index, reinforcement_type));
128  out_edges.push_back(Edge(inner_point_index, next_inner_index, reinforcement_type));
129  out_edges.push_back(Edge(outer_point_index, next_outer_index, reinforcement_type));
130  out_edges.push_back(Edge(outer_point_index, next_inner_index, reinforcement_type));
131 
132  // Rigidity beams
133  if (rigidity_point_ptr != nullptr)
134  {
135  unsigned rigidity_target_index = (rigidity_connect_outer_ring) ? outer_point_index : inner_point_index;
136  out_edges.push_back(Edge(rigidity_index, rigidity_target_index, bounded_type));
137  }
138  }
139 }
