102  virtual bool syncRemoteStreams()
103  {
104  lockStreams();
105  // first registrations
106  int changes = 0;
107  while (!stream_registrations.empty())
108  {
109  stream_reg_t reg = stream_registrations.front();
110  Streamable *s = createRemoteInstance(&reg);
111  if (s)
112  {
113  // add it to the streams list
114  NetworkStreamManager::getSingleton().addRemoteStream(s, reg.sourceid, reg.streamid);
115  reg.reg.status = 1;
116  } else
117  {
118  // error creating stream, tell the sourceid that it failed
119  reg.reg.status = -1;
120  }
121  // fixup the registration information
122  reg.reg.origin_sourceid = reg.sourceid;
123  reg.reg.origin_streamid = reg.streamid;
124 
125  // only save registration results for beam streams
126  // TODO: maybe enforce general design to allow all stream types to
127  // have a feedback channel
128  if (reg.reg.type == 0)
129  {
130  stream_creation_results.push_back(reg);
131  }
132  // remove registration from list
133  stream_registrations.pop_front();
134  changes++;
135  }
136 
137  // count the stream creation results into the changes
138  changes += (int)stream_creation_results.size();
139 
140  // then deletions:
141  // first registrations
142  while (!stream_deletions.empty())
143  {
144  stream_del_t del = stream_deletions.front();
145  removeInstance(&del);
146  stream_deletions.pop_front();
147  changes++;
148  }
149  unlockStreams();
150  return (changes > 0);
151  }
