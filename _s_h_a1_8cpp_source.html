00138 {
00139     uint32_t i, j;

00141     j = (m_count[0] >> 3) & 63;
00142 
00143     if ((m_count[0] += len << 3) < (len << 3)) m_count[1]++;
00144

00146 
00147     if ((j + len) > 63)
00148     {
00149         i = 64 - j;
