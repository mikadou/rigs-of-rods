63 void SkinManager::parseScript(DataStreamPtr& stream, const String& groupName)
64 {
65  try
66  {
67  String line = "";
68  Skin *pSkin = 0;
69 
70  while(!stream->eof())
71  {
72  line = stream->getLine();
73 
74  // Ignore blanks & comments
75  if (!line.length() || line.substr(0, 2) == "//")
76  {
77  continue;
78  }
79 
80  if (!pSkin)
81  {
82  // No current skin
83  // So first valid data should be skin name
84  pSkin = (Skin *)createResource(line, groupName).getPointer();
85  if (pSkin)
86  {
87  pSkin->_notifyOrigin(stream->getName());
88  stream->skipLine("{");
89  }
90  } else
91  {
92  // Already in skin
93  if (line == "}")
94  {
95  // Finished
96  //addImpl((Ogre::ResourcePtr)pSkin);
97  pSkin = 0;
98  // NB skin isn't loaded until required
99  } else
100  {
101  ParseSkinAttribute(line, pSkin);
102  }
103  }
104  }
105  } catch(Ogre::ItemIdentityException e)
106  {
107  // this catches duplicates -> to be ignored
108  // this happens since we load the full skin data off the cache, so we don't need
109  // to re-add it to the SkinManager
110  return;
111  }
112 }

114 void SkinManager::ParseSkinAttribute(const String& line, Skin *pSkin)
115 {
116  Ogre::StringVector params = StringUtil::split(line, "\t=,;\n");
117  for (unsigned int i=0; i < params.size(); i++)
118  {
119  StringUtil::trim(params[i]);
120  }
121  String& attrib = params[0];
122  StringUtil::toLowerCase(attrib);
123 
124  if (attrib == "replacetexture" && params.size() == 3) pSkin->addTextureReplace(params[1], params[2]);
125  if (attrib == "replacematerial" && params.size() == 3) pSkin->addMaterialReplace(params[1], params[2]);
126  else if (attrib == "preview" && params.size() >= 2) pSkin->thumbnail = params[1];
127  else if (attrib == "description" && params.size() >= 2) pSkin->description = params[1];
128  else if (attrib == "authorname" && params.size() >= 2) pSkin->authorName = params[1];
129  else if (attrib == "authorid" && params.size() == 2) pSkin->authorID = PARSEINT(params[1]);
130  else if (attrib == "guid" && params.size() >= 2) pSkin->guid = params[1];
131  else if (attrib == "name" && params.size() >= 2) pSkin->name = params[1];
132 
133  StringUtil::trim(pSkin->name);
134 }
