00063 void SkinManager::parseScript(DataStreamPtr& stream, const String& groupName)
00064 {
00065     try
00066     {
00067         String line = "";
00068         Skin *pSkin = 0;
00069 
00070         while(!stream->eof())
00071         {
00072             line = stream->getLine();
00073 
00074             // Ignore blanks & comments
00075             if (!line.length() || line.substr(0, 2) == "//")
00076             {
00077                 continue;
00078             }
00079 
00080             if (!pSkin)
00081             {
00082                 // No current skin
00083                 // So first valid data should be skin name
00084                 pSkin = (Skin *)createResource(line, groupName).getPointer();
00085                 if (pSkin)
00086                 {
00087                     pSkin->_notifyOrigin(stream->getName());
00088                     stream->skipLine("{");
00089                 }
00090             } else
00091             {
00092                 // Already in skin
00093                 if (line == "}")
00094                 {
00095                     // Finished
00096                     //addImpl((Ogre::ResourcePtr)pSkin);
00097                     pSkin = 0;
00098                     // NB skin isn't loaded until required
00099                 } else
00100                 {
00101                     ParseSkinAttribute(line, pSkin);
00102                 }
00103             }
00104         }
00105     } catch(Ogre::ItemIdentityException e)
00106     {
00107         // this catches duplicates -> to be ignored
00108         // this happens since we load the full skin data off the cache, so we don't need
00109         // to re-add it to the SkinManager
00110         return;
00111     }
00112 }

00114 void SkinManager::ParseSkinAttribute(const String& line, Skin *pSkin)
00115 {
00116     Ogre::StringVector params = StringUtil::split(line, "\t=,;\n");
00117     for (unsigned int i=0; i < params.size(); i++)
00118     {
00119         StringUtil::trim(params[i]);
00120     }
00121     String& attrib = params[0];
00122     StringUtil::toLowerCase(attrib);
00123 
00124     if      (attrib == "replacetexture"     && params.size() == 3) pSkin->addTextureReplace(params[1], params[2]);
00125     if      (attrib == "replacematerial"    && params.size() == 3) pSkin->addMaterialReplace(params[1], params[2]);
00126     else if (attrib == "preview"            && params.size() >= 2) pSkin->thumbnail = params[1];
00127     else if (attrib == "description"        && params.size() >= 2) pSkin->description = params[1];
00128     else if (attrib == "authorname"         && params.size() >= 2) pSkin->authorName = params[1];
00129     else if (attrib == "authorid"           && params.size() == 2) pSkin->authorID = PARSEINT(params[1]);
00130     else if (attrib == "guid"               && params.size() >= 2) pSkin->guid = params[1];
00131     else if (attrib == "name"               && params.size() >= 2) pSkin->name = params[1];
00132 
00133     StringUtil::trim(pSkin->name);
00134 }
