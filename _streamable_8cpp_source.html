112 void Streamable::addReceivedPacket(header_t header, char *buffer)
113 {
114  if (packets.size() > packetBufferSizeDiscardData && header.command == MSG2_STREAM_DATA)
115  // discard unimportant data packets for some while
116  return;
117 
118  if (receivedPackets.size() > packetBufferSize)
119  // buffer full, packet discarded

121  MUTEX_LOCK(&recv_work_mutex);
122 
123  // construct the data holding struct
124  recvPacket_t packet;
125  memset(&packet, 0, sizeof(packet)); // we need to do this, since we use String(buffer) at some point which will crash otherwise
126  packet.header = header;
