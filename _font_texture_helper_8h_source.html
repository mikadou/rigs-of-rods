41 int generateFontTexture(Ogre::String fontName, bool load=false)
42 {
43  Ogre::FontPtr font = Ogre::FontManager::getSingleton().getByName(fontName);
44  if (font.isNull()) return 1;
45 
46  if (font->getType() != Ogre::FT_TRUETYPE)
47  return 1;
48 
49  if (!font->isLoaded())
50  font->load();
51 
52  Ogre::String texname = font->getMaterial()->getTechnique(0)->getPass(0)->getTextureUnitState(0)->getTextureName();
53 
54 
55  Ogre::TexturePtr fontTexture = Ogre::TextureManager::getSingleton().getByName(texname);
56  Ogre::HardwarePixelBufferSharedPtr fontBuffer = fontTexture->getBuffer();
57 
58  if (fontBuffer.isNull())
59  return 1;
60 
61  // create a remporary buffer that holds the font
62  size_t nBuffSize = fontBuffer->getSizeInBytes();
63  Ogre::uint8* buffer = (Ogre::uint8*)calloc(nBuffSize, sizeof(Ogre::uint8));
64  Ogre::PixelBox fontPb(fontBuffer->getWidth(), fontBuffer->getHeight(),fontBuffer->getDepth(), fontBuffer->getFormat(), buffer);
65  fontBuffer->blitToMemory(fontPb);
66 
67  // create new render texture
68  Ogre::String texNameManual = "FontTextureRTT_"+font->getName();
69  Ogre::TexturePtr t = Ogre::TextureManager::getSingleton().createManual(texNameManual,"General",Ogre::TEX_TYPE_2D,(Ogre::uint)fontBuffer->getWidth(),(Ogre::uint)fontBuffer->getHeight(),(Ogre::uint)fontBuffer->getDepth(),1,fontBuffer->getFormat(), Ogre::TU_RENDERTARGET);
70  Ogre::TexturePtr fontTextureRTT = Ogre::TextureManager::getSingleton().getByName(texNameManual);
71  fontTextureRTT->getBuffer()->blitFromMemory(fontPb);
72 
73  // now render to file
74  Ogre::RenderTexture* pRenderTexture = fontTextureRTT->getBuffer()->getRenderTarget();
75  pRenderTexture->update();
76  Ogre::String outImageName = font->getName()+ cache_name + ".png";
77  pRenderTexture->writeContentsToFile(outImageName);
78 
79  // free stuff
80  free(buffer);
81  Ogre::TextureManager::getSingleton().remove(texNameManual);
82 
83 
84  // save texture font infos
85  Ogre::UTFString out_text = font->getName()+cache_name+"\n{\n\ttype\timage\n\tsource\t"+outImageName+"\n";
86 
87  Ogre::vector<Ogre::Font::CodePointRange>::type ranges = font->getCodePointRangeList();
88  for (Ogre::vector<Ogre::Font::CodePointRange>::type::iterator it = ranges.begin(); it != ranges.end(); it++)
89  {
90  // iterate over all known glyphs
91  for (Ogre::uint32 i=it->first;i<it->second;i++)
92  {
93  Ogre::Font::GlyphInfo gi(0, Ogre::Font::UVRect(), 0);
94  try
95  {
96  gi = font->getGlyphInfo(i);
97  }
98  catch(...)
99  {
100  // catch code point not found error
101  continue;
102  }
103 
104  // create character
105 #ifdef FONT_UNICODE
106  wchar_t tmp[20];
107  swprintf(tmp, 20, L"%c", i);
108 #else
109  char tmp[20] = "";
110  sprintf(tmp, "%c", i);
111 #endif
112  out_text.append("\tglyph " + Ogre::UTFString(tmp) + " " + \
113  TOSTRING(gi.uvRect.left)+" "+ \
114  TOSTRING(gi.uvRect.top)+" "+ \
115  TOSTRING(gi.uvRect.right)+" "+ \
116  TOSTRING(gi.uvRect.bottom)+"\n");
117 
118  }
119  }
120  out_text.append("}\n");
121 
122  Ogre::String defFileName = font->getName() + cache_name + ".fontdef";
123  std::ofstream f;
124  f.open(defFileName.c_str());
125  f << out_text;
126  f.close();
127  LOG("generated font cache for font "+fontName+" ("+outImageName+", "+defFileName+")");
128 
129  if (load)
130  {
131  // reload to find the .fontdef file
132  Ogre::ResourceGroupManager::getSingleton().initialiseAllResourceGroups();
133  // load now
134  Ogre::FontPtr cached_font = Ogre::FontManager::getSingleton().getByName(fontName+cache_name);
135  if (cached_font.isNull()) return 1;
136  cached_font->load();
137  return 0;
138  }
139  return 0;
140 }

145 int fontCacheInit(Ogre::String fontName)
146 {
147  Ogre::FontPtr font_org = Ogre::FontManager::getSingleton().getByName(fontName);
148  Ogre::FontPtr font_cache = Ogre::FontManager::getSingleton().getByName(fontName + cache_name);
149 
150  // base font not even existing?
151  if (font_org.isNull()) return 1;
152  // using texture fonts is a bit stupid ...
153  if (font_org->getType() != Ogre::FT_TRUETYPE) return 1;
154 
155  int res = 0;
156  if (font_cache.isNull())
157  res = generateFontTexture(fontName);
158 
159  if (font_org->isLoaded()) font_org->unload();
160  return res;
161 }
