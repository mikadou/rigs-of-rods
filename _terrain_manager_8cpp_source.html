141 void TerrainManager::loadTerrainConfigBasics(Ogre::DataStreamPtr &ds)
142 {
143  // now generate the hash of it
144  generateHashFromDataStream(ds, file_hash);
145 
146  m_terrain_config.load(ds, "\t:=", true);
147 
148  // read in the settings
149  terrain_name = m_terrain_config.getSetting("Name", "General");
150  if (terrain_name.empty())
151  {
152  ErrorUtils::ShowError(_L("Terrain loading error"), _L("the terrain name cannot be empty"));
153  exit(125);
154  }
155 
156  ogre_terrain_config_filename = m_terrain_config.getSetting("GeometryConfig", "General");
157  // otc = ogre terrain config
158  if (ogre_terrain_config_filename.find(".otc") == String::npos)
159  {
160  ErrorUtils::ShowError(_L("Terrain loading error"), _L("the new terrain mode only supports .otc configurations"));
161  exit(125);
162  }
163 
164  ambient_color = StringConverter::parseColourValue(m_terrain_config.getSetting("AmbientColor", "General"), ColourValue::White);
165  category_id = StringConverter::parseInt(m_terrain_config.getSetting("CategoryID", "General"), 129);
166  guid = m_terrain_config.getSetting("GUID", "General");
167  start_position = StringConverter::parseVector3(m_terrain_config.getSetting("StartPosition", "General"), Vector3(512.0f, 0.0f, 512.0f));
168  version = StringConverter::parseInt(m_terrain_config.getSetting("Version", "General"), 1);
169  gravity = StringConverter::parseReal(m_terrain_config.getSetting("Gravity", "General"), -9.81);
170 
171  // parse author info
172  ConfigFile::SettingsIterator it = m_terrain_config.getSettingsIterator("Authors");
173 
174  authors.clear();
175 
176  while (it.hasMoreElements())
177  {
178  String type = it.peekNextKey(); // e.g. terrain
179  String name = it.peekNextValue(); // e.g. john doe
180 
181  if (!name.empty())
182  {
183  authorinfo_t author;
184 
185  author.type = type;
186  author.name = name;
187 
188  authors.push_back(author);
189  }
190 
191  it.moveNext();
192  }
193 }

207  exit(125);

233  PROGRESS_WINDOW(90, _L("Loading Terrain Objects"));

584 void TerrainManager::initWater()
585 {
586  String waterSettingsString = SSETTING("Water effects", "Hydrax");
587 
588  // disabled in global config
589  if (waterSettingsString == "None") return;
590  // disabled in map config
591  if (!StringConverter::parseBool(m_terrain_config.getSetting("Water", "General"))) return;
592 
593  if (waterSettingsString == "Hydrax")
594  {
595  // try to load hydrax config
596  String hydraxConfig = m_terrain_config.getSetting("HydraxConfigFile", "General");
597 
598  if (!hydraxConfig.empty() && ResourceGroupManager::getSingleton().resourceExistsInAnyGroup(hydraxConfig))
599  {
600  hw = new HydraxWater(m_terrain_config, hydraxConfig);
601  }
602  else
603  {
604  // no config provided, fall back to the default one
605  hw = new HydraxWater(m_terrain_config);
606  }
607 
608 
609  water = hw;
610 
611  //Apply depth technique to the terrain
612  TerrainGroup::TerrainIterator ti = geometry_manager->getTerrainGroup()->getTerrainIterator();
613  while (ti.hasMoreElements())
614  {
615  Terrain* t = ti.getNext()->instance;
616  MaterialPtr ptr = t->getMaterial();
617  hw->GetHydrax()->getMaterialManager()->addDepthTechnique(ptr->createTechnique());
618  }
619 
620  } else
621  {
622  if (water == nullptr)
623  water = new Water(m_terrain_config);
624  else if (water != nullptr)
625  {
626  delete(water);
627  water = new Water(m_terrain_config);
628  }
629  }
630 }
