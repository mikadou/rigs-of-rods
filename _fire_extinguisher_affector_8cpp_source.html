00076     {
00077         ExtinguishableFireAffectorFactory::affectorIterator affIt = mEfaf->getAffectorIterator();
00078         ExtinguishableFireAffector* fire;
00079 
00080         while(affIt.hasMoreElements())
00081         {
00082             fire = (ExtinguishableFireAffector*)affIt.getNext();
00083 
00084             if (fire->isTemplate())
00085                 continue;
00086 
00087             Real squaredRadius = Math::Pow(fire->getRadius(), 2);
00088             Vector3 middlePoint = fire->getAbsoluteMiddlePoint();
00089 
00090             ParticleIterator pi = pSystem->_getIterator();
00091             Particle *p;
00092             int fireHits = 0;
00093             while (!pi.end())
00094             {
00095                 p = pi.getNext();
00096 
00097                 if ( middlePoint.squaredDistance(p->position) < squaredRadius )
00098                 {
00099                     // This particle is inside the fire, dispose of it in the next update
00100                     p->timeToLive = 0;
00101                     ++fireHits;
00102                 }
00103             }
00104             if (fireHits>0)
00105             {
00106                 Real intensity = fire->reduceIntensity(fireHits*mEffectiveness);
00107                 if (intensity<0) delete fire->getParticleSystem();
