125 {
126  g_ctxMgr = this;
127 
128  // TODO: Should have an optional time out for this function. If not all scripts executed before the
129  // time out, the next time the function is called the loop should continue
130  // where it left off.
131 
132  // TODO: There should be a time out per thread as well. If a thread executes for too
133  // long, it should be aborted. A group of co-routines count as a single thread.
134 
135  // Check if the system time is higher than the time set for the contexts
136  asUINT time = m_getTimeFunc ? m_getTimeFunc() : asUINT(-1);
137  for( m_currentThread = 0; m_currentThread < m_threads.size(); m_currentThread++ )
138  {
139  SContextInfo *thread = m_threads[m_currentThread];
140  if( thread->sleepUntil < time )
141  {
142  int currentCoRoutine = thread->currentCoRoutine;
143 
144  // Gather some statistics from the GC
145  asIScriptEngine *engine = thread->coRoutines[currentCoRoutine]->GetEngine();
146  asUINT gcSize1, gcSize2, gcSize3;
147  engine->GetGCStatistics(&gcSize1);
148 
149  // Execute the script for this thread and co-routine
150  int r = thread->coRoutines[currentCoRoutine]->Execute();
151 
152  // Determine how many new objects were created in the GC
153  engine->GetGCStatistics(&gcSize2);
154  m_numGCObjectsCreated += gcSize2 - gcSize1;
155  m_numExecutions++;
156 
157  if( r != asEXECUTION_SUSPENDED )
158  {
159  // TODO: It should be possible to retrieve the return value before the context is released.
160  // Maybe by calling a callback, or by storing the context somewhere until it has been
161  // accessed.
162 
163  // The context has terminated execution (for one reason or other)
164  thread->coRoutines[currentCoRoutine]->Release();

166

168  if( thread->currentCoRoutine >= thread->coRoutines.size() )
169  thread->currentCoRoutine = 0;
170

172  if( thread->coRoutines.size() == 0 )
173  {

175  m_threads.erase(m_threads.begin() + m_currentThread);
176  m_currentThread--;
177  }
178  }
179 
180  // Destroy all known garbage if any new objects were created
181  if( gcSize2 > gcSize1 )
182  {
183  engine->GarbageCollect(asGC_FULL_CYCLE | asGC_DESTROY_GARBAGE);
184 
185  // Determine how many objects were destroyed
186  engine->GetGCStatistics(&gcSize3);
187  m_numGCObjectsDestroyed += gcSize3 - gcSize2;
188  }
189 
190  // TODO: If more objects are created per execution than destroyed on average
191  // then it may be necessary to run more iterations of the detection of
192  // cyclic references. At the startup of an application there is usually
193  // a lot of objects created that will live on through out the application
194  // so the average number of objects created per execution will be higher
195  // than the number of destroyed objects in the beginning, but afterwards
196  // it usually levels out to be more or less equal.
