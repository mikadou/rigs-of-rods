00033 #define PROCESS_SECTION_IN_ANY_MODULE(_KEYWORD_, _FIELD_, _FUNCTION_)                      \
00034 {                                                                                          \
00035     SetCurrentKeyword(_KEYWORD_);                                                          \
00036     auto module_itor = m_selected_modules.begin();                                         \
00037     auto module_end  = m_selected_modules.end();                                           \
00038     for (; module_itor != module_end; ++module_itor)                                       \
00039     {                                                                                      \
00040         auto* def = module_itor->get()->_FIELD_.get();                                     \
00041         if (def != nullptr)                                                                \
00042         {                                                                                  \
00043             try {                                                                          \
00044                 _FUNCTION_(*def);                                                          \
00045             }                                                                              \
00046             catch (Exception ex)                                                           \
00047             {                                                                              \
00048                 AddMessage(Message::TYPE_ERROR, ex.what());                                \
00049             }                                                                              \
00050             catch (...)                                                                    \
00051             {                                                                              \
00052                 AddMessage(Message::TYPE_ERROR, "An unknown exception has occured");       \
00053             }                                                                              \
00054             break;                                                                         \
00055         }                                                                                  \
00056     }                                                                                      \
00057     SetCurrentKeyword(RigDef::File::KEYWORD_INVALID);                                      \
00058 }
