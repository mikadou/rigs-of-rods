00204 void SlideNode::UpdateForces(float dt)
00205 {
00206     // only do calcs if we have a beam to slide on
00207     if ( !mSlidingBeam || mSlidingBeam->broken || getFlag( MASK_SLIDE_BROKEN ) )
00208     {
00209         return;
00210     }
00211     
00212     // the threshold changes when attaching a node to a beam, if it's more than
00213     // the initial amount then then we're still attaching.
00214     if ( mCurThreshold > mInitThreshold )
00215     {
00216         mCurThreshold -= (mAttachRate * dt);
00217     }
00218     
00219     Ogre::Vector3 perpForces = getCorrectiveForces();
00220     // perpendicular Forces are distributed according to the position along the Beam
00221     if ( perpForces.length() > mBreakForce ) setFlag( MASK_SLIDE_BROKEN );
00222     mSlidingNode->Forces += -perpForces;
00223     mSlidingBeam->p1->Forces += perpForces * (1 - mRatio);
00224     mSlidingBeam->p2->Forces += perpForces * mRatio;
00225 }

00277 void SlideNode::UpdatePosition()
00278 {
00279     // only do calcs if we have a beam to slide on
00280     if ( !mSlidingBeam || mSlidingBeam->broken )
00281     {
00282         mIdealPosition = mSlidingNode->AbsPosition;
00283         return;
00284     }
00285         
00286     // find which beam to use
00287     mSlidingRail = getClosestRail( mSlidingRail, mSlidingNode->AbsPosition );
00288     mSlidingBeam = mSlidingRail->curBeam;
00289     
00290     // Get vector for beam
00291     Ogre::Vector3 b = mSlidingBeam->p2->AbsPosition;
00292     b -= mSlidingBeam->p1->AbsPosition;
00293     
00294     // pre-compute normal
00295     const Ogre::Real bLen = b.length();
00296     
00297     // normalize b instead of using another variable
00298     b.normalise();
00299     
00300     // Get dot product along the b beam
00301     const Ogre::Real aDotBUnit = (mSlidingNode->AbsPosition - mSlidingBeam->p1->AbsPosition).dotProduct( b );
00302     
00303     // constrain Value between the two end points
00304     const Ogre::Real len = std::max( 0.0f, std::min( aDotBUnit, bLen) );
00305     mIdealPosition  = b;
00306     mIdealPosition *= len;
00307     mIdealPosition += mSlidingBeam->p1->AbsPosition;
00308     
00309     // calculate(cache) the ratio between the the two end points,
00310     // if bLen = 0.0f it means the beam is zero length so pick an end point
00311     mRatio = (bLen > 0.0f) ? len/bLen : 0.0f;
00312 }
