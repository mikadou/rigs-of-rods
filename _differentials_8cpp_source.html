00108 void Axle::calcOpenDiff( differential_data_t& diff_data )
00109 {
00110     /* Open differential calculations *************************
00111      * These calculation are surprisingly tricky
00112      * the power ratio is based on normalizing the speed of the
00113      * wheel. Normalizing is when the sum of all values equals
00114      * one. more detail provided below
00115      */
00116 
00117     // velocity at which open diff calculations are used 100%
00118     // this value is twice the speed of the actual threshold
00119     // the factor of 2 comes from finding the average speed and
00120     // is done so to remove unessecary calculations
00121     const Ogre::Real threshold_vel = 10.0f;
00122 
00123     // combined total velocity
00124     const Ogre::Real sum_of_vel =
00125         fabs(diff_data.speed[0]) + fabs(diff_data.speed[1]);
00126 
00127     // normalizing the wheel speeds the wheels to sputter at near 0
00128     // speeds, to over come this we gradually transition from even
00129     // power distribution to splitting power based on speed. The
00130     // Transition ratio defines how much of each formula we take from,
00131     // Until a threshold has been reached
00132     const Ogre::Real transition_ratio = (sum_of_vel/threshold_vel < 1) ?
00133             sum_of_vel/threshold_vel : 1;
00134 
00135     // normalize the wheel speed, at a speed of 0 power is split evenly
00136     const Ogre::Real power_ratio = (fabs(sum_of_vel) > 0.0) ?
00137             fabs(diff_data.speed[0])/sum_of_vel : 0.5;
00138 
00139     diff_data.out_torque[0] = diff_data.out_torque[1] = diff_data.in_torque;
00140 
00141     // Diff model taken from Torcs, ror needs to model reaction torque for this to work.
00142     //const Ogre::Real spider_acc = (diff_data.speed[0] - diff_data.speed[1])/diff_data.dt;
00143     //DrTq0 = DrTq*0.5f + spiderTq;
00144     //DrTq1 = DrTq*0.5f - spiderTq;
00145 
00146     // get the final ratio based on the speed of the wheels
00147     diff_data.out_torque[0] *= 2*(transition_ratio*power_ratio + (1-transition_ratio)*0.5f);
00148     diff_data.out_torque[1] *= 2*(transition_ratio*(1 - power_ratio) + (1-transition_ratio)*0.5f);
00149 
00150     diff_data.delta_rotation = 0.0f;
00151 }

00153 void Axle::calcLockedDiff( differential_data_t& diff_data )
00154 {
00155     /* Locked axle calculation ********************************
00156      * This is straight forward, two wheels are joined together
00157      * by a torsion spring. the torsion spring keeps the two
00158      * wheels in the same orientation as when they were first
00159      * locked.
00160      */
00161 
00162     // Torsion spring rate that holds axles together when locked
00163     // keep as variable for now since this value will be user configurable
00164     const Ogre::Real torsion_rate = 1000000.0f;
00165     const Ogre::Real torsion_damp = torsion_rate/100.0f;
00166     const Ogre::Real delta_speed = diff_data.speed[0] - diff_data.speed[1];
00167 
00168     diff_data.out_torque[0] = diff_data.out_torque[1] = diff_data.in_torque;
00169 
00170     // derive how far wheels traveled relative to each during the last time step
00171     diff_data.delta_rotation += (delta_speed)*diff_data.dt;
00172 
00173     // torque cause by axle shafts
00174     diff_data.out_torque[0] -= diff_data.delta_rotation * torsion_rate;
00175     // damping
00176     diff_data.out_torque[0] -= (delta_speed)*torsion_damp;
00177 
00178     // torque cause by axle shafts
00179     diff_data.out_torque[1] += diff_data.delta_rotation * torsion_rate;
00180     // damping
00181     diff_data.out_torque[1] -= -(delta_speed)*torsion_damp;
00182 }
