00097 void Lens::recalculateDepthOfField(Real& _nearDepth, Real& _focalDepth, Real& _farDepth) {
00098   // Set focalDepth to the current focalDistance
00099   _focalDepth = mFocalDistance;
00100 
00101   // Recalculate the hyperfocal length
00102   recalculateHyperfocalLength();
00103 
00104   // Calculate the numerator of the optics equations
00105   Real numerator = (mFocalDistance * (mHyperfocalLength - mFocalLength));
00106 
00107   Real nearClear = numerator / (mHyperfocalLength + mFocalDistance - (2.0 * mFocalLength));
00108 
00109   // Adjust the nearDepth relative to the aperture. This is an approximation.
00110   _nearDepth = std::min(nearClear - nearClear * mFStop, 0.0f);
00111 
00112   if (mFocalDistance < mHyperfocalLength)
00113   {
00114     // Calculate the far clear plane
00115     Real farClear = numerator / (mHyperfocalLength - mFocalDistance);
00116 
00117     // Adjust the farDepth relative to the aperture. This is an approximation.
00118     _farDepth = farClear + farClear * mFStop;
00119   } else
00120   {
00121       // Far depth of field should be infinite
00122       _farDepth = Math::POS_INFINITY;
00123   }
