00259         if (inst && inst->truck == truck && inst->sound_link_type == linkType && inst->sound_link_item_id == linkItemID)

00355 {
00356     //first, search template
00357     SoundScriptTemplate* templ = NULL;
00358 
00359     if (templates.find(templatename) == templates.end())
00360     {
00361         return NULL; // found no template with this name

00364     templ = templates[templatename];
00365 
00366     if (templ->trigger_source == SS_TRIG_NONE)

00373         || free_pitches[templ->trigger_source] >= MAX_INSTANCES_PER_GROUP && templ->pitch_source != SS_MOD_NONE )
00374     {
00375         LOG("SoundScriptManager: Reached MAX_INSTANCES_PER_GROUP limit (" + TOSTRING(MAX_INSTANCES_PER_GROUP) + ")");

00377     }
00378 
00379     SoundScriptInstance* inst = new SoundScriptInstance(truck, templ, sound_manager, templ->file_name+"-"+TOSTRING(truck)+"-"+TOSTRING(instance_counter), soundLinkType, soundLinkItemId);
00380     instance_counter++;
00381 
00382     // register to lookup tables
00383     trigs[templ->trigger_source + free_trigs[templ->trigger_source] * SS_MAX_TRIG] = inst;
00384     free_trigs[templ->trigger_source]++;
00385 
00386     if (templ->gain_source != SS_MOD_NONE)
00387     {
00388         gains[templ->gain_source + free_gains[templ->gain_source] * SS_MAX_MOD] = inst;

00406 void SoundScriptManager::parseScript(DataStreamPtr& stream, const String& groupName)
00407 {
00408     SoundScriptTemplate* sst = 0;
00409     String line = "";
00410     std::vector<String> vecparams;
00411 
00412     LOG("SoundScriptManager: Parsing script "+stream->getName());
00413 
00414     while(!stream->eof())
00415     {
00416         line = stream->getLine();
00417         // ignore comments & blanks
00418         if (!(line.length() == 0 || line.substr(0,2) == "//"))
00419         {
00420             if (sst == 0)
00421             {
00422                 // no current SoundScript
00423                 // so first valid data should be a SoundScript name
00424                 LOG("SoundScriptManager: creating template "+line);
00425                 sst = createTemplate(line, groupName, stream->getName());
00426                 if (!sst)
00427                 {
00428                     // there is a name collision for this Sound Script
00429                     LOG("SoundScriptManager: Error, this sound script is already defined: "+line);
00430                     skipToNextOpenBrace(stream);
00431                     skipToNextCloseBrace(stream);
00432                     continue;
00433                 }
00434                 // skip to and over next {
00435                 skipToNextOpenBrace(stream);
00436             } else
00437             {
00438                 // already in a ss
00439                 if (line == "}")
00440                 {
00441                     // finished ss
00442                     sst = 0;
00443                 } else
00444                 {
00445                     // attribute
00446                     // split params on space
00447                     Ogre::StringVector veclineparams = StringUtil::split(line, "\t ", 0);
00448 
00449                     if (!sst->setParameter(veclineparams))
00450                     {
00451                         LOG("Bad SoundScript attribute line: '" + line + "' in " + stream->getName());
00452                     }
00453                 }
00454             }
00455         }
00456     }
00457 }

00883 float SoundScriptInstance::pitchgain_cutoff(float sourcepitch, float targetpitch)
00884 {
00885     if (sourcepitch == 0.0f)
00886     {
00887         return 1.0f; // unpitchable
00888     }
00889 
00890     if (targetpitch>sourcepitch / PITCHDOWN_FADE_FACTOR)
00891     {
00892         return 1.0f; // pass
00893     }
00894 
00895     if (targetpitch<sourcepitch / PITCHDOWN_CUTOFF_FACTOR)
00896     {
00897         return 0.0f; // cutoff
00898     }
00899 
00900     // linear fading
00901     return (targetpitch - sourcepitch / PITCHDOWN_CUTOFF_FACTOR) / (sourcepitch / PITCHDOWN_FADE_FACTOR - sourcepitch / PITCHDOWN_CUTOFF_FACTOR);
