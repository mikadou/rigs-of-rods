38 void AllWheelsAggregateData::AddWheel(LandVehicleWheel* wheel)
39 {
40  LandVehicleWheel::Type verified_type = LandVehicleWheel::TYPE_INVALID;
41  switch (wheel->GetType())
42  {
43  case LandVehicleWheel::TYPE_MESHWHEEL_2:
44  case LandVehicleWheel::TYPE_FLEXBODYWHEEL:
45  verified_type = wheel->GetType();
46  break;
47  default:
48  assert(false && "Unsupported wheel type");
49  break;
50  }
51  if (verified_type == LandVehicleWheel::TYPE_INVALID)
52  {
53  return;
54  }
55 
56  // Process wheel
57  this->num_elements++;
58  if (verified_type == LandVehicleWheel::TYPE_MESHWHEEL_2)
59  {
60  MeshWheel2* mw = static_cast<MeshWheel2*>(wheel);
61  this->meshwheels2_data.AddWheel(mw);
62  }
63  else if (verified_type == LandVehicleWheel::TYPE_FLEXBODYWHEEL)
64  {
65  FlexBodyWheel* flexwheel = static_cast<FlexBodyWheel*>(wheel);
66  this->flexbodywheels_data.AddWheel(flexwheel);
67  }
68 }
