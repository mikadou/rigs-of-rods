111 void ContentManager::AddResourcePack(ResourcePack const & resource_pack)
112 {
113  std::stringstream log_msg;
114  if (BITMASK_64_IS_1(m_loaded_resource_packs, resource_pack.mask)) // Already loaded?
115  {
116  log_msg << "[RoR|ContentManager] Resource pack \"" << resource_pack.name << "\" already loaded.";
117  LOG(log_msg.str());
118  return;
119  }
120  log_msg << "[RoR|ContentManager] Loading resource pack \"" << resource_pack.name << "\" from group \"" << resource_pack.resource_group_name << "\"";
121  Ogre::String resources_dir = SSETTING("Resources Path", "resources" + PlatformUtils::DIRECTORY_SEPARATOR);
122  Ogre::String zip_path = resources_dir + resource_pack.name + Ogre::String(".zip");
123  if (PlatformUtils::FileExists(zip_path))
124  {
125  log_msg << " (ZIP archive)";
126  LOG(log_msg.str());
127  ResourceGroupManager::getSingleton().addResourceLocation(zip_path, "Zip", resource_pack.resource_group_name);
128  BITMASK_64_SET_1(m_loaded_resource_packs, resource_pack.mask);
129  }
130  else
131  {
132  Ogre::String dir_path = resources_dir + resource_pack.name;
133  if (PlatformUtils::FolderExists(dir_path))
134  {
135  log_msg << " (directory)";
136  LOG(log_msg.str());
137  ResourceGroupManager::getSingleton().addResourceLocation(dir_path, "FileSystem", resource_pack.resource_group_name);
138  BITMASK_64_SET_1(m_loaded_resource_packs, resource_pack.mask);
139  }
140  else
141  {
142  log_msg << " failed, data not found.";
143  throw std::runtime_error(log_msg.str());
144  }
145  }
146 }

328 void ContentManager::exploreFolders(Ogre::String rg)
329 {
330  String dirsep="/";
331 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
332  dirsep="\\";
333 #endif
334  ResourceGroupManager& rgm = ResourceGroupManager::getSingleton();
335 
336  FileInfoListPtr files= rgm.findResourceFileInfo(rg, "*", true); // searching for dirs
337  FileInfoList::iterator iterFiles = files->begin();
338  for (; iterFiles!= files->end(); ++iterFiles)
339  {
340  if (!iterFiles->archive) continue;
341  if (iterFiles->filename==String(".svn")) continue;
342  // trying to get the full path
343  String fullpath = iterFiles->archive->getName() + dirsep;
344  rgm.addResourceLocation(fullpath+iterFiles->filename, "FileSystem", rg);
345  }
346  LOG("initialiseResourceGroups: "+rg);
347  try
348  {
349  if (BSETTING("Background Loading", false))
350  ResourceBackgroundQueue::getSingleton().initialiseResourceGroup(rg);
351  else
352  ResourceGroupManager::getSingleton().initialiseResourceGroup(rg);
353  } catch(Ogre::Exception& e)
