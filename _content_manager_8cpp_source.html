00112 {
00113     std::stringstream log_msg;
00114     if (BITMASK_64_IS_1(m_loaded_resource_packs, resource_pack.mask)) // Already loaded?
00115     {
00116         log_msg << "[RoR|ContentManager] Resource pack \"" << resource_pack.name << "\" already loaded.";
00117         LOG(log_msg.str());

00121     Ogre::String resources_dir = SSETTING("Resources Path", "resources" + PlatformUtils::DIRECTORY_SEPARATOR);
00122     Ogre::String zip_path = resources_dir + resource_pack.name + Ogre::String(".zip");
00123     if (PlatformUtils::FileExists(zip_path))
00124     {
00125         log_msg << " (ZIP archive)";
00126         LOG(log_msg.str());
00127         ResourceGroupManager::getSingleton().addResourceLocation(zip_path, "Zip", resource_pack.resource_group_name);
00128         BITMASK_64_SET_1(m_loaded_resource_packs, resource_pack.mask);
00129     }
00130     else
00131     {
00132         Ogre::String dir_path = resources_dir + resource_pack.name;
00133         if (PlatformUtils::FolderExists(dir_path))
00134         {
00135             log_msg << " (directory)";
00136             LOG(log_msg.str());
00137             ResourceGroupManager::getSingleton().addResourceLocation(dir_path, "FileSystem", resource_pack.resource_group_name);
00138             BITMASK_64_SET_1(m_loaded_resource_packs, resource_pack.mask);
00139         }
00140         else
00141         {
00142             log_msg << " failed, data not found.";
00143             throw std::runtime_error(log_msg.str());
00144         }
00145     }
