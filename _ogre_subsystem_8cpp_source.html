57 bool OgreSubsystem::Configure()
58 {
59  // Show the configuration dialog and initialise the system
60  // You can skip this and use root.restoreConfig() to load configuration
61  // settings if you were sure there are valid ones saved in ogre.cfg
62  bool use_ogre_config = BSETTING("USE_OGRE_CONFIG", false);
63 
64  bool ok = false;
65  if (use_ogre_config)
66  {
67  ok = m_ogre_root->showConfigDialog();
68  }
69  else
70  {
71  ok = m_ogre_root->restoreConfig();
72  }
73  if (ok)
74  {
75  // If returned true, user clicked OK so initialise
76  // Here we choose to let the system create a default rendering window by passing 'true'
77  m_render_window = m_ogre_root->initialise(true, "Rigs of Rods version " + Ogre::String(ROR_VERSION_STRING));
78 
79  // set window icon correctly
80  fixRenderWindowIcon(m_render_window);
81 
82  return true;
83  }
84  else
85  {
86  ErrorUtils::ShowError(_L("Configuration error"), _L("Run the RoRconfig program first."));
87  exit(1);
88  }
89  return true;
90 }

93 bool OgreSubsystem::LoadOgrePlugins(Ogre::String const & pluginsfile)
94 {
95  Ogre::StringVector pluginList;
96  Ogre::String pluginDir;
97  Ogre::ConfigFile cfg;
98 
99  try
100  {
101  cfg.load( pluginsfile );
102  }
103  catch (Ogre::Exception)
104  {
105  Ogre::LogManager::getSingleton().logMessage(pluginsfile + " not found, automatic plugin loading disabled.");
106  return false;
107  }
108 
109  pluginDir = cfg.getSetting("PluginFolder"); // Ignored on Mac OS X, uses Resources/ directory
110  pluginList = cfg.getMultiSetting("Plugin");
111 
112 #if OGRE_PLATFORM != OGRE_PLATFORM_APPLE && OGRE_PLATFORM != OGRE_PLATFORM_IPHONE
113  if (pluginDir.empty())
114  {
115  // User didn't specify plugins folder, try current one
116  pluginDir = ".";
117  }
118 #endif
119 
120  char last_char = pluginDir[pluginDir.length()-1];
121  if (last_char != '/' && last_char != '\\')
122  {
123 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
124  pluginDir += "\\";
125 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX
126  pluginDir += "/";
127 #endif
128  }
129 
130  for ( Ogre::StringVector::iterator it = pluginList.begin(); it != pluginList.end(); ++it )
131  {
132  Ogre::String pluginFilename = pluginDir + (*it);
133  try
134  {
135  m_ogre_root->loadPlugin(pluginFilename);
136  }
137  catch(Ogre::Exception &e)
138  {
139  LOG("failed to load plugin: " + pluginFilename + ": " + e.getFullDescription());
140  }
141  }
142  return true;
143 }

145 bool OgreSubsystem::StartOgre(Ogre::String const & name, Ogre::String const & hwnd, Ogre::String const & mainhwnd)
146 {
147  m_name = name;
148  m_hwnd = hwnd;
149  m_main_hwnd = mainhwnd;
150 
151  try
152  {
153  SETTINGS.loadSettings(SSETTING("Config Root", "")+"RoR.cfg");
154  }
155  catch(Ogre::Exception& e)
156  {
157  Ogre::String url = "http://wiki.rigsofrods.com/index.php?title=Error_" + TOSTRING(e.getNumber())+"#"+e.getSource();
158  ErrorUtils::ShowOgreWebError(_L("A fatal exception has occured!"), ANSI_TO_UTF(e.getFullDescription()), ANSI_TO_UTF(url));
159  ErrorUtils::ShowStoredOgreWebErrors();
160  exit(1);
161  }
162 
163  Ogre::String logFilename = SSETTING("Log Path", "") + name + Ogre::String(".log");
164  Ogre::String pluginsConfig = SSETTING("plugins.cfg", "plugins.cfg");
165  Ogre::String ogreConfig = SSETTING("ogre.cfg", "ogre.cfg");
166  m_ogre_root = new Ogre::Root("", ogreConfig, logFilename);
167 
168  // load plugins manually
169  LoadOgrePlugins(pluginsConfig);
170 
171  // configure RoR
172  Configure();
173 
174  m_viewport = m_render_window->addViewport(nullptr);
175 
176  m_viewport->setBackgroundColour(Ogre::ColourValue(0.5f, 0.5f, 0.5f, 1.0f));
177 
178  m_viewport->setCamera(nullptr);
179  m_viewport->setBackgroundColour(Ogre::ColourValue::Black);
180 
181  Ogre::TextureManager::getSingleton().setDefaultNumMipmaps(5);
182 
183  m_timer = new Ogre::Timer();
184  m_timer->reset();
185 
186  m_render_window->setActive(true);
187 
188  return true;
189 }
