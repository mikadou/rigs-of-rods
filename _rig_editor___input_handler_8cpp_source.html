160 bool InputHandler::keyPressed( const OIS::KeyEvent &arg )
161 {
162  if (RoR::Application::GetGuiManager()->keyPressed(arg))
163  {
164  return true;
165  }
166 
167  // HANDLE EVENTS \\
168 
169  const Event* event_ptr = m_event_key_mappings[arg.key];
170  if (event_ptr != nullptr)
171  {
172  m_events_fired[event_ptr->index] = true;
173  }
174 
175  // HANDLE MODES \\
176 
177  const Mode* mode_ptr = m_mode_key_mappings[arg.key];
178  if (mode_ptr != nullptr && mode_ptr->key_enters)
179  {
180  m_active_modes[mode_ptr->index] = true;
181  m_modes_entered[mode_ptr->index] = true;
182  }
183 
184  return true;
185 }
