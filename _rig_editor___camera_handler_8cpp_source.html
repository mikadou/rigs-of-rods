00242

00244             {
00245                 UpdateOrthoZoom();

00247             m_inverse_view_matrix_dirty = true;
00248             return true;
00249         }
00250     }
00251     else if (mStyle == CS_FREELOOK)
00252     {
00253         mCamera->yaw(Ogre::Degree(-x_rel * 0.15f));

00293     Vector2int & out_screen_position
00294 )
00295 {
00296     // Transform position: world space -> view space
00297     Ogre::Vector3 view_space_pos = mCamera->getViewMatrix(true) * _world_position;
00298 
00299     // Check if the position is in front of the camera
00300     if (view_space_pos.z < 0.f)
00301     {
00302         // Transform: view space -> clip space [-1, 1]
00303         Ogre::Vector3 clip_space_pos = mCamera->getProjectionMatrix() * view_space_pos;
00304 
00305         // Transform: clip space [-1, 1] -> to [0, 1]
00306         float screen_space_pos_x = (clip_space_pos.x / 2.f) + 0.5f;
00307         float screen_space_pos_y = 1 - ((clip_space_pos.y / 2.f) + 0.5f);

00309         // Transform: clip space -> absolute pixel coordinates
00310         Ogre::Viewport* viewport = mCamera->getViewport();

00312         out_screen_position.y = static_cast<int>(screen_space_pos_y * viewport->getActualHeight());

00314         return true;

00316     else

00318         return false;

00322 Ogre::Vector3 CameraHandler::ConvertScreenToWorldPosition(Vector2int const & screen_pos, Ogre::Vector3 const & pivot)
00323 {
00324     if (m_inverse_projection_matrix_dirty)
00325     {
00326         m_inverse_projection_matrix = mCamera->getProjectionMatrix().inverse();
00327         m_inverse_projection_matrix_dirty = false;
00328     }
00329 
00330     if (m_inverse_view_matrix_dirty)
00331     {
00332         m_inverse_view_matrix = mCamera->getViewMatrix().inverse();
00333         m_inverse_view_matrix_dirty = false;
00334     }
00335 
00336     // NOTE:
00337     // Screen pixel coordinates depend on your resolution:
00338     //     * start at top left corner of the screen (X:0, Y:0)
00339     //     * end in bottom right (for example X:1024, Y:768)
00340     // Clip space coordinates are always in range [-1, 1]:
00341     //     * start in the middle of the screen (X:0, Y:0, Z:0)
00342     //     * Bottom left corner of screen is (X:-1, Y:-1, Z:0)
00343     //     * Top right corner of screen is (X:1, Y:1, Z:0)
00344 
00345     // Transform: pixel coordinates -> clip space
00346     Ogre::Viewport* viewport = mCamera->getViewport();
00347     Ogre::Vector3 output(
00348             (((static_cast<float>(screen_pos.x) / static_cast<float>(viewport->getActualWidth()) ) *  2.f) - 1.f),
00349             (((static_cast<float>(screen_pos.y) / static_cast<float>(viewport->getActualHeight())) * -2.f) + 1.f),
00350             0.f
00351         );
00352 
00353     // Set point's depth, defined by pivot
00354     Ogre::Vector3 clip_space_pivot = mCamera->getProjectionMatrix() * mCamera->getViewMatrix() * pivot; // Transform pivot: world space -> clip space
00355     output.z = clip_space_pivot.z;
00356 
00357     // Transform clip space -> view space
00358     output = m_inverse_projection_matrix * output;
00359 
00360     // Transform view space -> world space
00361     return m_inverse_view_matrix * output;
00362 }
