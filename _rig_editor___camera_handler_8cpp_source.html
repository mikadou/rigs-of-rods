212 bool CameraHandler::InjectMouseMove(bool do_orbit, int x_rel, int y_rel, int wheel_rel)
213 {
214  if (mStyle == CS_ORBIT)
215  {
216  Ogre::Real dist = GetCameraTargetDistance();
217 
218  if (do_orbit) // yaw around the target, and pitch locally
219  {
220  mCamera->setPosition(mTarget->_getDerivedPosition());
221 
222  mCamera->yaw(Ogre::Degree(-x_rel * 0.25f));
223  mCamera->pitch(Ogre::Degree(-y_rel * 0.25f));
224 
225  mCamera->moveRelative(Ogre::Vector3(0, 0, dist));
226  m_inverse_view_matrix_dirty = true;
227  return true;
228 
229  // don't let the camera go over the top or around the bottom of the target
230  }
231  else if (mZooming) // (mouse-move zooming) move the camera toward or away from the target
232  {
233  // the further the camera is, the faster it moves
234  mCamera->moveRelative(Ogre::Vector3(0, 0, y_rel * 0.004f * dist));
235  m_inverse_view_matrix_dirty = true;
236  return true;
237  }
238  else if (wheel_rel != 0) // move the camera toward or away from the target
239  {
240  // the further the camera is, the faster it moves
241  mCamera->moveRelative(Ogre::Vector3(0, 0, -wheel_rel * 0.0008f * dist));
242 
243  if (mCamera->getProjectionType() == Ogre::PT_ORTHOGRAPHIC)
244  {
245  UpdateOrthoZoom();
246  }
247  m_inverse_view_matrix_dirty = true;
248  return true;
249  }
250  }
251  else if (mStyle == CS_FREELOOK)
252  {
253  mCamera->yaw(Ogre::Degree(-x_rel * 0.15f));
254  mCamera->pitch(Ogre::Degree(-y_rel * 0.15f));
255  }
256 
257  return false;
258 }

291 bool CameraHandler::ConvertWorldToScreenPosition(
292  Ogre::Vector3 const & _world_position,
293  Vector2int & out_screen_position
294 )
295 {
296  // Transform position: world space -> view space
297  Ogre::Vector3 view_space_pos = mCamera->getViewMatrix(true) * _world_position;
298 
299  // Check if the position is in front of the camera
300  if (view_space_pos.z < 0.f)
301  {
302  // Transform: view space -> clip space [-1, 1]
303  Ogre::Vector3 clip_space_pos = mCamera->getProjectionMatrix() * view_space_pos;
304 
305  // Transform: clip space [-1, 1] -> to [0, 1]
306  float screen_space_pos_x = (clip_space_pos.x / 2.f) + 0.5f;
307  float screen_space_pos_y = 1 - ((clip_space_pos.y / 2.f) + 0.5f);
308 
309  // Transform: clip space -> absolute pixel coordinates
310  Ogre::Viewport* viewport = mCamera->getViewport();
311  out_screen_position.x = static_cast<int>(screen_space_pos_x * viewport->getActualWidth());
312  out_screen_position.y = static_cast<int>(screen_space_pos_y * viewport->getActualHeight());
313 
314  return true;
315  }
316  else
317  {
318  return false;
319  }
320 }

322 Ogre::Vector3 CameraHandler::ConvertScreenToWorldPosition(Vector2int const & screen_pos, Ogre::Vector3 const & pivot)
323 {
324  if (m_inverse_projection_matrix_dirty)
325  {
326  m_inverse_projection_matrix = mCamera->getProjectionMatrix().inverse();
327  m_inverse_projection_matrix_dirty = false;
328  }
329 
330  if (m_inverse_view_matrix_dirty)
331  {
332  m_inverse_view_matrix = mCamera->getViewMatrix().inverse();
333  m_inverse_view_matrix_dirty = false;
334  }
335 
336  // NOTE:
337  // Screen pixel coordinates depend on your resolution:
338  // * start at top left corner of the screen (X:0, Y:0)
339  // * end in bottom right (for example X:1024, Y:768)
340  // Clip space coordinates are always in range [-1, 1]:
341  // * start in the middle of the screen (X:0, Y:0, Z:0)
342  // * Bottom left corner of screen is (X:-1, Y:-1, Z:0)
343  // * Top right corner of screen is (X:1, Y:1, Z:0)
344 
345  // Transform: pixel coordinates -> clip space
346  Ogre::Viewport* viewport = mCamera->getViewport();
347  Ogre::Vector3 output(
348  (((static_cast<float>(screen_pos.x) / static_cast<float>(viewport->getActualWidth()) ) * 2.f) - 1.f),
349  (((static_cast<float>(screen_pos.y) / static_cast<float>(viewport->getActualHeight())) * -2.f) + 1.f),
350  0.f
351  );
352 
353  // Set point's depth, defined by pivot
354  Ogre::Vector3 clip_space_pivot = mCamera->getProjectionMatrix() * mCamera->getViewMatrix() * pivot; // Transform pivot: world space -> clip space
355  output.z = clip_space_pivot.z;
356 
357  // Transform clip space -> view space
358  output = m_inverse_projection_matrix * output;
359 
360  // Transform view space -> world space
361  return m_inverse_view_matrix * output;
362 }
