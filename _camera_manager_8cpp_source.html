76 bool CameraManager::update(float dt) // Called every frame
77 {
78  if (RoR::Application::GetGuiManager()->GetPauseMenuVisible()) return true; //Stop everything when pause menu is visible
79 
80  if ( dt == 0 ) return false;
81 
82  mTransScale = mTransSpeed * dt;
83  mRotScale = mRotateSpeed * dt;
84 
85  ctx.mCurrTruck = BeamFactory::getSingleton().getCurrentTruck();
86  ctx.mDt = dt;
87  ctx.mRotScale = Degree(mRotScale);
88  ctx.mTransScale = mTransScale;
89  ctx.fovInternal = Degree(FSETTING("FOV Internal", 75.0f)); // FIXME: No reason for this to be done per frame.
90  ctx.fovExternal = Degree(FSETTING("FOV External", 60.0f)); // FIXME: No reason for this to be done per frame.
91 
92  if ( currentBehaviorID < CAMERA_BEHAVIOR_END && RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_CHANGE) )
93  {
94  switchToNextBehavior(false);
95  }
96 
97  if ( RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_FREE_MODE_FIX) )
98  {
99  toggleBehavior(CAMERA_BEHAVIOR_FIXED);
100  }
101 
102  if ( RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_FREE_MODE) )
103  {
104  toggleBehavior(CAMERA_BEHAVIOR_FREE);
105  }
106 
107  if ( currentBehavior )
108  {
109  currentBehavior->update(ctx);
110  } else
111  {
112  switchBehavior(CAMERA_BEHAVIOR_CHARACTER);
113  }
114 
115  return true;
116 }

127 void CameraManager::switchBehavior(int newBehaviorID, bool reset)
128 {
129  if (newBehaviorID == currentBehaviorID)
130  {
131  return;
132  }
133 
134  if ( globalBehaviors.find(newBehaviorID) == globalBehaviors.end() )
135  {
136  return;
137  }
138 
139  // deactivate old
140  if ( currentBehavior )
141  {
142  currentBehavior->deactivate(ctx);
143  }
144 
145  // set new
146  currentBehavior = globalBehaviors[newBehaviorID];
147  currentBehaviorID = newBehaviorID;
148 
149  // activate new
150  if (ctx.mCurrTruck != nullptr)
151  {
152  ctx.mCurrTruck->GetCameraContext()->behavior = RoR::PerVehicleCameraContext::CAMERA_BEHAVIOR_EXTERNAL;
153  }
154  currentBehavior->activate(ctx, reset);
155 }

185 void CameraManager::toggleBehavior(int behavior)
186 {
187  static std::stack<int> precedingBehaviors;
188 
189  if ( behavior != currentBehaviorID && (precedingBehaviors.empty() || precedingBehaviors.top() != behavior))
190  {
191  if ( currentBehaviorID >= 0 )
192  {
193  precedingBehaviors.push(currentBehaviorID);
194  }
195  switchBehavior(behavior);
196  } else if ( !precedingBehaviors.empty() )
197  {
198  switchBehavior(precedingBehaviors.top(), false);
199  precedingBehaviors.pop();
200  } else
201  {
202  switchToNextBehavior();
203  }
204 }

275 void CameraManager::NotifyVehicleChanged(Beam* old_vehicle, Beam* new_vehicle)
276 {
277  // Getting out of vehicle
278  if (new_vehicle == nullptr)
279  {
280  ctx.mCurrTruck = nullptr;
281  this->switchBehavior(CAMERA_BEHAVIOR_CHARACTER);
282  return;
283  }
284 
285  // Getting in vehicle
286  switch (new_vehicle->GetCameraContext()->behavior)
287  {
288  case RoR::PerVehicleCameraContext::CAMERA_BEHAVIOR_VEHICLE_3rdPERSON:
289  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE, true, old_vehicle, new_vehicle);
290  break;
291 
292  case RoR::PerVehicleCameraContext::CAMERA_BEHAVIOR_VEHICLE_SPLINE:
293  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE_SPLINE, true, old_vehicle, new_vehicle);
294  break;
295 
296  case RoR::PerVehicleCameraContext::CAMERA_BEHAVIOR_VEHICLE_CINECAM:
297  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE_CINECAM, true, old_vehicle, new_vehicle);
298  break;
299 
300  default:
301  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE, true, old_vehicle, new_vehicle);
302  }
