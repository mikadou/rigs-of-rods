57 void SequentialImporter::GenerateNodesForWheel(File::Keyword generated_from, int num_rays, bool has_rigidity_node)
58 {
59  // Old parser logic:
60  // Note: Axis nodes are A/B, where B has bigger Z coordinate.
61  //
62  // Section | Function() | Generated nodes
63  // -------------------------------------------------------------------------------------------
64  // wheels | addWheel() | Tyre nodes: A, B, A, B... (num_rays*2)
65  // wheels2 | addWheel2() | Rim nodes: A, B, A, B... (num_rays*2), Tyre nodes: A, B, A, B... (num_rays*2)
66  // meshwheels | addWheel() | Tyre nodes: A, B, A, B... (num_rays*2)
67  // meshwheels2 | addWheel() | Tyre nodes: A, B, A, B... (num_rays*2)
68  // flexbodywheels | addWheel3() | Rim nodes: A, B, A, B... (num_rays*2), Tyre nodes: A, B, A, B... (num_rays*2)
69 
70  NodeMapEntry::OriginDetail detail;
71  if (generated_from == File::KEYWORD_FLEXBODYWHEELS || generated_from == File::KEYWORD_WHEELS2)
72  {
73  // Rim nodes
74  for (int i = 0; i < num_rays*2; ++i)
75  {
76  detail = (i % 2 == 0) ? NodeMapEntry::DETAIL_WHEEL_RIM_A : NodeMapEntry::DETAIL_WHEEL_RIM_B;
77  this->AddGeneratedNode(generated_from, detail);
78  }
79  }
80  // Tyre nodes
81  for (int i = 0; i < num_rays*2; ++i)
82  {
83  detail = (i % 2 == 0) ? NodeMapEntry::DETAIL_WHEEL_TYRE_A : NodeMapEntry::DETAIL_WHEEL_TYRE_B;
84  this->AddGeneratedNode(generated_from, detail);
85  }
86 }

127 unsigned SequentialImporter::GetNodeArrayOffset(File::Keyword keyword)
128 {
129  // Reverse order of elements in nodes-array
130  // "break" keyword is omitted - control falls through!
131  unsigned out_offset = 0;
132  switch (keyword)
133  {
134  case File::KEYWORD_FLEXBODYWHEELS: out_offset += m_num_meshwheels2_nodes; // NO break!
135  case File::KEYWORD_MESHWHEELS2: out_offset += m_num_meshwheels_nodes; // NO break!
136  case File::KEYWORD_MESHWHEELS: out_offset += m_num_wheels2_nodes; // NO break!
137  case File::KEYWORD_WHEELS2: out_offset += m_num_wheels_nodes; // NO break!
138  case File::KEYWORD_WHEELS: out_offset += m_num_cinecam_nodes; // NO break!
139  case File::KEYWORD_CINECAM: out_offset += m_num_named_nodes; // NO break!
140  case File::KEYWORD_NODES2: out_offset += m_num_numbered_nodes; // NO break!
141  case File::KEYWORD_NODES: break; // Starts at 0
142  default: break;
143  }
144  return out_offset;
145 }

147 Node::Ref SequentialImporter::ResolveNodeByIndex(unsigned int index_in, unsigned int def_line_number)
148 {
149  if (index_in >= m_all_nodes.size())
150  {
151  std::stringstream msg;
152  msg << "Cannot resolve node by index [" << index_in << "], node is not defined, highest available number is: " << m_all_nodes.size() - 1;
153  this->AddMessage(Message::TYPE_ERROR, msg.str());
154  return Node::Ref(); // Invalid reference
155  }
156  NodeMapEntry node_entry = m_all_nodes[index_in];
157  unsigned index_out = this->GetNodeArrayOffset(node_entry.origin_keyword) + node_entry.node_sub_index;
158  if (index_out != index_in)
159  {
160  std::stringstream msg;
161  msg << "Node resolved by index.\n\tSource: [" << index_in << "]\n\tResult: [" << index_out << "]";
162  if (node_entry.origin_keyword != File::KEYWORD_NODES && node_entry.origin_keyword != File::KEYWORD_NODES2)
163  {
164  msg << " (generated by: " << File::KeywordToString(node_entry.origin_keyword)
165  << ", sub-index: " << node_entry.node_sub_index << ")";
166  }
167  this->AddMessage(Message::TYPE_INFO, msg.str());
168  }
169  else
170  {
171  ++m_num_resolved_to_self; // The happy statistic -> nodes which resolved to the same index.
172  }
173  return Node::Ref(TOSTRING(index_out), index_out, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, def_line_number);
174 }

176 Node::Ref SequentialImporter::ResolveNode(Node::Ref const & noderef_in)
177 {
178  ++m_total_resolved;
179  if (!noderef_in.GetImportState_IsValid())
180  {
181  return noderef_in;
182  }
183 
184  // Work like legacy function SerializedRig::parse_node_number()
185  // If node may be named, try resolving it as named first
186  // If that fails, try numbered
187 
188  if (noderef_in.GetImportState_MustCheckNamedFirst())
189  {
190  auto result = m_named_nodes.find(noderef_in.Str());
191  if (result != m_named_nodes.end())
192  {
193  return Node::Ref(noderef_in.Str(), 0, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NAMED, noderef_in.GetLineNumber());
194  }
195  }
196  if (noderef_in.Num() >= m_all_nodes.size())
197  {
198  // Return exactly what SerializedRig::parse_node_number() would return on this error, for compatibility.
199  // This definitely isn't valid, but it behaves exactly as RoR 0.38, so it's perfectly IMPORT_VALID! :D
200  Node::Ref out_ref("0", 0, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, noderef_in.GetLineNumber());
201  std::stringstream msg;
202  msg << "Cannot resolve " << noderef_in.ToString() << " - not a named node, and index is not defined (highest is: "
203  << m_all_nodes.size() - 1 << "). For backwards compatibility, converting to: " << out_ref.ToString();
204  this->AddMessage(Message::TYPE_ERROR, msg.str());
205  return out_ref; // Invalid
206  }
207  auto entry = m_all_nodes[noderef_in.Num()];
208  if (entry.node_id.IsTypeNamed())
209  {
210  Node::Ref out_ref(entry.node_id.Str(), 0, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NAMED, noderef_in.GetLineNumber());
211  std::stringstream msg;
212  msg << "Node resolved\n\tSource: " << noderef_in.ToString() << "\n\tResult: " << out_ref.ToString();
213  this->AddMessage(Message::TYPE_INFO, msg.str());
214  return out_ref;
215  }
216  else if (entry.node_id.IsTypeNumbered())
217  {
218  unsigned out_index = this->GetNodeArrayOffset(entry.origin_keyword) + entry.node_sub_index;
219  Node::Ref out_ref(TOSTRING(out_index), out_index, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, noderef_in.GetLineNumber());
220  std::stringstream msg;
221  msg << "Node resolved\n\tSource: " << noderef_in.ToString() << "\n\tResult: " << out_ref.ToString()
222  << "\n\tOrigin: " << RigDef::File::KeywordToString(entry.origin_keyword) << " SubIndex: " << entry.node_sub_index;
223  this->AddMessage(Message::TYPE_INFO, msg.str());
224  return out_ref;
225  }
226  else
227  {
228  std::stringstream msg;
229  msg << "Cannot resolve " << noderef_in.ToString() << " - found node is not valid";
230  this->AddMessage(Message::TYPE_ERROR, msg.str());
231  return Node::Ref(); // Invalid
232  }

235 void SequentialImporter::ResolveFlexbodyForset(std::vector<Node::Range>& in_ranges, std::vector<Node::Ref>& out_nodes)
236 {
237  auto range_itor = in_ranges.begin();
238  auto range_end = in_ranges.end();
239  for ( ; range_itor != range_end; ++range_itor)
240  {
241  Node::Range& range = *range_itor;
242  if (!range.IsRange())
243  {
244  if (!range.start.GetImportState_IsValid())
245  {
246  std::stringstream msg;
247  msg << "ResolveFlexbodyForset(): Skipping node because it's marked INVALID:" << range.start.ToString();
248  this->AddMessage(Message::TYPE_ERROR, msg.str());
249  }
250  else
251  {
252  Node::Ref node_ref = this->ResolveNodeByIndex(range.start.Num(), range.start.GetLineNumber()); // Forset nodes are numbered-only
253  // Invalid nodes are simply thrown away, for backwards compatibility
254  // (FlexBody loops through existing nodes first, and then evaluates if they are in the SET)
255  if (node_ref.IsValidAnyState())
256  {
257  out_nodes.push_back(node_ref);
258  }
259  else
260  {
261  std::stringstream msg;
262  msg << "ResolveFlexbodyForset(): Stand-alone node [" << range.start.ToString() << "] resolved invalid, removing from FORSET";
263  this->AddMessage(Message::TYPE_WARNING, msg.str());
264  }
265  }
266  }
267  else // It's a range
268  {
269  if (!range.start.GetImportState_IsValid() || !range.end.GetImportState_IsValid())
270  {
271  std::stringstream msg;
272  msg << "ResolveFlexbodyForset(): Skipping range because some nodes are marked INVALID, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
273  this->AddMessage(Message::TYPE_ERROR, msg.str());
274  }
275  else if (range.start.GetImportState_IsResolvedNamed() || range.end.GetImportState_IsResolvedNamed())
276  {
277  std::stringstream msg;
278  msg << "ResolveFlexbodyForset(): Some nodes in range are already resolved as named, unable to process, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
279  this->AddMessage(Message::TYPE_ERROR, msg.str());
280  }
281  else
282  {
283  unsigned int end_index = range.end.Num();
284  unsigned int line_num = range.start.GetLineNumber();
285  for (unsigned int i = range.start.Num(); i <= end_index; ++i)
286  {
287  Node::Ref node_ref = this->ResolveNodeByIndex(i, line_num); // Forset nodes are numbered-only
288  // Invalid nodes are simply thrown away, for backwards compatibility
289  // (FlexBody loops through existing nodes first, and then evaluates if they are in the SET -> invalid nodes are silently ignored)
290  if (node_ref.IsValidAnyState())
291  {
292  out_nodes.push_back(node_ref);
293  }
294  else
295  {
296  std::stringstream msg;
297  msg << "ResolveFlexbodyForset(): Node ["<<i<<"] from range [" << range.start.ToString() << " - " << range.end.ToString() << "] resolved invalid, removing from FORSET";
298  this->AddMessage(Message::TYPE_WARNING, msg.str());
299  }
300  }
301  }
302  }
303  }

306 void SequentialImporter::ResolveNodeRanges(std::vector<Node::Range>& ranges)
307 {
308  auto in_ranges = ranges; // Copy vector
309  ranges.clear();
310 
311  auto range_itor = in_ranges.begin();
312  auto range_end = in_ranges.end();
313  for ( ; range_itor != range_end; ++range_itor)
314  {
315  Node::Range& range = *range_itor;
316  if (!range.IsRange())
317  {
318  ranges.push_back(Node::Range(this->ResolveNode(range.start)));
319  }
320  else if (!range.start.GetImportState_IsValid() || !range.end.GetImportState_IsValid())
321  {
322  std::stringstream msg;
323  msg << "Some nodes in range are invalid, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
324  this->AddMessage(Message::TYPE_ERROR, msg.str());
325  }
326  else if (range.start.GetImportState_IsResolvedNamed() || range.end.GetImportState_IsResolvedNamed())
327  {
328  std::stringstream msg;
329  msg << "Some nodes in range are already resolved as named, unable to process, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
330  this->AddMessage(Message::TYPE_ERROR, msg.str());
331  }
332  else
333  {
334  unsigned int end_index = range.end.Num();
335  for (unsigned int i = range.start.Num(); i < end_index; ++i)
336  {
337  ranges.push_back(Node::Range(this->ResolveNodeByIndex(i, range.start.GetLineNumber())));
338  }
339  }
340  }
