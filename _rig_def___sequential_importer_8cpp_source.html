00062     // Section        | Function()    | Generated nodes
00063     // -------------------------------------------------------------------------------------------
00064     // wheels         | addWheel()    | Tyre nodes: A, B, A, B... (num_rays*2) 
00065     // wheels2        | addWheel2()   | Rim nodes: A, B, A, B... (num_rays*2), Tyre nodes: A, B, A, B... (num_rays*2)
00066     // meshwheels     | addWheel()    | Tyre nodes: A, B, A, B... (num_rays*2)
00067     // meshwheels2    | addWheel()    | Tyre nodes: A, B, A, B... (num_rays*2)
00068     // flexbodywheels | addWheel3()   | Rim nodes: A, B, A, B... (num_rays*2), Tyre nodes: A, B, A, B... (num_rays*2)
00069 
00070     NodeMapEntry::OriginDetail detail;
00071     if (generated_from == File::KEYWORD_FLEXBODYWHEELS || generated_from == File::KEYWORD_WHEELS2)
00072     {
00073         // Rim nodes
00074         for (int i = 0; i < num_rays*2; ++i)
00075         {
00076             detail = (i % 2 == 0) ? NodeMapEntry::DETAIL_WHEEL_RIM_A : NodeMapEntry::DETAIL_WHEEL_RIM_B;
00077             this->AddGeneratedNode(generated_from, detail);
00078         }
00079     }
00080     // Tyre nodes
00081     for (int i = 0; i < num_rays*2; ++i)
00082     {
00083         detail = (i % 2 == 0) ? NodeMapEntry::DETAIL_WHEEL_TYRE_A : NodeMapEntry::DETAIL_WHEEL_TYRE_B;
00084         this->AddGeneratedNode(generated_from, detail);
00085     }

00127 unsigned SequentialImporter::GetNodeArrayOffset(File::Keyword keyword)
00128 {
00129     // Reverse order of elements in nodes-array
00130     // "break" keyword is omitted - control falls through!
00131     unsigned out_offset = 0;
00132     switch (keyword)
00133     {
00134     case File::KEYWORD_FLEXBODYWHEELS: out_offset += m_num_meshwheels2_nodes; // NO break! 
00135     case File::KEYWORD_MESHWHEELS2:    out_offset += m_num_meshwheels_nodes;  // NO break!
00136     case File::KEYWORD_MESHWHEELS:     out_offset += m_num_wheels2_nodes;     // NO break!
00137     case File::KEYWORD_WHEELS2:        out_offset += m_num_wheels_nodes;      // NO break!
00138     case File::KEYWORD_WHEELS:         out_offset += m_num_cinecam_nodes;     // NO break!
00139     case File::KEYWORD_CINECAM:        out_offset += m_num_named_nodes;       // NO break!
00140     case File::KEYWORD_NODES2:         out_offset += m_num_numbered_nodes;    // NO break!
00141     case File::KEYWORD_NODES:          break; // Starts at 0
00142     default: break;
00143     }
00144     return out_offset;
00145 }

00149     if (index_in >= m_all_nodes.size())

00160         std::stringstream msg;

00166         }
00167         this->AddMessage(Message::TYPE_INFO, msg.str());
00168     }
00169     else
00170     {
00171         ++m_num_resolved_to_self; // The happy statistic -> nodes which resolved to the same index.
00172     }
00173     return Node::Ref(TOSTRING(index_out), index_out, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, def_line_number);

00235 void SequentialImporter::ResolveFlexbodyForset(std::vector<Node::Range>& in_ranges, std::vector<Node::Ref>& out_nodes)
00236 {
00237     auto range_itor = in_ranges.begin();
00238     auto range_end  = in_ranges.end();
00239     for ( ; range_itor != range_end; ++range_itor)
00240     {
00241         Node::Range& range = *range_itor;
00242         if (!range.IsRange())
00243         {
00244             if (!range.start.GetImportState_IsValid())
00245             {
00246                 std::stringstream msg;
00247                 msg << "ResolveFlexbodyForset(): Skipping node because it's marked INVALID:" << range.start.ToString();
00248                 this->AddMessage(Message::TYPE_ERROR, msg.str());
00249             }
00250             else
00251             {
00252                 Node::Ref node_ref = this->ResolveNodeByIndex(range.start.Num(), range.start.GetLineNumber()); // Forset nodes are numbered-only
00253                 // Invalid nodes are simply thrown away, for backwards compatibility 
00254                 // (FlexBody loops through existing nodes first, and then evaluates if they are in the SET)
00255                 if (node_ref.IsValidAnyState())
00256                 {
00257                     out_nodes.push_back(node_ref);
00258                 }
00259                 else
00260                 {
00261                     std::stringstream msg;
00262                     msg << "ResolveFlexbodyForset(): Stand-alone node [" << range.start.ToString() << "] resolved invalid, removing from FORSET";
00263                     this->AddMessage(Message::TYPE_WARNING, msg.str());
00264                 }
00265             }
00266         }
00267         else // It's a range
00268         {
00269             if (!range.start.GetImportState_IsValid() || !range.end.GetImportState_IsValid())
00270             {
00271                 std::stringstream msg;
00272                 msg << "ResolveFlexbodyForset(): Skipping range because some nodes are marked INVALID, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
00273                 this->AddMessage(Message::TYPE_ERROR, msg.str());
00274             }
00275             else if (range.start.GetImportState_IsResolvedNamed() || range.end.GetImportState_IsResolvedNamed())
00276             {
00277                 std::stringstream msg;
00278                 msg << "ResolveFlexbodyForset(): Some nodes in range are already resolved as named, unable to process, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
00279                 this->AddMessage(Message::TYPE_ERROR, msg.str());
00280             }
00281             else
00282             {
00283                 unsigned int end_index = range.end.Num();

00286                 {
00287                     Node::Ref node_ref = this->ResolveNodeByIndex(i, line_num); // Forset nodes are numbered-only

00292                         out_nodes.push_back(node_ref);
00293                     }
00294                     else
00295                     {
00296                         std::stringstream msg;
00297                         msg << "ResolveFlexbodyForset(): Node ["<<i<<"] from range [" << range.start.ToString() << " - " << range.end.ToString() << "] resolved invalid, removing from FORSET";
00298                         this->AddMessage(Message::TYPE_WARNING, msg.str());
00299                     }
00300                 }
00301             }
