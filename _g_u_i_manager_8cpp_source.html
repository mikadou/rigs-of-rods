227 String GUIManager::getRandomWallpaperImage()
228 {
229 
230  FileInfoListPtr files = ResourceGroupManager::getSingleton().findResourceFileInfo("Wallpapers", "*.jpg", false);
231  if (files.isNull() || files->empty())
232  {
233  files = ResourceGroupManager::getSingleton().findResourceFileInfo("Wallpapers", "*.png", false);
234  if (files.isNull() || files->empty())
235  return "";
236  }
237  srand ( time(NULL) );
238 
239  int num = 0;
240  for (int i = 0; i<Math::RangeRandom(0, 10); i++)
241  num = Math::RangeRandom(0, files->size());
