00044 void Screwprop::updateForces(int update)
00045 {
00046     if (!gEnv->terrainManager->getWater()) return;
00047     float depth=gEnv->terrainManager->getWater()->getHeightWaves(nodes[noderef].AbsPosition)-nodes[noderef].AbsPosition.y;
00048     if (depth<0) return; //out of water!
00049     Vector3 dir=nodes[nodeback].RelPosition-nodes[noderef].RelPosition;
00050     Vector3 rudaxis=nodes[noderef].RelPosition-nodes[nodeup].RelPosition;
00051     dir.normalise();
00052     if (reverse) dir=-dir;
00053     rudaxis.normalise();
00054     dir=(throtle*fullpower)*(Quaternion(Degree(rudder),rudaxis)*dir);
00055     nodes[noderef].Forces += dir;
00056     
00057     if (update && splashp && throtle>0.1)
00058     {
00059         if (depth<0.2)
00060             splashp->allocSplash(nodes[noderef].AbsPosition, 10.0*dir/fullpower);
00061         else
00062             splashp->allocSplash(nodes[noderef].AbsPosition, 5.0*dir/fullpower);
00063         ripplep->allocRipple(nodes[noderef].AbsPosition, 10.0*dir/fullpower);
00064     }
00065 }

00067 void Screwprop::setThrottle(float val)
00068 {
00069     if (val>1.0) val=1.0;
00070     if (val<-1.0) val=-1.0;
00071     throtle=fabs(val);

00073     //pseudo-rpm

00076     SoundScriptManager::getSingleton().modulate(trucknum, SS_MOD_ENGINE, prpm);
