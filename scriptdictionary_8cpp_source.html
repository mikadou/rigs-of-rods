158 bool CScriptDictionary::Get(const string &key, void *value, int typeId) const
159 {
160  map<string, valueStruct>::const_iterator it;
161  it = dict.find(key);
162  if( it != dict.end() )
163  {
164  // Return the value
165  if( typeId & asTYPEID_OBJHANDLE )
166  {
167  // A handle can be retrieved if the stored type is a handle of same or compatible type
168  // or if the stored type is an object that implements the interface that the handle refer to.
169  if( (it->second.typeId & asTYPEID_MASK_OBJECT) &&
170  engine->IsHandleCompatibleWithObject(it->second.valueObj, it->second.typeId, typeId) )
171  {
172  engine->AddRefScriptObject(it->second.valueObj, it->second.typeId);
173  *(void**)value = it->second.valueObj;
174 
175  return true;
176  }
177  }
178  else if( typeId & asTYPEID_MASK_OBJECT )
179  {
180  // Verify that the copy can be made
181  bool isCompatible = false;
182  if( it->second.typeId == typeId )
183  isCompatible = true;
184 
185  // Copy the object into the given reference
186  if( isCompatible )
187  {
188  engine->CopyScriptObject(value, it->second.valueObj, typeId);
189 
190  return true;
191  }
192  }
193  else
194  {
195  if( it->second.typeId == typeId )
196  {
197  int size = engine->GetSizeOfPrimitiveType(typeId);
198  memcpy(value, &it->second.valueInt, size);
199  return true;
200  }
201 
202  // We know all numbers are stored as either int64 or double, since we register overloaded functions for those
203  if( it->second.typeId == asTYPEID_INT64 && typeId == asTYPEID_DOUBLE )
204  {
205  *(double*)value = double(it->second.valueInt);
206  return true;
207  }
208  else if( it->second.typeId == asTYPEID_DOUBLE && typeId == asTYPEID_INT64 )
209  {
210  *(asINT64*)value = asINT64(it->second.valueFlt);
211  return true;
212  }
213  }
214  }
215 
216  // AngelScript has already initialized the value with a default value,
217  // so we don't have to do anything if we don't find the element, or if
218  // the element is incompatible with the requested type.
219 
220  return false;
221 }
