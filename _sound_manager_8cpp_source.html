00161

00163     {
00164         audio_sources[i]->computeAudibility(camera_position);
00165         audio_sources_most_audible[i].first = i;
00166         audio_sources_most_audible[i].second = audio_sources[i]->audibility;
00167     }
00168     // sort first 'num_hardware_sources' sources by audibility
00169     // see: https://en.wikipedia.org/wiki/Selection_algorithm
00170     if ((audio_sources_in_use_count - 1) > hardware_sources_num)
00171     {
00172         std::nth_element(audio_sources_most_audible, audio_sources_most_audible+hardware_sources_num, audio_sources_most_audible+audio_sources_in_use_count-1, compareByAudibility);
00173     }
00174     // retire out of range sources first
00175     for (int i=0; i < audio_sources_in_use_count; i++)
00176     {
00177         if (audio_sources[audio_sources_most_audible[i].first]->hardware_index != -1 && (i >= hardware_sources_num || audio_sources_most_audible[i].second == 0))
00178             retire(audio_sources_most_audible[i].first);
00179     }
00180     // assign new sources
00181     for (int i=0; i < std::min(audio_sources_in_use_count, hardware_sources_num); i++)
00182     {
00183         if (audio_sources[audio_sources_most_audible[i].first]->hardware_index == -1 && audio_sources_most_audible[i].second > 0)
00184         {
00185             for (int j=0; j < hardware_sources_num; j++)

00197 void SoundManager::recomputeSource(int source_index, int reason, float vfl, Vector3 *vvec)
00198 {
00199     if (!audio_device) return;
00200     audio_sources[source_index]->computeAudibility(camera_position);
00201 
00202     if (audio_sources[source_index]->audibility == 0.0f)
00203     {
00204         if (audio_sources[source_index]->hardware_index != -1)
00205         {
00206             // retire the source if it is currently assigned
00207             retire(source_index);
00208         }
00209     } else
00210     {
00211         // this is a potentially audible m_audio_sources[source_index]
00212         if (audio_sources[source_index]->hardware_index != -1)
00213         {
00214             // m_audio_sources[source_index] already playing
00215             // update the AL settings
00216             switch (reason)
00217             {
00218             case Sound::REASON_PLAY: alSourcePlay(hardware_sources[audio_sources[source_index]->hardware_index]); break;
00219             case Sound::REASON_STOP: alSourceStop(hardware_sources[audio_sources[source_index]->hardware_index]); break;
00220             case Sound::REASON_GAIN: alSourcef(hardware_sources[audio_sources[source_index]->hardware_index], AL_GAIN, vfl * master_volume); break;
00221             case Sound::REASON_LOOP: alSourcei(hardware_sources[audio_sources[source_index]->hardware_index], AL_LOOPING, (vfl > 0.5) ? AL_TRUE : AL_FALSE); break;
00222             case Sound::REASON_PTCH: alSourcef(hardware_sources[audio_sources[source_index]->hardware_index], AL_PITCH, vfl); break;
00223             case Sound::REASON_POSN: alSource3f(hardware_sources[audio_sources[source_index]->hardware_index], AL_POSITION, vvec->x, vvec->y, vvec->z); break;
00224             case Sound::REASON_VLCT: alSource3f(hardware_sources[audio_sources[source_index]->hardware_index], AL_VELOCITY, vvec->x, vvec->y, vvec->z); break;
00225             default: break;
00226             }
00227         } else
00228         {
00229             // try to make it play by the hardware
00230             // check if there is one free m_audio_sources[source_index] in the pool
00231             if (hardware_sources_in_use_count < hardware_sources_num)
00232             {
00233                 for (int i=0; i < hardware_sources_num; i++)
00234                 {
00235                     if (hardware_sources_map[i] == -1)
00236                     {
00237                         assign(source_index, i);
00238                         break;
00239                     }
00240                 }
00241             } else
00242             {
00243                 // now, compute who is the faintest
00244                 // note: we know the table m_hardware_sources_map is full!
00245                 float fv = 1.0f;
00246                 int al_faintest = 0;
00247                 for (int i=0; i < hardware_sources_num; i++)
00248                 {
00249                     if (hardware_sources_map[i] >= 0 && audio_sources[hardware_sources_map[i]]->audibility < fv)
00250                     {
00251                         fv = audio_sources[hardware_sources_map[i]]->audibility;
00252                         al_faintest = i;
00253                     }
00254                 }
00255                 // check to ensure that the sound is louder than the faintest sound currently playing
00256                 if (fv < audio_sources[source_index]->audibility)
00257                 {
00258                     // this new m_audio_sources[source_index] is louder than the faintest!
00259                     retire(hardware_sources_map[al_faintest]);
00260                     assign(source_index, al_faintest);
00261                 }
00262                 // else this m_audio_sources[source_index] is too faint, we don't play it!
00263             }
00264         }
00265     }
