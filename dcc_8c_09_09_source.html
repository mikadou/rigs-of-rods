94 static void libirc_dcc_add_descriptors (irc_session_t * ircsession, fd_set *in_set, fd_set *out_set, int * maxfd)
95 {
96  irc_dcc_session_t * dcc, *dcc_next;
97  time_t now = time (0);
98 
99  libirc_mutex_lock (&ircsession->mutex_dcc);
100 
101  // Preprocessing DCC list:
102  // - ask DCC send callbacks for data;
103  // - remove unused DCC structures
104  for ( dcc = ircsession->dcc_sessions; dcc; dcc = dcc_next )
105  {
106  dcc_next = dcc->next;
107 
108  // Remove timed-out sessions
109  if ( (dcc->state == LIBIRC_STATE_CONNECTING
110  || dcc->state == LIBIRC_STATE_INIT
111  || dcc->state == LIBIRC_STATE_LISTENING)
112  && now - dcc->timeout > ircsession->dcc_timeout )
113  {
114  // Inform the caller about DCC timeout.
115  // Do not inform when state is LIBIRC_STATE_INIT - session
116  // was initiated from someone else, and callbacks aren't set yet.
117  if ( dcc->state != LIBIRC_STATE_INIT )
118  {
119  libirc_mutex_unlock (&ircsession->mutex_dcc);
120 
121  if ( dcc->cb )
122  (*dcc->cb)(ircsession, dcc->id, LIBIRC_ERR_TIMEOUT, dcc->ctx, 0, 0);
123 
124  libirc_mutex_lock (&ircsession->mutex_dcc);
125  }
126 
127  libirc_remove_dcc_session (ircsession, dcc, 0);
128  }
129 
130  /*
131  * If we're sending file, and the output buffer is empty, we need
132  * to provide some data.
133  */
134  if ( dcc->state == LIBIRC_STATE_CONNECTED
135  && dcc->dccmode == LIBIRC_DCC_SENDFILE
136  && dcc->dccsend_file_fp
137  && dcc->outgoing_offset == 0 )
138  {
139  int len = fread (dcc->outgoing_buf, 1, sizeof (dcc->outgoing_buf), dcc->dccsend_file_fp);
140 
141  if ( len <= 0 )
142  {
143  int err = (len < 0 ? LIBIRC_ERR_READ : 0);
144 
145  libirc_mutex_unlock (&ircsession->mutex_dcc);
146 
147  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, 0, 0);
148  libirc_mutex_lock (&ircsession->mutex_dcc);
149  libirc_dcc_destroy_nolock (ircsession, dcc->id);
150  }
151  else
152  dcc->outgoing_offset = len;
153  }
154 
155  // Clean up unused sessions
156  if ( dcc->state == LIBIRC_STATE_REMOVED )
157  libirc_remove_dcc_session (ircsession, dcc, 0);
158  }
159 
160  for ( dcc = ircsession->dcc_sessions; dcc; dcc = dcc->next )
161  {
162  switch (dcc->state)
163  {
164  case LIBIRC_STATE_LISTENING:
165  // While listening, only in_set descriptor should be set
166  libirc_add_to_set (dcc->sock, in_set, maxfd);
167  break;
168 
169  case LIBIRC_STATE_CONNECTING:
170  // While connection, only out_set descriptor should be set
171  libirc_add_to_set (dcc->sock, out_set, maxfd);
172  break;
173 
174  case LIBIRC_STATE_CONNECTED:
175  // Add input descriptor if there is space in input buffer
176  // and it is DCC chat (during DCC send, there is nothing to recv)
177  if ( dcc->incoming_offset < sizeof(dcc->incoming_buf) - 1 )
178  libirc_add_to_set (dcc->sock, in_set, maxfd);
179 
180  // Add output descriptor if there is something in output buffer
181  libirc_mutex_lock (&dcc->mutex_outbuf);
182 
183  if ( dcc->outgoing_offset > 0 )
184  libirc_add_to_set (dcc->sock, out_set, maxfd);
185 
186  libirc_mutex_unlock (&dcc->mutex_outbuf);
187  break;
188 
189  case LIBIRC_STATE_CONFIRM_SIZE:
190  /*
191  * If we're receiving file, then WE should confirm the transferred
192  * part (so we have to sent data). But if we're sending the file,
193  * then RECEIVER should confirm the packet, so we have to receive
194  * data.
195  *
196  * We don't need to LOCK_DCC_OUTBUF - during file transfer, buffers
197  * can't change asynchronously.
198  */
199  if ( dcc->dccmode == LIBIRC_DCC_RECVFILE && dcc->outgoing_offset > 0 )
200  libirc_add_to_set (dcc->sock, out_set, maxfd);
201 
202  if ( dcc->dccmode == LIBIRC_DCC_SENDFILE && dcc->incoming_offset < 4 )
203  libirc_add_to_set (dcc->sock, in_set, maxfd);
204  }
205  }
206 
207  libirc_mutex_unlock (&ircsession->mutex_dcc);
208 }

211 static void libirc_dcc_process_descriptors (irc_session_t * ircsession, fd_set *in_set, fd_set *out_set)
212 {
213  irc_dcc_session_t * dcc;
214 
215  /*
216  * We need to use such a complex scheme here, because on every callback
217  * a number of DCC sessions could be destroyed.
218  */
219  libirc_mutex_lock (&ircsession->mutex_dcc);
220 
221  for ( dcc = ircsession->dcc_sessions; dcc; dcc = dcc->next )
222  {
223  if ( dcc->state == LIBIRC_STATE_LISTENING
224  && FD_ISSET (dcc->sock, in_set) )
225  {
226  socklen_t len = sizeof(dcc->remote_addr);
227  int nsock, err = 0;
228 
229  // New connection is available; accept it.
230  if ( socket_accept (&dcc->sock, (socket_t *)&nsock, (struct sockaddr *) &dcc->remote_addr, &len) )
231  err = LIBIRC_ERR_ACCEPT;
232 
233  // On success, change the active socket and change the state
234  if ( err == 0 )
235  {
236  // close the listen socket, and replace it by a newly
237  // accepted
238  socket_close (&dcc->sock);
239  dcc->sock = nsock;
240  dcc->state = LIBIRC_STATE_CONNECTED;
241  }
242 
243  // If this is DCC chat, inform the caller about accept()
244  // success or failure.
245  // Otherwise (DCC send) there is no reason.
246  if ( dcc->dccmode == LIBIRC_DCC_CHAT )
247  {
248  libirc_mutex_unlock (&ircsession->mutex_dcc);
249  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, 0, 0);
250  libirc_mutex_lock (&ircsession->mutex_dcc);
251  }
252 
253  if ( err )
254  libirc_dcc_destroy_nolock (ircsession, dcc->id);
255  }
256 
257  if ( dcc->state == LIBIRC_STATE_CONNECTING
258  && FD_ISSET (dcc->sock, out_set) )
259  {
260  // Now we have to determine whether the socket is connected
261  // or the connect is failed
262  struct sockaddr_in saddr;
263  socklen_t slen = sizeof(saddr);
264  int err = 0;
265 
266  if ( getpeername (dcc->sock, (struct sockaddr*)&saddr, &slen) < 0 )
267  err = LIBIRC_ERR_CONNECT;
268 
269  // On success, change the state
270  if ( err == 0 )
271  dcc->state = LIBIRC_STATE_CONNECTED;
272 
273  // If this is DCC chat, inform the caller about connect()
274  // success or failure.
275  // Otherwise (DCC send) there is no reason.
276  if ( dcc->dccmode == LIBIRC_DCC_CHAT )
277  {
278  libirc_mutex_unlock (&ircsession->mutex_dcc);
279  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, 0, 0);
280  libirc_mutex_lock (&ircsession->mutex_dcc);
281  }
282 
283  if ( err )
284  libirc_dcc_destroy_nolock (ircsession, dcc->id);
285  }
286 
287  if ( dcc->state == LIBIRC_STATE_CONNECTED
288  || dcc->state == LIBIRC_STATE_CONFIRM_SIZE )
289  {
290  if ( FD_ISSET (dcc->sock, in_set) )
291  {
292  int length, offset = 0, err = 0;
293 
294  unsigned int amount = sizeof (dcc->incoming_buf) - dcc->incoming_offset;
295 
296  length = socket_recv (&dcc->sock, dcc->incoming_buf + dcc->incoming_offset, amount);
297 
298  if ( length < 0 )
299  {
300  err = LIBIRC_ERR_READ;
301  }
302  else if ( length == 0 )
303  {
304  err = LIBIRC_ERR_CLOSED;
305 
306  if ( dcc->dccsend_file_fp )
307  {
308  fclose (dcc->dccsend_file_fp);
309  dcc->dccsend_file_fp = 0;
310  }
311  }
312  else
313  {
314  dcc->incoming_offset += length;
315 
316  if ( dcc->dccmode != LIBIRC_DCC_CHAT )
317  offset = dcc->incoming_offset;
318  else
319  offset = libirc_findcrorlf (dcc->incoming_buf, dcc->incoming_offset);
320 
321  /*
322  * In LIBIRC_STATE_CONFIRM_SIZE state we don't call any
323  * callbacks (except there is an error). We just receive
324  * the data, and compare it with the amount sent.
325  */
326  if ( dcc->state == LIBIRC_STATE_CONFIRM_SIZE )
327  {
328  if ( dcc->dccmode != LIBIRC_DCC_SENDFILE )
329  abort();
330 
331  if ( dcc->incoming_offset == 4 )
332  {
333  unsigned int received_size = ntohl (*((unsigned int*)dcc->incoming_buf));
334 
335  // Sent size confirmed
336  if ( dcc->file_confirm_offset == received_size )
337  {
338  dcc->state = LIBIRC_STATE_CONNECTED;
339  dcc->incoming_offset = 0;
340  }
341  else
342  err = LIBIRC_ERR_WRITE;
343  }
344  }
345  else
346  {
347  /*
348  * If it is DCC_CHAT, we send a 0-terminated string
349  * (which is smaller than offset). Otherwise we send
350  * a full buffer.
351  */
352  libirc_mutex_unlock (&ircsession->mutex_dcc);
353 
354  if ( dcc->dccmode != LIBIRC_DCC_CHAT )
355  {
356  if ( dcc->dccmode != LIBIRC_DCC_RECVFILE )
357  abort();
358 
359  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, dcc->incoming_buf, offset);
360 
361  /*
362  * If the session is not terminated in callback,
363  * put the sent amount into the sent_packet_size_net_byteorder
364  */
365  if ( dcc->state != LIBIRC_STATE_REMOVED )
366  {
367  dcc->state = LIBIRC_STATE_CONFIRM_SIZE;
368  dcc->file_confirm_offset += offset;
369  *((unsigned int*)dcc->outgoing_buf) = htonl (dcc->file_confirm_offset);
370  dcc->outgoing_offset = 4;
371  }
372  }
373  else
374  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, dcc->incoming_buf, strlen(dcc->incoming_buf));
375 
376  libirc_mutex_lock (&ircsession->mutex_dcc);
377 
378  if ( dcc->incoming_offset - offset > 0 )
379  memmove (dcc->incoming_buf, dcc->incoming_buf + offset, dcc->incoming_offset - offset);
380 
381  dcc->incoming_offset -= offset;
382  }
383  }
384 
385  /*
386  * If error arises somewhere above, we inform the caller
387  * of failure, and destroy this session.
388  */
389  if ( err )
390  {
391  libirc_mutex_unlock (&ircsession->mutex_dcc);
392  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, 0, 0);
393  libirc_mutex_lock (&ircsession->mutex_dcc);
394  libirc_dcc_destroy_nolock (ircsession, dcc->id);
395  }
396  }
397 
398  /*
399  * Session might be closed (with sock = -1) after the in_set
400  * processing, so before out_set processing we should check
401  * for this case
402  */
403  if ( dcc->state == LIBIRC_STATE_REMOVED )
404  continue;
405 
406  /*
407  * Write bit set - we can send() something, and it won't block.
408  */
409  if ( FD_ISSET (dcc->sock, out_set) )
410  {
411  int length, offset, err = 0;
412 
413  /*
414  * Because in some cases outgoing_buf could be changed
415  * asynchronously (by another thread), we should lock
416  * it.
417  */
418  libirc_mutex_lock (&dcc->mutex_outbuf);
419 
420  offset = dcc->outgoing_offset;
421 
422  if ( offset > 0 )
423  {
424  length = socket_send (&dcc->sock, dcc->outgoing_buf, offset);
425 
426  if ( length < 0 )
427  err = LIBIRC_ERR_WRITE;
428  else if ( length == 0 )
429  err = LIBIRC_ERR_CLOSED;
430  else
431  {
432  /*
433  * If this was DCC_SENDFILE, and we just sent a packet,
434  * change the state to wait for confirmation (and store
435  * sent packet size)
436  */
437  if ( dcc->state == LIBIRC_STATE_CONNECTED
438  && dcc->dccmode == LIBIRC_DCC_SENDFILE )
439  {
440  dcc->file_confirm_offset += offset;
441  dcc->state = LIBIRC_STATE_CONFIRM_SIZE;
442 
443  libirc_mutex_unlock (&ircsession->mutex_dcc);
444  libirc_mutex_unlock (&dcc->mutex_outbuf);
445  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, 0, offset);
446  libirc_mutex_lock (&ircsession->mutex_dcc);
447  libirc_mutex_lock (&dcc->mutex_outbuf);
448  }
449 
450  if ( dcc->outgoing_offset - length > 0 )
451  memmove (dcc->outgoing_buf, dcc->outgoing_buf + length, dcc->outgoing_offset - length);
452 
453  dcc->outgoing_offset -= length;
454 
455  /*
456  * If we just sent the confirmation data, change state
457  * back.
458  */
459  if ( dcc->state == LIBIRC_STATE_CONFIRM_SIZE
460  && dcc->dccmode == LIBIRC_DCC_RECVFILE
461  && dcc->outgoing_offset == 0 )
462  {
463  /*
464  * If the file is already received, we should inform
465  * the caller, and close the session.
466  */
467  if ( dcc->received_file_size == dcc->file_confirm_offset )
468  {
469  libirc_mutex_unlock (&ircsession->mutex_dcc);
470  libirc_mutex_unlock (&dcc->mutex_outbuf);
471  (*dcc->cb)(ircsession, dcc->id, 0, dcc->ctx, 0, 0);
472  libirc_dcc_destroy_nolock (ircsession, dcc->id);
473  }
474  else
475  {
476  /* Continue to receive the file */
477  dcc->state = LIBIRC_STATE_CONNECTED;
478  }
479  }
480  }
481  }
482 
483  libirc_mutex_unlock (&dcc->mutex_outbuf);
484 
485  /*
486  * If error arises somewhere above, we inform the caller
487  * of failure, and destroy this session.
488  */
489  if ( err )
490  {
491  libirc_mutex_unlock (&ircsession->mutex_dcc);
492  (*dcc->cb)(ircsession, dcc->id, err, dcc->ctx, 0, 0);
493  libirc_mutex_lock (&ircsession->mutex_dcc);
494 
495  libirc_dcc_destroy_nolock (ircsession, dcc->id);
496  }
497  }
498  }
499  }
500 
501  libirc_mutex_unlock (&ircsession->mutex_dcc);
502 }

505 static int libirc_new_dcc_session (irc_session_t * session, unsigned long ip, unsigned short port, int dccmode, void * ctx, irc_dcc_session_t ** pdcc)
506 {
507  irc_dcc_session_t * dcc = (irc_dcc_session_t *)malloc (sizeof(irc_dcc_session_t));
508 
509  if ( !dcc )
510  return LIBIRC_ERR_NOMEM;
511 
512  // setup
513  memset (dcc, 0, sizeof(irc_dcc_session_t));
514 
515  dcc->dccsend_file_fp = 0;
516 
517  if ( libirc_mutex_init (&dcc->mutex_outbuf) )
518  goto cleanup_exit_error;
519 
520  if ( socket_create (PF_INET, SOCK_STREAM, &dcc->sock) )
521  goto cleanup_exit_error;
522 
523  if ( !ip )
524  {
525  struct sockaddr_in saddr;
526  unsigned long arg = 1;
527 
528  setsockopt (dcc->sock, SOL_SOCKET, SO_REUSEADDR, (char*)&arg, sizeof(arg));
529 
530  memset (&saddr, 0, sizeof(saddr));
531  saddr.sin_family = AF_INET;
532  memcpy (&saddr.sin_addr, &session->local_addr, sizeof(session->local_addr));
533  saddr.sin_port = htons (0);
534 
535  if ( bind (dcc->sock, (struct sockaddr *) &saddr, sizeof(saddr)) < 0 )
536  goto cleanup_exit_error;
537 
538  if ( listen (dcc->sock, 5) < 0 )
539  goto cleanup_exit_error;
540 
541  dcc->state = LIBIRC_STATE_LISTENING;
542  }
543  else
544  {
545  // make socket non-blocking, so connect() call won't block
546  if ( socket_make_nonblocking (&dcc->sock) )
547  goto cleanup_exit_error;
548 
549  memset (&dcc->remote_addr, 0, sizeof(dcc->remote_addr));
550  dcc->remote_addr.sin_family = AF_INET;
551  dcc->remote_addr.sin_addr.s_addr = htonl (ip); // what idiot came up with idea to send IP address in host-byteorder?
552  dcc->remote_addr.sin_port = htons(port);
553 
554  dcc->state = LIBIRC_STATE_INIT;
555  }
556 
557  dcc->dccmode = dccmode;
558  dcc->ctx = ctx;
559  time (&dcc->timeout);
560 
561  // and store it
562  libirc_mutex_lock (&session->mutex_dcc);
563 
564  dcc->id = session->dcc_last_id++;
565  dcc->next = session->dcc_sessions;
566  session->dcc_sessions = dcc;
567 
568  libirc_mutex_unlock (&session->mutex_dcc);
569 
570  *pdcc = dcc;
571  return 0;
572 
573 cleanup_exit_error:
574  if ( dcc->sock >= 0 )
575  socket_close (&dcc->sock);
576 
577  free (dcc);
578  return LIBIRC_ERR_SOCKET;
579 }

649 int irc_dcc_msg (irc_session_t * session, irc_dcc_t dccid, const char * text)

653  if ( !dcc )
654  return 1;

795 int irc_dcc_sendfile (irc_session_t * session, void * ctx, const char * nick, const char * filename, irc_dcc_callback_t callback, irc_dcc_t * dccid)
796 {
797  struct sockaddr_in saddr;
798  socklen_t len = sizeof(saddr);
799  char cmdbuf[128], notbuf[128];
800  irc_dcc_session_t * dcc;
801  const char * p;
802  int err;
803  long filesize;
804 
805  if ( !session )
806  {
807  return 1;
808  }
809 
810  if ( !dccid || !filename || !callback )
811  {
812  session->lasterror = LIBIRC_ERR_INVAL;
813  return 1;
814  }
815 
816  if ( session->state != LIBIRC_STATE_CONNECTED )
817  {
818  session->lasterror = LIBIRC_ERR_STATE;
819  return 1;
820  }
821 
822  if ( (err = libirc_new_dcc_session (session, 0, 0, LIBIRC_DCC_SENDFILE, ctx, &dcc)) != 0 )
823  {
824  session->lasterror = err;
825  return 1;
826  }
827 
828  if ( (dcc->dccsend_file_fp = fopen (filename, "rb")) == 0 )
829  {
830  libirc_remove_dcc_session (session, dcc, 1);
831  session->lasterror = LIBIRC_ERR_OPENFILE;
832  return 1;
833  }
834 
835  /* Get file length */
836  if ( fseek (dcc->dccsend_file_fp, 0, SEEK_END)
837  || (filesize = ftell (dcc->dccsend_file_fp)) == -1
838  || fseek (dcc->dccsend_file_fp, 0, SEEK_SET) )
839  {
840  libirc_remove_dcc_session (session, dcc, 1);
841  session->lasterror = LIBIRC_ERR_NODCCSEND;
842  return 1;
843  }
844 
845  if ( getsockname (dcc->sock, (struct sockaddr*) &saddr, &len) < 0 )
846  {
847  libirc_remove_dcc_session (session, dcc, 1);
848  session->lasterror = LIBIRC_ERR_SOCKET;
849  return 1;
850  }
851 
852  // Remove path from the filename
853  if ( (p = strrchr (filename, '\\')) == 0
854  && (p = strrchr (filename, '/')) == 0 )
855  p = filename;
856  else
857  p++; // skip directory slash
858 
859  sprintf (notbuf, "DCC Send %s (%s)", p, inet_ntoa (saddr.sin_addr));
860  sprintf (cmdbuf, "DCC SEND %s %lu %u %ld", p, (unsigned long) ntohl (saddr.sin_addr.s_addr), ntohs (saddr.sin_port), filesize);
861 
862  if ( irc_cmd_notice (session, nick, notbuf)
863  || irc_cmd_ctcp_request (session, nick, cmdbuf) )
864  {
865  libirc_remove_dcc_session (session, dcc, 1);
866  return 1;
867  }
868 
869  *dccid = dcc->id;
870  dcc->cb = callback;
871 
872  return 0;
873 }
