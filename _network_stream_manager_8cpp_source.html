64  stream->isOrigin = true;
65 
66  // add new stream map to the streams map
67  if (streams.find(mysourceid) == streams.end())
68  streams[mysourceid] = std::map < unsigned int, Streamable *>();
69  // map the stream
70  streams[mysourceid][streamid] = stream;
71  LOG("adding local stream: " + TOSTRING(mysourceid) + ":"+ TOSTRING(streamid) + ", type: " + TOSTRING(reg->type));
72  // send stream setup notice to server
73  if (size == 0) size = sizeof(stream_register_t);
74  stream->addPacket(MSG2_STREAM_REGISTER, size, (char*)reg);
75 
76  // increase stream counter
77  streamid++;
78  MUTEX_UNLOCK(&stream_mutex);
79 #endif // USE_SOCKETW

203  //char *buffer = 0;

205 
206  std::map < int, std::map < unsigned int, Streamable *> >::iterator it;
207  for (it=streams.begin(); it!=streams.end(); it++)
208  {
209  std::map<unsigned int,Streamable *>::iterator it2;
210  for (it2=it->second.begin(); it2!=it->second.end(); it2++)
211  {
212  if (!it2->second) continue;
213  std::deque <Streamable::bufferedPacket_t> *packets = it2->second->getPacketQueue();
214 
215  while (!packets->empty())
216  {
217  // remove oldest packet in queue

219

225  swprintf(emsg, 256, tmp.asWStr_c_str(), etype);
226  net->netFatalError(UTFString(emsg));
227  MUTEX_UNLOCK(&stream_mutex);
228  return;
229  }
230 
231  packets->pop_front();
232  }
233
