376 struct RigAggregateBeams2Data
377 {
378  RigAggregatePlainBeamsData plain_beams;
379  RigAggregateHydrosData hydros;
380  RigAggregateCommands2Data commands2;
381  RigAggregateShocksData shocks;
382  RigAggregateShocks2Data shocks2;
383 
384  inline int GetTotalNumSelectedBeams() const
385  {
386  return plain_beams.num_selected + hydros.num_selected + commands2.num_selected + shocks.num_selected + shocks2.num_selected;
387  }
388 
389  inline bool HasMixedBeamTypes() const
390  {
391  int total = GetTotalNumSelectedBeams();
392  return (total != plain_beams.num_selected) && (total != hydros.num_selected) && (total != commands2.num_selected)
393  && (total != shocks.num_selected) && (total != shocks2.num_selected);
394  }
395 
396  bool IsDetacherGroupUniform()
397  {
398  bool detacher_group_is_uniform
399  = plain_beams.IsDetacherGroupUniform() && shocks .IsDetacherGroupUniform()
400  && hydros .IsDetacherGroupUniform() && shocks2.IsDetacherGroupUniform()
401  && commands2 .IsDetacherGroupUniform();
402 
403  if (detacher_group_is_uniform)
404  {
405  int detacher_group_id = plain_beams.detacher_group;
406  detacher_group_is_uniform = (hydros.detacher_group != detacher_group_id) ? false : detacher_group_is_uniform;
407  detacher_group_is_uniform = (shocks.detacher_group != detacher_group_id) ? false : detacher_group_is_uniform;
408  detacher_group_is_uniform = (shocks2.detacher_group != detacher_group_id) ? false : detacher_group_is_uniform;
409  detacher_group_is_uniform = (commands2.detacher_group != detacher_group_id) ? false : detacher_group_is_uniform;
410  }
411  return detacher_group_is_uniform;
412  }
413 };
