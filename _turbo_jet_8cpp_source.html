227 void Turbojet::updateForces(float dt, int doUpdate)
228 {
229  if (doUpdate)
230  {
231 #ifdef USE_OPENAL
232  //sound update
233  SoundScriptManager::getSingleton().modulate(trucknum, mod_id, rpm);
234 #endif //OPENAL
235  }
236  timer+=dt;
237  axis=nodes[nodefront].RelPosition-nodes[nodeback].RelPosition;
238  float axlen=axis.length();
239  axis=axis/axlen; //normalize
240  if (fabs(reflen-axlen)>0.1) {rpm=0;failed=true;}; //check for broken
241 
242  float warmupfactor=1.0;
243  if (warmup)
244  {
245  warmupfactor=(timer-warmupstart)/warmuptime;
246  if (warmupfactor>=1.0) warmup=false;
247  }
248 
249  //turbine RPM
250  //braking (compression)
251  if (rpm<0) rpm=0;
252  float torque=-rpm/100.0;
253  //powering with limiter
254  if (rpm<100.0 && !failed && ignition) torque+=((0.2+throtle*0.8)*warmupfactor);
255  //integration
256  rpm+=(double)dt*torque*30.0;
257 
258  float enginethrust=0;
259  if (!failed && ignition)
260  {
261  enginethrust=maxdrythrust*rpm/100.0;
262  afterburner=(afterburnable && throtle>0.95 && rpm>80);
263  if (afterburner) enginethrust+=(afterburnthrust-maxdrythrust);
264  } else afterburner=false;
265 #ifdef USE_OPENAL
266  if (afterburner)
267  SoundScriptManager::getSingleton().trigStart(trucknum, ab_id);
268  else
269  SoundScriptManager::getSingleton().trigStop(trucknum, ab_id);
270 #endif //OPENAL
271 
272  nodes[nodeback].Forces+=(enginethrust*1000.0)*axis;
273  exhaust_velocity=enginethrust*5.6/area;
274 }
