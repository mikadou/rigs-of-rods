02867 
02868     Matrix3 forward;
02869     Vector3 bx;
02870     Vector3 by;
02871     Vector3 bz;
02872     Vector3 forcevec;
02873     Vector3 plnormal;
02874     Vector3 point;
02875     Vector3 vecrelVel;
02876     float trwidth;
02877     int hitnodeid;
02878     int tmpv;
02879     node_t* hitnode;
02880     node_t* na;
02881     node_t* nb;
02882     node_t* no;
02883 
02884     trwidth=collrange;
02885 
02886     int chunk_size = free_collcab / chunk_number;
02887     int end_index = (chunk_index+1)*chunk_size;
02888 
02889     if (chunk_index+1 == chunk_number)
02890     {
02891         end_index = free_collcab;
02892     }
02893 
02894     for (int i=chunk_index*chunk_size; i<end_index; i++)
02895     {
02896         intra_collcabrate[i].update=true;
02897         if (intra_collcabrate[i].rate>0)
02898         {
02899             intra_collcabrate[i].rate--;
02900             intra_collcabrate[i].update=false;

02910             , na->AbsPosition

02962                     (no->Velocity*(-point.x-point.y+1.0f)+na->Velocity*point.x
02963                     +nb->Velocity*point.y));
02964 
02965                 //Find the velocity perpendicular to the triangle
02966                 float velForce=vecrelVel.dotProduct(plnormal);
02967                 //if it points away from the triangle the ignore it (set it to 0)
02968                 if (velForce<0.0f) velForce=-velForce;
02969                 else velForce=0.0f;
02970 
02971                 //Velocity impulse
02972                 float vi=hitnode->mass*inverted_dt*(velForce+inverted_dt*penetration)*0.5f;
02973 
02974                 //MUTEX_LOCK(&itc_node_access_mutex);

02977                     +nb->Forces*point.y).dotProduct(plnormal);
02978                 //(applied only when it is towards the point)
02979                 if (trfnormal<0.0f) trfnormal=0.0f;
02980 
02981                 //The force that the point puts on the triangle
02982                 float pfnormal=hitnode->Forces.dotProduct(plnormal);
02983                 //(applied only when it is towards the triangle)
02984                 if (pfnormal>0.0f) pfnormal=0.0f;
02985 
02986                 float fl=(vi+trfnormal-pfnormal)*0.5f;
02987

02989                 float nso;
02990 
02991                 //Calculate the collision forces
02992                 gEnv->collisions->primitiveCollision(hitnode, forcevec, vecrelVel, plnormal, ((float) dt), submesh_ground_model, &nso, penetration, fl);
02993 
02994                 hitnode->Forces+=forcevec;

03053 Quaternion Beam::specialGetRotationTo(const Vector3& src, const Vector3& dest) const
03054 {
03055     // Based on Stan Melax's article in Game Programming Gems
03056     Quaternion q;
03057     // Copy, since cannot modify local
03058     Vector3 v0 = src;
03059     Vector3 v1 = dest;
03060     v0.normalise();
03061     v1.normalise();
03062 
03063 
03064     // NB if the crossProduct approaches zero, we get unstable because ANY axis will do
03065     // when v0 == -v1
03066     Real d = v0.dotProduct(v1);
03067     // If dot == 1, vectors are the same
03068     if (d >= 1.0f)
03069     {
03070         return Quaternion::IDENTITY;
03071     }
03072     if (d < (1e-6f - 1.0f))
03073     {
03074         // Generate an axis
03075         Vector3 axis = Vector3::UNIT_X.crossProduct(src);
03076         if (axis.isZeroLength()) // pick another if colinear
03077             axis = Vector3::UNIT_Y.crossProduct(src);
03078         axis.normalise();
03079         q.FromAngleAxis(Radian(Math::PI), axis);
03080     }
03081     else
03082     {
03083         Real s = fast_sqrt( (1+d)*2 );
03084         if (s==0) return Quaternion::IDENTITY;
03085 
03086         Vector3 c = v0.crossProduct(v1);
03087         Real invs = 1 / s;
03088 
03089 
03090         q.x = c.x * invs;
03091         q.y = c.y * invs;
03092         q.z = c.z * invs;
03093         q.w = s * 0.5;
03094     }
03095     return q;
03096 }

04524 void Beam::hookToggle(int group, hook_states mode, int node_number)
04525 {
04526     Beam **trucks = BeamFactory::getSingleton().getTrucks();
04527     int trucksnum = BeamFactory::getSingleton().getTruckCount();
04528 
04529     // iterate over all hooks
04530     for (std::vector <hook_t>::iterator it = hooks.begin(); it!=hooks.end(); it++)
04531     {
04532         if (mode == MOUSE_HOOK_TOGGLE && it->hookNode->id != node_number)
04533         {
04534             //skip all other nodes except the one manually toggled by mouse
04535             continue;
04536         }
04537         if (mode == HOOK_TOGGLE && group == -1)
04538         {
04539             //manually triggerd (EV_COMMON_LOCK). Toggle all hooks groups with group#: -1, 0, 1 ++
04540             if (it->group <= -2)
04541                 continue;
04542         }
04543         if (mode == HOOK_LOCK && group == -2)
04544         {
04545             //automatic lock attempt (cyclic with doupdate). Toggle all hooks groups with group#: -2, -3, -4 --, skip the ones which are not autolock (triggered only)
04546             if (it->group >= -1 || !it->autolock)
04547                 continue;
04548         }
04549         if (mode == HOOK_UNLOCK && group == -2)
04550         {
04551             //manual unlock ALL autolock and triggerlock, do not unlock standard hooks (EV_COMMON_AUTOLOCK)
04552             if (it->group >= -1 || !it->autolock)
04553                 continue;
04554         }
04555         if ((mode == HOOK_LOCK || mode == HOOK_UNLOCK) && group <= -3)
04556         {
04557             //trigger beam lock or unlock. Toggle one hook group with group#: group
04558             if (it->group != group)
04559                 continue;
04560         }
04561         if ((mode == HOOK_LOCK || mode == HOOK_UNLOCK) && group >= -1)
04562         {
04563             continue;
04564         }
04565         if (mode == HOOK_LOCK && it->timer > 0.0f)
04566         {
04567             //check relock delay timer for autolock nodes and skip if not 0
04568             continue;
04569         }
04570 
04571         Beam* lastLockTruck = it->lockTruck; // memorize current value
04572 
04573         //this is a locked or prelocked hook and its not a locking attempt
04574         if ((it->locked == LOCKED || it->locked == PRELOCK) && mode != HOOK_LOCK)
04575         {
04576             // we unlock ropes
04577             it->locked = UNLOCKED;
04578             // remove node locking
04579             if (it->lockNode)
04580             {
04581                 it->lockNode->lockednode = 0;
04582             }
04583             if (it->group <= -2)
04584             {
04585                 it->timer = it->timer_preset;   //timer reset for autolock nodes
04586             }
04587             it->lockNode  = 0;
04588             it->lockTruck = 0;
04589             //disable hook-assistance beam
04590             it->beam->mSceneNode->detachAllObjects();
04591             it->beam->p2       = &nodes[0];
04592             it->beam->p2truck  = 0;
04593             it->beam->L        = (nodes[0].AbsPosition - it->hookNode->AbsPosition).length();
04594             it->beam->disabled = true;
04595         }
04596         // do this only for toggle or lock attempts, skip prelocked or locked nodes for performance
04597         else if (mode != HOOK_UNLOCK && it->locked == UNLOCKED)
04598         {
04599             // we lock hooks
04600             // search new remote ropable to lock to
04601             float mindist = it->lockrange;
04602             node_t *shorter=0;
04603             Beam *shtruck=0;
04604             float distance = 100000000.0f;
04605             // iterate over all trucks
04606             for (int t=0; t<trucksnum; t++)
04607             {
04608                 if (!trucks[t]) continue;
04609                 if (trucks[t]->state >= SLEEPING) continue;
04610                 if (t == this->trucknum && !it->selflock) continue; // don't lock to self
04611 
04612                 // do we lock against all nodes or just against ropables?
04613                 bool found = false;
04614                 if (it->lockNodes)
04615                 {
04616                     int last_node = 0; // node number storage
04617                     // all nodes, so walk them
04618                     for (int i=0; i<trucks[t]->free_node; i++)
04619                     {
04620                         // skip all nodes with lockgroup 9999 (deny lock)
04621                         if (trucks[t]->nodes[i].lockgroup == 9999)
04622                             continue;
04623 
04624                         // exclude this truck and its current hooknode from the locking search
04625                         if (this == trucks[t] && i == it->hookNode->id)
04626                             continue;
04627 
04628                         // a lockgroup for this hooknode is set -> skip all nodes that do not have the same lockgroup (-1 = default(all nodes))
04629                         if (it->lockgroup != -1 && it->lockgroup != trucks[t]->nodes[i].lockgroup)
04630                             continue;
04631 
04632                         // measure distance
04633                         float n2n_distance = (it->hookNode->AbsPosition - trucks[t]->nodes[i].AbsPosition).length();
04634                         if (n2n_distance < mindist)
04635                         {
04636                             if (distance >= n2n_distance)
04637                             {
04638                                 // located a node that is closer
04639                                 distance  = n2n_distance;
04640                                 last_node = i;
04641                                 found     = true;
04642                             }
04643                         }
04644                     }
04645                     if (found)
04646                     {
04647                         // we found a node, lock to it
04648                         it->lockNode  = &(trucks[t]->nodes[last_node]);
04649                         it->lockTruck = trucks[t];
04650                         it->locked    = PRELOCK;
04651                     }
04652                 } else
04653                 {
04654                     // we lock against ropables
04655 
04656                     // and their ropables
04657                     for (std::vector <ropable_t>::iterator itr = trucks[t]->ropables.begin(); itr!=trucks[t]->ropables.end(); itr++)
04658                     {
04659                         // if the ropable is not multilock and used, then discard this ropable
04660                         if (!itr->multilock && itr->used)
04661                             continue;
04662 
04663                         // calculate the distance and record the nearest ropable
04664                         float dist = (it->hookNode->AbsPosition - itr->node->AbsPosition).length();
04665                         if (dist < mindist)
04666                         {
04667                             mindist = dist;
04668                             shorter = itr->node;
04669                             shtruck = trucks[t];
04670                         }
04671                     }
04672                 }
04673                 // if we found a ropable, then lock it
04674                 if (shorter)
04675                 {
04676                     // we found a ropable, lock to it
04677                     it->lockNode  = shorter;
04678                     it->lockTruck = shtruck;
04679                     it->locked    = PRELOCK;
04680                 }
04681             }
04682         }
04683 
04684         if (it->lockTruck != lastLockTruck)
04685         {
04686             std::list<Beam*> linkedBeams (linkedBeams);
04687 
04688             linkedBeams.push_back(this);
04689             linkedBeams.push_back(std::max(lastLockTruck, it->lockTruck));
04690             linkedBeams.splice(linkedBeams.end(), linkedBeams.back()->linkedBeams);
04691             linkedBeams.sort();
04692             linkedBeams.unique();
04693 
04694             for (std::list<Beam*>::iterator it = linkedBeams.begin(); it != linkedBeams.end(); ++it)
04695             {
04696                 (*it)->determineLinkedBeams();
04697 
04698                 if (skeleton && this != (*it) && !(*it)->skeleton)
04699                 {
04700                     (*it)->showSkeleton(true, skeleton==2, false);
04701                 } else if (skeleton && this != (*it) && (*it)->skeleton)
04702                 {
04703                     (*it)->hideSkeleton(skeleton==2, false);
04704                 }
04705             }
04706         }
04707     }
04708 }

05787 Vector3 Beam::getGForces()
05788 {
05789     if (cameranodecount > 0 && cameranodepos[0] >= 0 && cameranodepos[0] < MAX_NODES && cameranodedir[0] >= 0 && cameranodedir[0] < MAX_NODES && cameranoderoll[0] >= 0 && cameranoderoll[0] < MAX_NODES)
05790     {
05791         Vector3 acc      = cameranodeacc / cameranodecount;
05792         cameranodeacc    = Vector3::ZERO;
05793         cameranodecount  = 0;
05794 
05795         float longacc     = acc.dotProduct((nodes[cameranodepos[0]].RelPosition - nodes[cameranodedir[0]].RelPosition).normalisedCopy());
05796         float latacc      = acc.dotProduct((nodes[cameranodepos[0]].RelPosition - nodes[cameranoderoll[0]].RelPosition).normalisedCopy());
05797         
05798         Vector3 diffdir  = nodes[cameranodepos[0]].RelPosition - nodes[cameranodedir[0]].RelPosition;
05799         Vector3 diffroll = nodes[cameranodepos[0]].RelPosition - nodes[cameranoderoll[0]].RelPosition;
05800         
05801         Vector3 upv = diffdir.crossProduct(-diffroll);
05802         upv.normalise();
05803 
05804         float gravity = DEFAULT_GRAVITY;
05805 
05806         if (gEnv->terrainManager)
05807         {
05808             gravity = gEnv->terrainManager->getGravity();
05809         }
05810 
05811         float vertacc = std::abs(gravity) - acc.dotProduct(-upv);
05812 
05813         return Vector3(vertacc / std::abs(gravity), longacc / std::abs(gravity), latacc / std::abs(gravity));
05814     }
05815 
05816     return Vector3::ZERO;
05817 }

05854 void Beam::runThreadTask(Beam* truck, ThreadTask task, bool shared /*= false*/)
05855 {
05856     ThreadTask old_thread_task = truck->thread_task;
05857 
05858     truck->thread_index = 0;
05859     truck->thread_number = 1;
05860     truck->thread_task = task;
05861 
05862     if (gEnv->threadPool && (!shared || gEnv->threadPool->getSize() / truck->num_simulated_trucks > 1))
05863     {
05864         truck->thread_number = gEnv->threadPool->getSize();
05865         if (shared)
05866         {
05867             truck->thread_number /= truck->num_simulated_trucks;
05868         }
05869         truck->task_count[task] = truck->thread_number;
05870 
05871         std::list<IThreadTask*> tasks;
05872 
05873         // Push tasks into thread pool
05874         for (int i=0; i<truck->thread_number; i++)
05875         {
05876             tasks.emplace_back(truck);
05877         }
05878 
05879         gEnv->threadPool->enqueue(tasks);
05880 
05881         // Wait for all tasks to complete
05882         MUTEX_LOCK(&truck->task_count_mutex[task]);
05883         while (truck->task_count[task] > 0)
05884         {
05885             pthread_cond_wait(&truck->task_count_cv[task], &truck->task_count_mutex[task]);
05886         }
05887         MUTEX_UNLOCK(&truck->task_count_mutex[task]);
05888     } else
05889     {
05890         truck->run();
05891     }
05892 
05893     truck->thread_task = old_thread_task;
05894 }
