00046 void LandVehicleWheel::BuildWheelNodes( 
00047     std::vector<Ogre::Vector3> & out_positions,
00048     unsigned int num_rays,
00049     Ogre::Vector3 const & axis_node_inner,
00050     Ogre::Vector3 const & axis_node_outer,
00051     Ogre::Vector3 const & reference_arm_node_pos,
00052     Ogre::Vector3* rigidity_node_ptr,
00053     float wheel_radius,
00054     Ogre::AxisAlignedBox & out_aabb
00055 )
00056 {
00057     // Prepare buffer & insert initial points
00058     out_positions.clear();
00059     out_positions.reserve(num_rays * 2);
00060     out_positions.push_back(axis_node_inner);
00061     out_positions.push_back(axis_node_outer);
00062     out_positions.push_back( (rigidity_node_ptr != nullptr) ? *rigidity_node_ptr : Ogre::Vector3::ZERO );
00063 
00064     // Find near attach
00065     Ogre::Real length_1 = axis_node_inner.distance(reference_arm_node_pos);
00066     Ogre::Real length_2 = axis_node_outer.distance(reference_arm_node_pos);
00067     Ogre::Vector3 const & near_attach = (length_1 < length_2) ? axis_node_inner : axis_node_outer;
00068 
00069     // Axis
00070     Ogre::Vector3 axis_vector = axis_node_outer - axis_node_inner;
00071     axis_vector.normalise();
00072 
00073     // Generated nodes
00074     Ogre::Vector3 ray_vector = axis_vector.perpendicular() * wheel_radius;
00075     Ogre::Quaternion ray_rotator = Ogre::Quaternion(Ogre::Degree(-360.0 / (num_rays * 2)), axis_vector);
00076     Ogre::AxisAlignedBox aabb;
00077     aabb.setNull();
00078 
00079     for (unsigned int i = 0; i < num_rays; i++)
00080     {
00081         // Outer ring
00082         Ogre::Vector3 pos = axis_node_inner + ray_vector;
00083         aabb.merge(pos);
00084         out_positions.push_back(pos);
00085         ray_vector = ray_rotator * ray_vector;
00086 
00087         // Inner ring
00088         pos = axis_node_outer + ray_vector;
00089         aabb.merge(pos);
00090         out_positions.push_back(pos);
00091         ray_vector = ray_rotator * ray_vector;
00092     }
00093     out_aabb = aabb;
00094 }
