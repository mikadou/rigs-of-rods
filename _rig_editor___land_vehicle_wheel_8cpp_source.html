46 void LandVehicleWheel::BuildWheelNodes(
47  std::vector<Ogre::Vector3> & out_positions,
48  unsigned int num_rays,
49  Ogre::Vector3 const & axis_node_inner,
50  Ogre::Vector3 const & axis_node_outer,
51  Ogre::Vector3 const & reference_arm_node_pos,
52  Ogre::Vector3* rigidity_node_ptr,
53  float wheel_radius,
54  Ogre::AxisAlignedBox & out_aabb
55 )
56 {
57  // Prepare buffer & insert initial points
58  out_positions.clear();
59  out_positions.reserve(num_rays * 2);
60  out_positions.push_back(axis_node_inner);
61  out_positions.push_back(axis_node_outer);
62  out_positions.push_back( (rigidity_node_ptr != nullptr) ? *rigidity_node_ptr : Ogre::Vector3::ZERO );
63 
64  // Find near attach
65  Ogre::Real length_1 = axis_node_inner.distance(reference_arm_node_pos);
66  Ogre::Real length_2 = axis_node_outer.distance(reference_arm_node_pos);
67  Ogre::Vector3 const & near_attach = (length_1 < length_2) ? axis_node_inner : axis_node_outer;
68 
69  // Axis
70  Ogre::Vector3 axis_vector = axis_node_outer - axis_node_inner;
71  axis_vector.normalise();
72 
73  // Generated nodes
74  Ogre::Vector3 ray_vector = axis_vector.perpendicular() * wheel_radius;
75  Ogre::Quaternion ray_rotator = Ogre::Quaternion(Ogre::Degree(-360.0 / (num_rays * 2)), axis_vector);
76  Ogre::AxisAlignedBox aabb;
77  aabb.setNull();
78 
79  for (unsigned int i = 0; i < num_rays; i++)
80  {
81  // Outer ring
82  Ogre::Vector3 pos = axis_node_inner + ray_vector;
83  aabb.merge(pos);
84  out_positions.push_back(pos);
85  ray_vector = ray_rotator * ray_vector;
86 
87  // Inner ring
88  pos = axis_node_outer + ray_vector;
89  aabb.merge(pos);
90  out_positions.push_back(pos);
91  ray_vector = ray_rotator * ray_vector;
92  }
93  out_aabb = aabb;
94 }

96 void LandVehicleWheel::RigidityNodeUpdateConnectivity()
97 {
98  // Find out where to connect rigidity node
99  if (m_rigidity_node != nullptr)
100  {
101  const Ogre::Vector3 rigidity_pos = m_rigidity_node->GetPosition();
102  const float distance_inner = rigidity_pos.squaredDistance(m_axis_inner_node->GetPosition()); // Squared is faster
103  const float distance_outer = rigidity_pos.squaredDistance(m_axis_inner_node->GetPosition());
104  m_rigidity_node_connect_outer_ring = distance_inner > distance_outer;
105  }
