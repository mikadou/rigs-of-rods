00029 #define SET_BIT(var, pos)   ((var) |= (1<<(pos)))
00030 #define CLEAR_BIT(var, pos) ((var) &= ~(1<<(pos)))

00057     void write()
00058     {
00059         win->update();
00060 
00061         // please do not misinterpret this feature
00062         // its used in the forums like an EXTIF data to display things beside the screenshots
00063         int mWidth  = win->getWidth();
00064         int mHeight = win->getHeight();
00065 
00066         // grab the image data
00067         Ogre::PixelFormat pf = Ogre::PF_B8G8R8; //win->suggestPixelFormat();
00068         long isize = mWidth * mHeight * (long)Ogre::PixelUtil::getNumElemBytes(pf);
00069         Ogre::uchar *data = OGRE_ALLOC_T(Ogre::uchar, isize, Ogre::MEMCATEGORY_RENDERSYS);
00070         Ogre::PixelBox pb(mWidth, mHeight, 1, pf, data);
00071         win->copyContentsToMemory(pb);
00072         
00073         // now do the fancy stuff ;)
00074 
00075         // 0. allocate enough buffer
00076         Ogre::uchar *databuf = OGRE_ALLOC_T(Ogre::uchar, 32768, Ogre::MEMCATEGORY_RENDERSYS);
00077         char *ptr = (char *)databuf;
00078         // header
00079         long dsize = 0;
00080         int w  = sprintf(ptr, "RORED\n");
00081         ptr   += w;
00082         dsize += w;
00083 
00084         // 1. convert the std::map to something properly
00085         for (std::map<Ogre::String, Ogre::String>::iterator it = map.begin(); it != map.end(); it++)
00086         {
00087             int w2 = sprintf(ptr, "%s:%s\n", it->first.c_str(), it->second.c_str());
00088             ptr   += w2;
00089             dsize += w2;
00090         }
00091 
00092         // now buffer ready, put it into the image
00093         Ogre::uchar *ptri = data;
00094         // set data pointer to the start again
00095         ptr = (char *)databuf;
00096         int bc = 7;
00097         for (long b = 0; b < isize && b < dsize * 8 + 40; b++, ptri++) // 40 = 5 zero bytes
00098         {
00099             if ( CHECK_BIT(*ptr, bc) )
00100                 SET_LSB(*ptri);
00101             else
00102                 CLEAR_LSB(*ptri);
00103             bc--;
00104             if ( bc < 0 )
00105             {
00106                 bc=7;
00107                 ptr++;
00108             }
00109         }
00110         //float used_per = ((float)(dsize * 8 + 40) / (float)isize) * 100.0f;
00111         //LOG("used " + TOSTRING(used_per) + " %");
00112 
00113         //save it
00114         Ogre::Image img;
00115         img.loadDynamicImage(data, mWidth, mHeight, 1, pf, false, 1, 0);
00116         img.save(filename);
00117 
00118         OGRE_FREE(data, Ogre::MEMCATEGORY_RENDERSYS);
00119         OGRE_FREE(databuf, Ogre::MEMCATEGORY_RENDERSYS);
00120     }
