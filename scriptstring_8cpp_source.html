708 // This is where we register the string type
709 void RegisterScriptString_Native(asIScriptEngine *engine)
710 {
711  int r;
712 
713  // Register the type
714  r = engine->RegisterObjectType("string", 0, asOBJ_REF); assert( r >= 0 );
715 
716  // Register the object operator overloads
717  // Note: We don't have to register the destructor, since the object uses reference counting
718  r = engine->RegisterObjectBehaviour("string", asBEHAVE_FACTORY, "string @f()", asFUNCTION(StringDefaultFactory), asCALL_CDECL); assert( r >= 0 );
