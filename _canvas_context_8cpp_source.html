501 void Context::arc(float x, float y, float r, float startRadian, float endRadian, bool anticlockwise)
502 {
503  SkScalar cx = x;
504  SkScalar cy = y;
505  SkScalar radius = r;
506  SkScalar sa = startRadian;
507  SkScalar ea = endRadian;
508 
509  SkRect oval;
510  oval.set(cx - radius, cy - radius, cx + radius, cy + radius);
511 
512  float sweep = ea - sa;
513  // check for a circle
514  if (sweep >= 2 * Ogre::Math::PI || sweep <= -2 * Ogre::Math::PI)
515  mPath.addOval(oval);
516  else {
517  SkScalar startDegrees = (sa * 180 / Ogre::Math::PI);
518  SkScalar sweepDegrees = (sweep * 180 / Ogre::Math::PI);
519 
520  // Counterclockwise arcs should be drawn with negative sweeps, while
521  // clockwise arcs should be drawn with positive sweeps. Check to see
522  // if the situation is reversed and correct it by adding or subtracting
523  // a full circle
524  if (anticlockwise && sweepDegrees > 0) {
525  sweepDegrees -= SkIntToScalar(360);
526  } else if (!anticlockwise && sweepDegrees < 0) {
527  sweepDegrees += SkIntToScalar(360);
528  }
529 
530  mPath.arcTo(oval, startDegrees, sweepDegrees, false);
531  }
532 }

659 Ogre::ColourValue ColourConverter::fromHexa(const std::string& text)
660 {
661  std::string c = text;
662  if (c.substr(0,1) == "#")
663  c = c.substr(1, c.length() - 1);
664 
665  int r,g,b;
666 
667  std::stringstream ss;
668 
669  if (c.length() == 3)
670  {
671  ss << std::setbase(16) << "0x" << c[0] << c[0];
672  ss >> r;
673  ss.clear();
674  ss << std::setbase(16) << "0x" << c[1] << c[1];
675  ss >> g;
676  ss.clear();
677  ss << std::setbase(16) << "0x" << c[2] << c[2];
678  ss >> b;
679  ss.clear();
680  }
681  else if (c.length() == 6)
682  {
683  ss << std::setbase(16) << "0x" << c[0] << c[1];
684  ss >> r;
685  ss.clear();
686  ss << std::setbase(16) << "0x" << c[2] << c[3];
687  ss >> g;
688  ss.clear();
689  ss << std::setbase(16) << "0x" << c[4] << c[5];
690  ss >> b;
691  ss.clear();
692  }
693  else
694  return Ogre::ColourValue::Black;
695 
696  return Ogre::ColourValue(r / 255.0f, g / 255.0f, b / 255.0f, 1);
697 }

743 void Gradient::createShader()
744 {
745  //FIXME : how is std::sort behaving with two identical values ?
746  //std::sort(mColorStops.begin(), mColorStops.end(), colorStopsSorting);
747 
748  SkPoint pts[2];
749  pts[0].set(mP0.x, mP0.y);
750  pts[1].set(mP1.x, mP1.y);
751 
752  SkColor* colors = new SkColor[mColorStops.size()];
753  SkScalar* pos = new SkScalar[mColorStops.size()];
754  for (unsigned int i=0; i<mColorStops.size(); ++i)
755  {
756  Ogre::ColourValue color = mColorStops[i].second;
757  colors[i] = SkColorSetARGB((U8CPU)(color.a*255), (U8CPU)(color.r*255), (U8CPU)(color.g*255), (U8CPU)(color.b*255));
758  pos[i] = mColorStops[i].first;
759  }
760 
761  if (mType == GradientType_Linear)
762  mShader = SkGradientShader::CreateLinear(pts, colors, pos, mColorStops.size(), SkShader::kClamp_TileMode);
763  else if (mType == GradientType_Radial)
764  mShader = SkGradientShader::CreateRadial(pts[1], mRadius1, colors, pos, mColorStops.size(), SkShader::kClamp_TileMode);
765  else
766  mShader = SkGradientShader::CreateTwoPointRadial(pts[0], mRadius0, pts[1], mRadius1, colors, pos, mColorStops.size(), SkShader::kClamp_TileMode);
767 }

780 Pattern::Pattern(const Ogre::Image& img, Repetition repeat)
781 {
782  SkBitmap bitmap;// = new SkBitmap();
783  bitmap.setConfig(SkBitmap::kARGB_8888_Config, img.getWidth(), img.getHeight());
784  bitmap.allocPixels();
785  bitmap.eraseARGB(0x00, 0x00, 0x00, 0x00);
786 
787  Ogre::PixelBox src = img.getPixelBox();
788  Ogre::PixelBox dst = Ogre::PixelBox(src.getWidth(), src.getHeight(), src.getDepth(), Ogre::PF_A8R8G8B8); //PF_A8B8G8R8
789 
790  dst.data = bitmap.getPixels();
791  Ogre::PixelUtil::bulkPixelConversion(src, dst);
792 
793  if (repeat == Repetition_Repeat)
794  mShader = SkShader::CreateBitmapShader(bitmap, SkShader::kRepeat_TileMode, SkShader::kRepeat_TileMode);
795  else
796  {
797  bool repeatX = repeat == Repetition_RepeatX;
798  bool repeatY = repeat == Repetition_RepeatY;
799  SkShader::TileMode tileModeX = repeatX ? SkShader::kRepeat_TileMode : SkShader::kClamp_TileMode;
800  SkShader::TileMode tileModeY = repeatY ? SkShader::kRepeat_TileMode : SkShader::kClamp_TileMode;
801  int expandW = repeatX ? 0 : 1;
802  int expandH = repeatY ? 0 : 1;
803 
804  SkBitmap bitmapTmp;
805  bitmapTmp.setConfig(bitmap.config(), bitmap.width() + expandW, bitmap.height() + expandH);
806  bitmapTmp.allocPixels();
807  bitmapTmp.eraseARGB(0x00, 0x00, 0x00, 0x00);
808  SkCanvas canvas(bitmapTmp);
809  canvas.drawBitmap(bitmap, 0, 0);
810  mShader = SkShader::CreateBitmapShader(bitmapTmp, tileModeX, tileModeY);
811  }
812 }
