00501 void Context::arc(float x, float y, float r, float startRadian, float endRadian, bool anticlockwise)
00502 {
00503     SkScalar cx = x;
00504     SkScalar cy = y;
00505     SkScalar radius = r;
00506     SkScalar sa = startRadian;
00507     SkScalar ea = endRadian;    
00508 
00509     SkRect oval;
00510     oval.set(cx - radius, cy - radius, cx + radius, cy + radius);
00511 
00512     float sweep = ea - sa;
00513     // check for a circle
00514     if (sweep >= 2 * Ogre::Math::PI || sweep <= -2 * Ogre::Math::PI)
00515         mPath.addOval(oval);
00516     else {
00517         SkScalar startDegrees = (sa * 180 / Ogre::Math::PI);
00518         SkScalar sweepDegrees = (sweep * 180 / Ogre::Math::PI);
00519 
00520         // Counterclockwise arcs should be drawn with negative sweeps, while
00521         // clockwise arcs should be drawn with positive sweeps. Check to see
00522         // if the situation is reversed and correct it by adding or subtracting
00523         // a full circle
00524         if (anticlockwise && sweepDegrees > 0) {
00525             sweepDegrees -= SkIntToScalar(360);
00526         } else if (!anticlockwise && sweepDegrees < 0) {
00527             sweepDegrees += SkIntToScalar(360);
00528         }
00529 
00530         mPath.arcTo(oval, startDegrees, sweepDegrees, false);
00531     }
00532 }
