265 void Character::update(float dt)
266 {
267  if (physicsEnabled && !remote)
268  {
269  // disable character movement when using the free camera mode or when the menu is opened
270  // TODO: check for menu being opened
271  if (gEnv->cameraManager && gEnv->cameraManager->gameControlsLocked()) return;
272 
273  // small hack: if not visible do not apply physics
274  Vector3 position = mCharacterNode->getPosition();
275 
276  // gravity force is always on
277  position.y += characterVSpeed * dt;
278  characterVSpeed += dt * -9.8f;
279 
280  // object contact
281  Vector3 position2 = position + Vector3(0.0f, 0.075f, 0.0f);
282  Vector3 position3 = position + Vector3(0.0f, 0.25, 0.0f);
283 
284  if (gEnv->collisions->collisionCorrect(&position))
285  {
286  characterVSpeed = std::max(0.0f, characterVSpeed);
287  if (gEnv->collisions->collisionCorrect(&position2) && !gEnv->collisions->collisionCorrect(&position3))
288  {
289  characterVSpeed = 2.0f; // autojump
290  } else
291  {
292  canJump = true;
293  }
294  } else
295  {
296  // double check
297  if (mLastPosition.squaredDistance(position) < 25.0f)
298  {
299  const int numstep = 100;
300  Vector3 dvec = (position - mLastPosition);
301  for (int i=1; i <= numstep; i++)
302  {
303  Vector3 cposition = mLastPosition + dvec * ((float)i / numstep);
304  if (gEnv->collisions->collisionCorrect(&cposition))
305  {
306  position = cposition;
307  characterVSpeed = std::max(0.0f, characterVSpeed);
308  canJump = true;
309  break;
310  }
311  }
312  }
313 
314  }
315 
316  mLastPosition = position;
317 
318  // ground contact
319  float pheight = gEnv->terrainManager->getHeightFinder()->getHeightAt(position.x,position.z);
320 
321  if (position.y < pheight)
322  {
323  position.y = pheight;
324  characterVSpeed = 0.0f;
325  canJump = true;
326  }
327 
328  // water stuff
329  bool isswimming = false;
330  float wheight = -99999;
331 
332  if (gEnv->terrainManager->getWater())
333  {
334  wheight = gEnv->terrainManager->getWater()->getHeightWaves(position);
335  if (position.y < wheight - 1.8f)
336  {
337  position.y = wheight - 1.8f;
338  characterVSpeed = 0.0f;
339  }
340  }
341 
342  // 0.1 due to 'jumping' from waves -> not nice looking
343  if (gEnv->terrainManager->getWater() && (wheight - pheight > 1.8f) && (position.y + 0.1f <= wheight))
344  {
345  isswimming = true;
346  }
347 
348  float tmpJoy = 0.0f;
349  if (canJump)
350  {
351  if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_CHARACTER_JUMP))
352  {
353  characterVSpeed = 2.0f;
354  canJump = false;
355  }
356  }
357 
358  bool idleanim = true;
359 
360  tmpJoy = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_RIGHT);
361  if (tmpJoy > 0.0f)
362  {
363  setRotation(characterRotation + dt * 2.0f * Radian(tmpJoy));
364  if (!isswimming)
365  {
366  setAnimationMode("Turn", -dt);
367  idleanim = false;
368  }
369  }
370 
371  tmpJoy = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_LEFT);
372  if (tmpJoy > 0.0f)
373  {
374  setRotation(characterRotation - dt * 2.0f * Radian(tmpJoy));
375  if (!isswimming)
376  {
377  setAnimationMode("Turn", dt);
378  idleanim = false;
379  }
380  }
381 
382  float tmpRun = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_RUN);
383  float accel = 1.0f;
384 
385  tmpJoy = accel = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_SIDESTEP_LEFT);
386  if (tmpJoy > 0.0f)
387  {
388  if (tmpRun > 0.0f) accel = 3.0f * tmpRun;
389  // animation missing for that
390  position += dt * characterSpeed *1.5f * accel * Vector3(cos(characterRotation.valueRadians() - Math::HALF_PI), 0.0f, sin(characterRotation.valueRadians() - Math::HALF_PI));
391  }
392 
393  tmpJoy = accel = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_SIDESTEP_RIGHT);
394  if (tmpJoy > 0.0f)
395  {
396  if (tmpRun > 0.0f) accel = 3.0f * tmpRun;
397  // animation missing for that
398  position += dt * characterSpeed * 1.5f * accel * Vector3(cos(characterRotation.valueRadians() + Math::HALF_PI), 0.0f, sin(characterRotation.valueRadians() + Math::HALF_PI));
399  }
400 
401  tmpJoy = accel = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_FORWARD) + RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_ROT_UP);
402  float tmpBack = RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_BACKWARDS) + RoR::Application::GetInputEngine()->getEventValue(EV_CHARACTER_ROT_DOWN);
403 
404  tmpJoy = std::min(tmpJoy, 1.0f);
405  tmpBack = std::min(tmpBack, 1.0f);
406 
407  if (tmpJoy > 0.0f || tmpRun > 0.0f)
408  {
409  if (tmpRun > 0.0f) accel = 3.0f * tmpRun;
410 
411  float time = dt * tmpJoy * characterSpeed;
412 
413  if (isswimming)
414  {
415  setAnimationMode("Swim_loop", time);
416  idleanim = false;
417  } else
418  {
419  if (tmpRun > 0.0f)
420  {
421  setAnimationMode("Run", time);
422  idleanim = false;
423  } else
424  {
425  setAnimationMode("Walk", time);
426  idleanim = false;
427  }
428  }
429  // 0.005f fixes character getting stuck on meshes
430  position += dt * characterSpeed * 1.5f * accel * Vector3(cos(characterRotation.valueRadians()), 0.01f, sin(characterRotation.valueRadians()));
431  } else if (tmpBack > 0.0f)
432  {
433  float time = -dt * characterSpeed;
434  if (isswimming)
435  {
436  setAnimationMode("Spot_swim", time);
437  idleanim = false;
438  } else
439  {
440  setAnimationMode("Walk", time);
441  idleanim = false;
442  }
443  // 0.005f fixes character getting stuck on meshes
444  position -= dt * characterSpeed * tmpBack * Vector3(cos(characterRotation.valueRadians()), 0.01f, sin(characterRotation.valueRadians()));
445  }
446 
447  if (idleanim)
448  {
449  if (isswimming)
450  {
451  setAnimationMode("Spot_swim", dt * 0.5f);
452  } else
453  {
454  setAnimationMode("Idle_sway", dt * 0.05f);
455  }
456  }
457 
458  mCharacterNode->setPosition(position);
459  updateMapIcon();
460  } else if (beamCoupling)
461  {
462  // if physics is disabled, its attached to a beam truck maybe?
463  Vector3 pos;
464  Quaternion rot;
465  float angle = beamCoupling->hydrodirwheeldisplay * -1.0f; // not getSteeringAngle(), but this, as its smoothed
466  int res = beamCoupling->calculateDriverPos(pos, rot);
467  if (!res)
468  {
469  setPosition(pos + rot * Vector3(0.0f,-0.6f,-0.1f)); // hack to position the character right perfect on the default seat
470  mCharacterNode->setOrientation(rot);
471  setAnimationMode("driving");
472  Real lenght = mAnimState->getAnimationState("driving")->getLength();
473  float timePos = ((angle + 1.0f) * 0.5f) * lenght;
474  //LOG("angle: " + TOSTRING(angle) + " / " + TOSTRING(timePos));
475  // prevent animation flickering on the borders:
476  if (timePos < 0.01f)
477  {
478  timePos = 0.01f;
479  }
480  if (timePos > lenght - 0.01f)
481  {
482  timePos = lenght - 0.01f;
483  }
484  mAnimState->getAnimationState("driving")->setTimePosition(timePos);
485  }
486 
487  }
488 
489 #ifdef USE_SOCKETW
490  if (gEnv->network && !remote)
491  {
492  sendStreamData();
493  }
494 #endif // USE_SOCKETW
495 }

561 {
562  if (type == MSG2_STREAM_DATA && this->source == source && this->streamid == streamid)
563  {
564  header_netdata_t *header = (header_netdata_t *)buffer;
565  if (header->command == CHARCMD_POSITION)
566  {
567  // position
568  pos_netdata_t *data = (pos_netdata_t *)buffer;
569  Vector3 pos(data->posx, data->posy, data->posz);
570  this->setPosition(pos);

572  mCharacterNode->setOrientation(rot);

574  } else if (header->command == CHARCMD_ATTACH)
