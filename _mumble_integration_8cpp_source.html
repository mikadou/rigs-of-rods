00054 void MumbleIntegration::initMumble()
00055 {
00056 #ifdef _WIN32
00057     HANDLE hMapObject = OpenFileMappingW(FILE_MAP_ALL_ACCESS, FALSE, L"MumbleLink");
00058     if (hMapObject == NULL)
00059         return;
00060 
00061     lm = (LinkedMem *) MapViewOfFile(hMapObject, FILE_MAP_ALL_ACCESS, 0, 0, sizeof(LinkedMem));
00062     if (lm == NULL)
00063     {
00064         CloseHandle(hMapObject);
00065         hMapObject = NULL;
00066         return;
00067     }
00068 #else
00069     char memname[256];
00070     snprintf(memname, 256, "/MumbleLink.%d", getuid());
00071 
00072     int shmfd = shm_open(memname, O_RDWR, S_IRUSR | S_IWUSR);
00073 
00074     if (shmfd < 0)
00075     {
00076         return;
00077     }
00078 
00079     lm = (LinkedMem *)(mmap(NULL, sizeof(struct LinkedMem), PROT_READ | PROT_WRITE, MAP_SHARED, shmfd,0));
00080 
00081     if (lm == (void *)(-1))
00082     {
00083         lm = NULL;
00084         return;
00085     }
00086 #endif // _WIN32
00087 }
