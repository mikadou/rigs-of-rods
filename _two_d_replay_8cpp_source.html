55 void TwoDReplay::recordFrame()
56 {
57  if (!file)
58  start();
59 
60  Beam *v = BeamFactory::getSingleton().getCurrentTruck();
61  if (!v) return;
62 
63  // write entry first
64  r2d_file_entry_t entry;
65  memset(&entry, 0, sizeof(entry));
66  entry.type = R2DTYPE_FRAME;
67  entry.rignum = v->trucknum;
68  entry.time_ms = Ogre::Root::getSingleton().getTimer()->getMilliseconds();
69  entry.physframe = BeamFactory::getSingleton().getPhysFrame();
70  fwrite(&entry, 1, sizeof(entry), file);
71 
72  // then the actual data
73  r2d_file_frame_t frame;
74  memset(&frame, 0, sizeof(frame));
75 
76  // input things first
77  frame.brake = v->brake;
78  frame.steering = v->hydrodircommand;
79  frame.pbrake = (v->parkingbrake > 0);
80  if (v->engine)
81  {
82  frame.accel = v->engine->getAcc();
83  frame.clutch = v->engine->getClutch();
84  frame.gear = v->engine->getGear();
85  frame.gearMode = v->engine->getAutoMode();
86  }
87  fwrite(&frame, 1, sizeof(frame), file);
88 
89  // then the wheels
90  r2d_file_wheel_frame_t wf;
91  for (int w=0; w < v->free_wheel; w++)
92  {
93  memset(&wf, 0, sizeof(wf));
94  Vector3 idir = v->wheels[w].refnode0->AbsPosition - v->wheels[w].refnode1->AbsPosition;
95  Vector3 mpos = v->wheels[w].refnode1->AbsPosition + idir * 0.5f;
96  float angle = atan2(idir.dotProduct(Vector3::UNIT_X), idir.dotProduct(-Vector3::UNIT_Z));
97  wf.wheel_pos[0] = mpos.x;
98  wf.wheel_pos[1] = mpos.y;
99  wf.wheel_pos[2] = mpos.z;
100  wf.wheel_direction = angle;
101  wf.wheel_slip = v->wheels[w].lastSlip;
102  wf.wheel_speed = v->wheels[w].speed;
103  fwrite(&wf, 1, sizeof(wf), file);
104  }
105 
106  // TODO: actually record any coupled/locked trucks
107 
108 }

110 void TwoDReplay::start()
111 {
112  if (file) return;
113  file = fopen("data.tdr", "w");
114 
115  // first: write header
116  r2d_file_header_t header;
117  memset(&header, 0, sizeof(header));
118  strncpy(header.fileformat, R2DFILEFORMAT, 8);
119  header.num_rigs = BeamFactory::getSingleton().getTruckCount();
120  fwrite(&header, 1, sizeof(header), file);
121 
122  // second: write entry for every truck
123  for (int t = 0; t < header.num_rigs; t++)
124  {
125  Beam *v = BeamFactory::getSingleton().getTruck(t);
126  r2d_file_rig_t rig;
127  memset(&rig, 0, sizeof(rig));
128  rig.num = t;
129  if (v)
130  {
131  strncpy(rig.name, v->getTruckName().c_str(), std::min<int>(v->getTruckName().size(), 255));
132  strncpy(rig.filename, v->getTruckFileName().c_str(), std::min<int>(v->getTruckFileName().size(), 255));
133  strncpy(rig.hash, v->getTruckHash().c_str(), std::min<int>(v->getTruckHash().size(), 255));
134  rig.num_wheels = v->free_wheel;
135  }
136  fwrite(&rig, 1, sizeof(rig), file);
137 
138  // write wheel setup
139  if (v && rig.num_wheels > 0)
140  {
141  r2d_file_wheel_setup_t ws;
142  for (int w=0; w < rig.num_wheels; w++)
143  {
144  memset(&ws, 0, sizeof(ws));
145  ws.width = v->wheels[w].width;
146  ws.radius = v->wheels[w].radius;
147  ws.type = v->wheels[w].propulsed;
148  fwrite(&ws, 1, sizeof(ws), file);
149  }
150  }
151  }
152 
153  // third: header done, now the data
154 }
