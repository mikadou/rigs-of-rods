221 void GUI_MainMenu::addUserToMenu(user_info_t &user)
222 {
223  int numTrucks = BeamFactory::getSingleton().getTruckCount();
224  Beam **trucks = BeamFactory::getSingleton().getTrucks();
225 
226  // now search the vehicles of that user together
227  std::vector<int> matches;
228  for (int j = 0; j < numTrucks; j++)
229  {
230  if (!trucks[j]) continue;
231 
232  if (trucks[j]->getSourceID() == user.uniqueid)
233  {
234  // match, found truck :)
235  matches.push_back(j);
236  }
237  }
238 
239  // now add this user to the list
240  {
241  MyGUI::UString userStr = "- " + convertToMyGUIString(getUserString(user, (int)matches.size()));
242  // finally add the user line
243  m_vehicles_menu_widget->addItem(userStr, MyGUI::MenuItemType::Normal, "USER_"+TOSTRING(user.uniqueid));
244 
245  // and add the vehicles below the user name
246  if (!matches.empty())
247  {
248  for (unsigned int j = 0; j < matches.size(); j++)
249  {
250  char tmp[512] = "";
251  sprintf(tmp, " + %s (%s)", trucks[matches[j]]->realtruckname.c_str(), trucks[matches[j]]->realtruckfilename.c_str());
252  MyGUI::UString vehName = convertToMyGUIString(ANSI_TO_UTF(tmp));
253  m_vehicles_menu_widget->addItem(vehName, MyGUI::MenuItemType::Normal, "TRUCK_"+TOSTRING(matches[j]));
254  }
255  }
256  }

259 void GUI_MainMenu::vehiclesListUpdate()
260 {
261  m_vehicles_menu_widget->removeAllItems();
262 
263  if (!gEnv->network)
264  {
265  // single player mode: add vehicles simply, no users
266  int numTrucks = BeamFactory::getSingleton().getTruckCount();
267  Beam **trucks = BeamFactory::getSingleton().getTrucks();
268 
269  // simple iterate through :)
270  for (int i = 0; i < numTrucks; i++)
271  {
272  if (!trucks[i]) continue;
273 
274  if (trucks[i]->hideInChooser) continue;
275 
276  char tmp[255] = {};
277  sprintf(tmp, "[%d] %s", i, trucks[i]->realtruckname.c_str());
278 
279  m_vehicles_menu_widget->addItem(String(tmp), MyGUI::MenuItemType::Normal, "TRUCK_"+TOSTRING(i));
280  }
281  } else
282  {
283  // sort the list according to the network users
284 
285  // add self first
286  user_info_t *local_user = gEnv->network->getLocalUserData();
287  addUserToMenu(*local_user);
288 
289  // get network clients
290  client_t c[MAX_PEERS];
291  gEnv->network->getClientInfos(c);
292  // iterate over them
293  for (int i = 0; i < MAX_PEERS; i++)
294  {
295  if (!c[i].used) continue;
296  addUserToMenu(c[i].user);
297  }
298  }
