608 void MyApp::recurseCopy(wxString sourceDir, wxString destinationDir)
609 {
610  if (!wxDir::Exists(sourceDir) || !wxDir::Exists(destinationDir)) return;
611 
612  wxDir srcd(sourceDir);
613  wxString src;
614  if (!srcd.GetFirst(&src)) return; //empty dir!

616  {

618  if (src.StartsWith(wxT("."))) continue;
619  //check if it id a directory
620  wxFileName tsfn=wxFileName(sourceDir, wxEmptyString);
621  tsfn.AppendDir(src);
622  if (wxDir::Exists(tsfn.GetPath()))
623  {
624  //this is a directory, create dest dir and recurse
625  wxFileName tdfn=wxFileName(destinationDir, wxEmptyString);
626  tdfn.AppendDir(src);

628  recurseCopy(tsfn.GetPath(), tdfn.GetPath());
629  }
630  else
631  {
632  //this is a file, copy file
633  tsfn=wxFileName(sourceDir, src);
634  wxFileName tdfn=wxFileName(destinationDir, src);
635  //policy to be defined

637  {
638  //file already exists, should we overwrite?

640  }

644  }
645  }

665

667  {
668  // access meta-data
669  wxString name = entry->GetName();
670  name = aTargetDir + wxFileName::GetPathSeparator() + name;
671 
672  // read 'zip' to access the entry's data
673  if (entry->IsDir())
674  {
675  int perm = entry->GetMode();
676  wxFileName::Mkdir(name, perm, wxPATH_MKDIR_FULL);
677  } else
678  {
679  zip.OpenEntry(*entry.get());
680  if (!zip.CanRead())
681  {
682  wxLogError(_T("Can not read zip entry '") + entry->GetName() + wxT("'."));
683  ret = false;
684  break;
685  }
686 
687  wxFileOutputStream file(name);
688 
689  if (!file)
690  {
691  wxLogError(_T("Can not create file '")+name+wxT("'."));
692  ret = false;
693  break;
694  }
695  zip.Read(file);
696  }
697  }
698  } while(false);
