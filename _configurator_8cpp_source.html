02313 {
02314     //RoR config
02315     Ogre::ImprovedConfigFile cfg;
02316     try
02317     {
02318         wxLogStatus(wxT("Loading RoR.cfg"));
02319 
02320         wxString rorcfg=app->UserPath + wxFileName::GetPathSeparator() + wxT("config") + wxFileName::GetPathSeparator() + wxT("RoR.cfg");
02321         wxLogStatus(wxT("reading from Config file: ") + rorcfg);
02322 
02323         // Don't trim whitespace
02324         cfg.load((const char *)rorcfg.mb_str(wxConvUTF8), "=:\t", false);
02325     } catch(...)
02326     {
02327         wxLogError(wxT("error loading RoR.cfg"));
02328         return false;
02329     }
02330 
02331     // load all settings into a map!
02332     Ogre::ConfigFile::SettingsIterator i = cfg.getSettingsIterator();
02333     Ogre::String svalue, sname;
02334     while (i.hasMoreElements())
02335     {
02336         sname = i.peekNextKey();
02337         Ogre::StringUtil::trim(sname);
02338         svalue = i.getNext();
02339         Ogre::StringUtil::trim(svalue);
02340         // filter out some things that shouldnt be in there (since we cannot use RoR normally anymore after those)
02341         if(sname == Ogre::String("Benchmark") || sname == Ogre::String("streamCacheGenerationOnly")|| sname == Ogre::String("regen-cache-only"))
02342             continue;
02343         settings[sname] = svalue;
02344     }
02345 
02346 
02347     // enforce default settings
02348     if(settings["Allow NVPerfHUD"] == "") settings["Allow NVPerfHUD"] = "No";
02349     if(settings["Floating-point mode"] == "") settings["Floating-point mode"] = "Fastest";
02350     if(settings["Colour Depth"] == "") settings["Colour Depth"] = "32";
02351 
02352     // then update the controls!
02353     updateSettingsControls();
02354     return false;
02355 }
