93 int TorqueCurve::processLine(Ogre::StringVector args, String model)
94 {
95  // if its just one arguments, it must be a known model
96  if (args.size() == 1)
97  return setTorqueModel(args[0]);
98 
99  // we only accept 2 arguments
100  if (args.size() != 2)
101  return 1;
102  // parse the data
103  float pointx = StringConverter::parseReal(args[0]);
104  float pointy = StringConverter::parseReal(args[1]);
105  Vector3 point = Vector3(pointx,pointy,0);
106 
107  // find the spline to attach the points
108  if (splines.find(model) == splines.end())
109  splines[model] = SimpleSpline();
110 
111  // attach the points to the spline
112  // LOG("curve "+model+" : " + TOSTRING(point));
113  splines[model].addPoint(point);
114 
115  // special case for custom model:
116  // we set it as active curve as well!
117  if (model == TorqueCurve::customModel)
118  setTorqueModel(TorqueCurve::customModel);
119 
120  return 0;
121 }

160 int TorqueCurve::spaceCurveEvenly(Ogre::SimpleSpline *spline)
161 {
162  if (!spline) return 2;
163 
164  SimpleSpline tmpSpline = *spline;
165  Real points = tmpSpline.getNumPoints();
166 
167  if (points > 1)
168  {
169  // clear the original spline, so it's prepared for the new corrected numbers
170  spline->clear();
171  Real minDistance = tmpSpline.getPoint(1).x - tmpSpline.getPoint(0).x;
172  // looking for the minimum distance (spacing) in the current spline
173  for (int i=2; i < points; i++)
174  {
175  Real distance = tmpSpline.getPoint(i).x - tmpSpline.getPoint(i - 1).x;
176  minDistance = std::min(distance, minDistance);
177  }
178  // the rpm points must be in an ascending order, as the points should be added at the end of the spline
179  if (minDistance < 0) return 1;
180  // first(smallest)- and last(greatest) rpm
181  Vector3 minPoint = tmpSpline.getPoint(0);
182  Vector3 maxPoint = tmpSpline.getPoint(points - 1);
183 
184  Real rpmPoint = minPoint.x;
185  int pointIndex = 1; // this is the index used to interpolate between the rpm points
186  while (rpmPoint <= maxPoint.x && pointIndex < points)
187  {
188  // if actual rpm is higher than point of the spline, proceed to the next point
189  if (rpmPoint > tmpSpline.getPoint(pointIndex).x)
190  pointIndex++;
191  // interpolate(linear)
192  Real newPoint = tmpSpline.getPoint(pointIndex - 1).y + (tmpSpline.getPoint(pointIndex).y - tmpSpline.getPoint(pointIndex - 1).y) /
193  (tmpSpline.getPoint(pointIndex).x - tmpSpline.getPoint(pointIndex - 1).x) * (rpmPoint-tmpSpline.getPoint(pointIndex - 1).x);
194  spline->addPoint(Vector3(rpmPoint, newPoint, 0));
195  rpmPoint += minDistance;
196  }
197  // if the last point is missing due the even spacing, we add the last point manually
198  // criterion is that it must be smaller than 1% of the maximum rpm.
199  if (spline->getPoint(spline->getNumPoints() - 1).x < maxPoint.x && (rpmPoint - maxPoint.x) < 0.01 * maxPoint.x)
200  {
201  spline->addPoint(Vector3(rpmPoint, maxPoint.y, 0));
202  }
203  }
204 
205  return 0;
206 }
