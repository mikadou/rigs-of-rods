00093 int TorqueCurve::processLine(Ogre::StringVector args, String model)
00094 {
00095     // if its just one arguments, it must be a known model
00096     if (args.size() == 1)
00097         return setTorqueModel(args[0]);
00098 
00099     // we only accept 2 arguments
00100     if (args.size() != 2)
00101         return 1;
00102     // parse the data
00103     float pointx = StringConverter::parseReal(args[0]);
00104     float pointy = StringConverter::parseReal(args[1]);
00105     Vector3 point = Vector3(pointx,pointy,0);
00106 
00107     // find the spline to attach the points
00108     if (splines.find(model) == splines.end())
00109         splines[model] = SimpleSpline();
00110 
00111     // attach the points to the spline
00112     // LOG("curve "+model+" : " + TOSTRING(point));
00113     splines[model].addPoint(point);
00114 
00115     // special case for custom model:
00116     // we set it as active curve as well!
00117     if (model == TorqueCurve::customModel)
00118         setTorqueModel(TorqueCurve::customModel);
00119 
00120     return 0;

00160 int TorqueCurve::spaceCurveEvenly(Ogre::SimpleSpline *spline)
00161 {
00162     if (!spline) return 2;
00163 
00164     SimpleSpline tmpSpline = *spline;
00165     Real points = tmpSpline.getNumPoints();
00166 
00167     if (points > 1)
00168     {
00169         // clear the original spline, so it's prepared for the new corrected numbers
00170         spline->clear();
00171         Real minDistance = tmpSpline.getPoint(1).x - tmpSpline.getPoint(0).x;
00172         // looking for the minimum distance (spacing) in the current spline
00173         for (int i=2; i < points; i++)
00174         {
00175             Real distance = tmpSpline.getPoint(i).x - tmpSpline.getPoint(i - 1).x;
00176             minDistance = std::min(distance, minDistance);
00177         }
00178         // the rpm points must be in an ascending order, as the points should be added at the end of the spline
00179         if (minDistance < 0) return 1;
00180         // first(smallest)- and last(greatest) rpm
00181         Vector3 minPoint = tmpSpline.getPoint(0);
00182         Vector3 maxPoint = tmpSpline.getPoint(points - 1);
00183 
00184         Real rpmPoint = minPoint.x;
00185         int pointIndex = 1; // this is the index used to interpolate between the rpm points
00186         while (rpmPoint <= maxPoint.x && pointIndex < points)
00187         {
00188             // if actual rpm is higher than point of the spline, proceed to the next point
00189             if (rpmPoint > tmpSpline.getPoint(pointIndex).x)
00190                 pointIndex++;
00191             // interpolate(linear)
00192             Real newPoint = tmpSpline.getPoint(pointIndex - 1).y + (tmpSpline.getPoint(pointIndex).y - tmpSpline.getPoint(pointIndex - 1).y) /
00193                 (tmpSpline.getPoint(pointIndex).x - tmpSpline.getPoint(pointIndex - 1).x) * (rpmPoint-tmpSpline.getPoint(pointIndex - 1).x);
00194             spline->addPoint(Vector3(rpmPoint, newPoint, 0));
00195             rpmPoint += minDistance;
00196         }
00197         // if the last point is missing due the even spacing, we add the last point manually
00198         // criterion is that it must be smaller than 1% of the maximum rpm.
00199         if (spline->getPoint(spline->getNumPoints() - 1).x < maxPoint.x && (rpmPoint - maxPoint.x) < 0.01 * maxPoint.x)
00200         {
00201             spline->addPoint(Vector3(rpmPoint, maxPoint.y, 0));
00202         }
00203     }
00204 
00205     return 0;
00206 }
