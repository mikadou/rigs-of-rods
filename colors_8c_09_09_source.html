116 /*
117  * IRC to [code] color conversion. Or strip.
118  */
119 static char * libirc_colorparser_irc2code (const char * source, int strip)
120 {
121  unsigned int mask = 0, destlen = 0;
122  char * destline = 0, *d = 0;
123  const char *p;
124  int current_color = 1, current_bg = 0;
125 
126  /*
127  * There will be two passes. First pass calculates the total length of
128  * the destination string. The second pass allocates memory for the string,
129  * and fills it.
130  */
131  while ( destline == 0 ) // destline will be set after the 2nd pass
132  {
133  if ( destlen > 0 )
134  {
135  // This is the 2nd pass; allocate memory.
136  if ( (destline = (char *)malloc (destlen)) == 0 )
137  return 0;
138 
139  d = destline;
140  }
141 
142  for ( p = source; *p; p++ )
143  {
144  switch (*p)
145  {
146  case 0x02: // bold
147  if ( strip )
148  continue;
149 
150  libirc_colorparser_applymask (&mask, &d, &destlen, LIBIRC_COLORPARSER_BOLD, "[B]", "[/B]");
151  break;
152 
153  case 0x1F: // underline
154  if ( strip )
155  continue;
156 
157  libirc_colorparser_applymask (&mask, &d, &destlen, LIBIRC_COLORPARSER_UNDERLINE, "[U]", "[/U]");
158  break;
159 
160  case 0x16: // reverse
161  if ( strip )
162  continue;
163 
164  libirc_colorparser_applymask (&mask, &d, &destlen, LIBIRC_COLORPARSER_REVERSE, "[I]", "[/I]");
165  break;
166 
167  case 0x0F: // reset colors
168  if ( strip )
169  continue;
170 
171  libirc_colorparser_closetags (&mask, &d, &destlen);
172  break;
173 
174  case 0x03: // set color
175  if ( isdigit (p[1]) )
176  {
177  // Parse
178  int bgcolor = -1, color = p[1] - 0x30;
179  p++;
180 
181  if ( isdigit (p[1]) )
182  {
183  color = color * 10 + (p[1] - 0x30);
184  p++;
185  }
186 
187  // If there is a comma, search for the following
188  // background color
189  if ( p[1] == ',' && isdigit (p[2]) )
190  {
191  bgcolor = p[2] - 0x30;
192  p += 2;
193 
194  if ( isdigit (p[1]) )
195  {
196  bgcolor = bgcolor * 10 + (p[1] - 0x30);
197  p++;
198  }
199  }
200 
201  // Check for range
202  if ( color <= LIBIRC_COLORPARSER_MAXCOLORS
203  && bgcolor <= LIBIRC_COLORPARSER_MAXCOLORS )
204  {
205  if ( strip )
206  continue;
207 
208  current_color = color;
209 
210  if ( bgcolor != -1 )
211  current_bg = bgcolor;
212 
213  libirc_colorparser_applycolor (&mask, &d, &destlen, color, current_bg);
214  }
215  }
216  break;
217 
218  default:
219  if ( destline )
220  *d++ = *p;
221  else
222  destlen++;
223  break;
224  }
225  }
226 
227  // Close all the opened tags
228  libirc_colorparser_closetags (&mask, &d, &destlen);
229  destlen++; // for 0-terminator
230  }
231 
232  *d = '\0';
233  return destline;
234 }

251 char * irc_color_convert_to_mirc (const char * source)
252 {
253  unsigned int destlen = 0;
254  char * destline = 0, *d = 0;
255  const char *p1, *p2, *cur;
256 
257  /*
258  * There will be two passes. First pass calculates the total length of
259  * the destination string. The second pass allocates memory for the string,
260  * and fills it.
261  */
262  while ( destline == 0 ) // destline will be set after the 2nd pass
263  {
264  if ( destlen > 0 )
265  {
266  // This is the 2nd pass; allocate memory.
267  if ( (destline = (char *)malloc (destlen)) == 0 )
268  return 0;
269 
270  d = destline;
271  }
272 
273  cur = source;
274  while ( (p1 = strchr (cur, '[')) != 0 )
275  {
276  const char * replacedval = 0;
277  p2 = 0;
278 
279  // Check if the closing bracket is available after p1
280  // and the tag length is suitable
281  if ( p1[1] != '\0'
282  && (p2 = strchr (p1, ']')) != 0
283  && (p2 - p1) > 1
284  && (p2 - p1) < 31 )
285  {
286  // Get the tag
287  char tagbuf[32];
288  int taglen = p2 - p1 - 1;
289 
290  memcpy (tagbuf, p1 + 1, taglen);
291  tagbuf[taglen] = '\0';
292 
293  if ( !strcmp (tagbuf, "/COLOR") )
294  replacedval = "\x0F";
295  else if ( strstr (tagbuf, "COLOR=") == tagbuf )
296  {
297  int color, bgcolor = -2;
298  char * bcol;
299 
300  bcol = strchr (tagbuf + 6, '/');
301 
302  if ( bcol )
303  {
304  *bcol++ = '\0';
305  bgcolor = libirc_colorparser_colorlookup (bcol);
306  }
307 
308  color = libirc_colorparser_colorlookup (tagbuf + 6);
309 
310  if ( color != -1 && bgcolor == -2 )
311  {
312  sprintf (tagbuf, "\x03%02d", color);
313  replacedval = tagbuf;
314  }
315  else if ( color != -1 && bgcolor >= 0 )
316  {
317  sprintf (tagbuf, "\x03%02d,%02d", color, bgcolor);
318  replacedval = tagbuf;
319  }
320  }
321  else if ( !strcmp (tagbuf, "B") || !strcmp (tagbuf, "/B") )
322  replacedval = "\x02";
323  else if ( !strcmp (tagbuf, "U") || !strcmp (tagbuf, "/U") )
324  replacedval = "\x1F";
325  else if ( !strcmp (tagbuf, "I") || !strcmp (tagbuf, "/I") )
326  replacedval = "\x16";
327  }
328 
329  if ( replacedval )
330  {
331  // add a part before the tag
332  int partlen = p1 - cur;
333 
334  if ( destline )
335  {
336  memcpy (d, cur, partlen);
337  d += partlen;
338  }
339  else
340  destlen += partlen;
341 
342  // Add the replacement
343  libirc_colorparser_addorcat (&d, &destlen, replacedval);
344 
345  // And move the pointer
346  cur = p2 + 1;
347  }
348  else
349  {
350  // add a whole part before the end tag
351  int partlen;
352 
353  if ( !p2 )
354  p2 = cur + strlen(cur);
355 
356  partlen = p2 - cur + 1;
357 
358  if ( destline )
359  {
360  memcpy (d, cur, partlen);
361  d += partlen;
362  }
363  else
364  destlen += partlen;
365 
366  // And move the pointer
367  cur = p2 + 1;
368  }
369  }
370 
371  // Add the rest of string
372  libirc_colorparser_addorcat (&d, &destlen, cur);
373  destlen++; // for 0-terminator
374  }
375 
376  *d = '\0';
377  return destline;
378 }
