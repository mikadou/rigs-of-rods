00366 int irc_add_select_descriptors (irc_session_t * session, fd_set *in_set, fd_set *out_set, int * maxfd)
00367 {
00368     if ( session->sock < 0
00369     || session->state == LIBIRC_STATE_INIT
00370     || session->state == LIBIRC_STATE_DISCONNECTED )
00371     {
00372         session->lasterror = LIBIRC_ERR_STATE;
00373         return 1;
00374     }
00375 
00376     libirc_mutex_lock (&session->mutex_session);
00377 
00378     switch (session->state)
00379     {
00380     case LIBIRC_STATE_CONNECTING:
00381         // While connection, only out_set descriptor should be set
00382         libirc_add_to_set (session->sock, out_set, maxfd);
00383         break;
00384 
00385     case LIBIRC_STATE_CONNECTED:
00386         // Add input descriptor if there is space in input buffer
00387         if ( session->incoming_offset < (sizeof (session->incoming_buf) - 1) )
00388             libirc_add_to_set (session->sock, in_set, maxfd);
00389 
00390         // Add output descriptor if there is something in output buffer
00391         if ( libirc_findcrlf (session->outgoing_buf, session->outgoing_offset) > 0 )
00392             libirc_add_to_set (session->sock, out_set, maxfd);
00393 
00394         break;
00395     }
00396 
00397     libirc_mutex_unlock (&session->mutex_session);
00398 
00399     libirc_dcc_add_descriptors (session, in_set, out_set, maxfd);
00400     return 0;

00408     const char * command = 0, *prefix = 0, *params[MAX_PARAMS_ALLOWED+1];

00410 
00411     if ( process_length > sizeof(buf) )
00412         abort(); // should be impossible
00413 
00414     memcpy (buf, session->incoming_buf, process_length);

00416 
00417     memset ((char *)params, 0, sizeof(params));
00418     p = buf;
00419 
00420     /*
00421      * From RFC 1459:
00422      *  <message>  ::= [':' <prefix> <SPACE> ] <command> <params> <crlf>
00423      *  <prefix>   ::= <servername> | <nick> [ '!' <user> ] [ '@' <host> ]
00424      *  <command>  ::= <letter> { <letter> } | <number> <number> <number>
00425      *  <SPACE>    ::= ' ' { ' ' }
00426      *  <params>   ::= <SPACE> [ ':' <trailing> | <middle> <params> ]
00427      *  <middle>   ::= <Any *non-empty* sequence of octets not including SPACE
00428      *                 or NUL or CR or LF, the first of which may not be ':'>
00429      *  <trailing> ::= <Any, possibly *empty*, sequence of octets not including
00430      *                   NUL or CR or LF>
00431      */
00432 
00433     // Parse <prefix>
00434     if ( buf[0] == ':' )
00435     {
00436         while ( *p && *p != ' ')
00437             p++;
00438 
00439         *p++ = '\0';
00440 
00441         // we use buf+1 to skip the leading colon
00442         prefix = buf + 1;
00443 
00444         // If LIBIRC_OPTION_STRIPNICKS is set, we should 'clean up' nick
00445         // right here
00446         if ( session->options & LIBIRC_OPTION_STRIPNICKS )

00449             {
00450                 if ( *s == '@' || *s == '!' )
00451                 {
