326 int irc_run (irc_session_t * session)
327 {
328  if ( session->state != LIBIRC_STATE_CONNECTING )
329  {
330  session->lasterror = LIBIRC_ERR_STATE;
331  return 1;
332  }
333 
334  while ( irc_is_connected(session) )
335  {
336  struct timeval tv;
337  fd_set in_set, out_set;
338  int maxfd = 0;
339 
340  tv.tv_usec = 250000;
341  tv.tv_sec = 0;
342 
343  // Init sets
344  FD_ZERO (&in_set);
345  FD_ZERO (&out_set);
346 
347  irc_add_select_descriptors (session, &in_set, &out_set, &maxfd);
348 
349  if ( select (maxfd + 1, &in_set, &out_set, 0, &tv) < 0 )
350  {
351  if ( socket_error() == EINTR )
352  continue;
353 
354  session->lasterror = LIBIRC_ERR_TERMINATED;
355  return 1;
356  }
357 
358  if ( irc_process_select_descriptors (session, &in_set, &out_set) )
359  return 1;
360  }
361 
362  return 0;
363 }

366 int irc_add_select_descriptors (irc_session_t * session, fd_set *in_set, fd_set *out_set, int * maxfd)
367 {
368  if ( session->sock < 0
369  || session->state == LIBIRC_STATE_INIT
370  || session->state == LIBIRC_STATE_DISCONNECTED )
371  {
372  session->lasterror = LIBIRC_ERR_STATE;
373  return 1;
374  }
375 
376  libirc_mutex_lock (&session->mutex_session);
377 
378  switch (session->state)
379  {
380  case LIBIRC_STATE_CONNECTING:
381  // While connection, only out_set descriptor should be set
382  libirc_add_to_set (session->sock, out_set, maxfd);
383  break;
384 
385  case LIBIRC_STATE_CONNECTED:
386  // Add input descriptor if there is space in input buffer
387  if ( session->incoming_offset < (sizeof (session->incoming_buf) - 1) )
388  libirc_add_to_set (session->sock, in_set, maxfd);
389 
390  // Add output descriptor if there is something in output buffer
391  if ( libirc_findcrlf (session->outgoing_buf, session->outgoing_offset) > 0 )
392  libirc_add_to_set (session->sock, out_set, maxfd);
393 
394  break;
395  }
396 
397  libirc_mutex_unlock (&session->mutex_session);
398 
399  libirc_dcc_add_descriptors (session, in_set, out_set, maxfd);
400  return 0;
401 }

404 static void libirc_process_incoming_data (irc_session_t * session, int process_length)
405 {
406  #define MAX_PARAMS_ALLOWED 10
407  char buf[2*512], *p, *s;
408  const char * command = 0, *prefix = 0, *params[MAX_PARAMS_ALLOWED+1];
409  int code = 0, paramindex = 0;
410 
411  if ( process_length > sizeof(buf) )
412  abort(); // should be impossible
413 
414  memcpy (buf, session->incoming_buf, process_length);
415  buf[process_length] = '\0';
416 
417  memset ((char *)params, 0, sizeof(params));
418  p = buf;
419 
420  /*
421  * From RFC 1459:
422  * <message> ::= [':' <prefix> <SPACE> ] <command> <params> <crlf>
423  * <prefix> ::= <servername> | <nick> [ '!' <user> ] [ '@' <host> ]
424  * <command> ::= <letter> { <letter> } | <number> <number> <number>
425  * <SPACE> ::= ' ' { ' ' }
426  * <params> ::= <SPACE> [ ':' <trailing> | <middle> <params> ]
427  * <middle> ::= <Any *non-empty* sequence of octets not including SPACE
428  * or NUL or CR or LF, the first of which may not be ':'>
429  * <trailing> ::= <Any, possibly *empty*, sequence of octets not including
430  * NUL or CR or LF>
431  */
432 
433  // Parse <prefix>
434  if ( buf[0] == ':' )
435  {
436  while ( *p && *p != ' ')
437  p++;
438 
439  *p++ = '\0';
440 
441  // we use buf+1 to skip the leading colon
442  prefix = buf + 1;
443 
444  // If LIBIRC_OPTION_STRIPNICKS is set, we should 'clean up' nick
445  // right here
446  if ( session->options & LIBIRC_OPTION_STRIPNICKS )
447  {
448  for ( s = buf + 1; *s; s++ )
449  {
450  if ( *s == '@' || *s == '!' )
451  {
452  *s = '\0';
453  break;
454  }
455  }
456  }
457  }
458 
459  // Parse <command>
460  if ( isdigit (p[0]) && isdigit (p[1]) && isdigit (p[2]) )
461  {
462  p[3] = '\0';
463  code = atoi (p);
464  p += 4;
465  }
466  else
467  {
468  s = p;
469 
470  while ( *p && *p != ' ')
471  p++;
472 
473  *p++ = '\0';
474 
475  command = s;
476  }
477 
478  // Parse middle/params
479  while ( *p && paramindex < MAX_PARAMS_ALLOWED )
480  {
481  // beginning from ':', this is the last param
482  if ( *p == ':' )
483  {
484  params[paramindex++] = p + 1; // skip :
485  break;
486  }
487 
488  // Just a param
489  for ( s = p; *p && *p != ' '; p++ )
490  ;
491 
492  params[paramindex++] = s;
493 
494  if ( !*p )
495  break;
496 
497  *p++ = '\0';
498  }
499 
500  // Handle PING/PONG
501  if ( command && !strcmp (command, "PING") && params[0] )
502  {
503  irc_send_raw (session, "PONG %s", params[0]);
504  return;
505  }
506 
507  // and dump
508  if ( code )
509  {
510  // We use session->motd_received to check whether it is the first
511  // RPL_ENDOFMOTD or ERR_NOMOTD after the connection.
512  if ( (code == 376 || code == 422) && !session->motd_received )
513  {
514  session->motd_received = 1;
515 
516  if ( session->callbacks.event_connect )
517  (*session->callbacks.event_connect) (session, "CONNECT", prefix, params, paramindex);
518  }
519 
520  if ( session->callbacks.event_numeric )
521  (*session->callbacks.event_numeric) (session, code, prefix, params, paramindex);
522  }
523  else
524  {
525  if ( command && !strcmp (command, "NICK") )
526  {
527  /*
528  * If we're changed our nick, we should save it.
529  */
530  char nickbuf[256];
531 
532  irc_target_get_nick (prefix, nickbuf, sizeof(nickbuf));
533 
534  if ( !strcmp (nickbuf, session->nick) && paramindex > 0 )
535  {
536  free (session->nick);
537  session->nick = strdup (params[0]);
538  }
539 
540  if ( session->callbacks.event_nick )
541  (*session->callbacks.event_nick) (session, command, prefix, params, paramindex);
542  }
543  else if ( !strcmp (command, "QUIT") )
544  {
545  if ( session->callbacks.event_quit )
546  (*session->callbacks.event_quit) (session, command, prefix, params, paramindex);
547  }
548  else if ( !strcmp (command, "JOIN") )
549  {
550  if ( session->callbacks.event_join )
551  (*session->callbacks.event_join) (session, command, prefix, params, paramindex);
552  }
553  else if ( !strcmp (command, "PART") )
554  {
555  if ( session->callbacks.event_part )
556  (*session->callbacks.event_part) (session, command, prefix, params, paramindex);
557  }
558  else if ( !strcmp (command, "MODE") )
559  {
560  if ( paramindex > 0 && !strcmp (params[0], session->nick) )
561  {
562  params[0] = params[1];
563  paramindex = 1;
564 
565  if ( session->callbacks.event_umode )
566  (*session->callbacks.event_umode) (session, command, prefix, params, paramindex);
567  }
568  else
569  {
570  if ( session->callbacks.event_mode )
571  (*session->callbacks.event_mode) (session, command, prefix, params, paramindex);
572  }
573  }
574  else if ( !strcmp (command, "TOPIC") )
575  {
576  if ( session->callbacks.event_topic )
577  (*session->callbacks.event_topic) (session, command, prefix, params, paramindex);
578  }
579  else if ( !strcmp (command, "KICK") )
580  {
581  if ( session->callbacks.event_kick )
582  (*session->callbacks.event_kick) (session, command, prefix, params, paramindex);
583  }
584  else if ( !strcmp (command, "PRIVMSG") )
585  {
586  if ( paramindex > 1 )
587  {
588  unsigned int msglen = strlen (params[1]);
589 
590  /*
591  * Check for CTCP request (a CTCP message starts from 0x01
592  * and ends by 0x01
593  */
594  if ( params[1][0] == 0x01 && params[1][msglen-1] == 0x01 )
595  {
596  char ctcp_buf[128];
597 
598  msglen -= 2;
599  if ( msglen > sizeof(ctcp_buf) - 1 )
600  msglen = sizeof(ctcp_buf) - 1;
601 
602  memcpy (ctcp_buf, params[1] + 1, msglen);
603  ctcp_buf[msglen] = '\0';
604 
605  if ( strstr(ctcp_buf, "DCC ") == ctcp_buf )
606  libirc_dcc_request (session, prefix, ctcp_buf);
607  else if ( strstr(ctcp_buf, "ACTION ") == ctcp_buf
608  && session->callbacks.event_ctcp_action )
609  {
610  params[0] = ctcp_buf + 7; // the length of "ACTION "
611  paramindex = 2;
612 
613  (*session->callbacks.event_ctcp_action) (session, "ACTION", prefix, params, paramindex);
614  }
615  else
616  {
617  params[0] = ctcp_buf;
618  paramindex = 1;
619 
620  if ( session->callbacks.event_ctcp_req )
621  (*session->callbacks.event_ctcp_req) (session, "CTCP", prefix, params, paramindex);
622  }
623  }
624  else if ( !strcmp (params[0], session->nick) )
625  {
626  if ( session->callbacks.event_privmsg )
627  (*session->callbacks.event_privmsg) (session, command, prefix, params, paramindex);
628  }
629  else
630  {
631  if ( session->callbacks.event_channel )
632  (*session->callbacks.event_channel) (session, command, prefix, params, paramindex);
633  }
634  }
635  }
636  else if ( !strcmp (command, "NOTICE") )
637  {
638  unsigned int msglen = strlen (params[1]);
639 
640  /*
641  * Check for CTCP request (a CTCP message starts from 0x01
642  * and ends by 0x01
643  */
644  if ( paramindex > 1 && params[1][0] == 0x01 && params[1][msglen-1] == 0x01 )
645  {
646  char ctcp_buf[512];
647 
648  msglen -= 2;
649  if ( msglen > sizeof(ctcp_buf) - 1 )
650  msglen = sizeof(ctcp_buf) - 1;
651 
652  memcpy (ctcp_buf, params[1] + 1, msglen);
653  ctcp_buf[msglen] = '\0';
654 
655  params[0] = ctcp_buf;
656  paramindex = 1;
657 
658  if ( session->callbacks.event_ctcp_rep )
659  (*session->callbacks.event_ctcp_rep) (session, "CTCP", prefix, params, paramindex);
660  }
661  else
662  {
663  if ( session->callbacks.event_notice )
664  (*session->callbacks.event_notice) (session, command, prefix, params, paramindex);
665  }
666  }
667  else if ( !strcmp (command, "INVITE") )
668  {
669  if ( session->callbacks.event_invite )
670  (*session->callbacks.event_invite) (session, command, prefix, params, paramindex);
671  }
672  else if ( !strcmp (command, "KILL") )
673  {
674  ; /* ignore this event - not all servers generate this */
675  }
676  else
677  {
678  /*
679  * The "unknown" event is triggered upon receipt of any number of
680  * unclassifiable miscellaneous messages, which aren't handled by
681  * the library.
682  */
683 
684  if ( session->callbacks.event_unknown )
685  (*session->callbacks.event_unknown) (session, command, prefix, params, paramindex);
686  }
687  }

691 int irc_process_select_descriptors (irc_session_t * session, fd_set *in_set, fd_set *out_set)
692 {
693  char buf[256], hname[256];
694 
695  if ( session->sock < 0
696  || session->state == LIBIRC_STATE_INIT
697  || session->state == LIBIRC_STATE_DISCONNECTED )
698  {
699  session->lasterror = LIBIRC_ERR_STATE;
700  return 1;
701  }
702 
703  libirc_dcc_process_descriptors (session, in_set, out_set);
704 
705  // Handle "connection succeed" / "connection failed"
706  if ( session->state == LIBIRC_STATE_CONNECTING
707  && FD_ISSET (session->sock, out_set) )
708  {
709  // Now we have to determine whether the socket is connected
710  // or the connect is failed
711  struct sockaddr_storage saddr, laddr;
712  socklen_t slen = sizeof(saddr);
713  socklen_t llen = sizeof(laddr);
714 
715  if ( getsockname (session->sock, (struct sockaddr*)&laddr, &llen) < 0
716  || getpeername (session->sock, (struct sockaddr*)&saddr, &slen) < 0 )
717  {
718  // connection failed
719  session->lasterror = LIBIRC_ERR_CONNECT;
720  session->state = LIBIRC_STATE_DISCONNECTED;
721  return 1;
722  }
723 
724  if (saddr.ss_family == AF_INET)
725  memcpy (&session->local_addr, &((struct sockaddr_in *)&laddr)->sin_addr, sizeof(struct in_addr));
726  else
727  memcpy (&session->local_addr, &((struct sockaddr_in6 *)&laddr)->sin6_addr, sizeof(struct in6_addr));
728 
729 #if defined (ENABLE_DEBUG)
730  if ( IS_DEBUG_ENABLED(session) )
731  fprintf (stderr, "[DEBUG] Detected local address: %s\n", inet_ntoa(session->local_addr));
732 #endif
733 
734  session->state = LIBIRC_STATE_CONNECTED;
735 
736  // Get the hostname
737  if ( gethostname (hname, sizeof(hname)) < 0 )
738  strcpy (hname, "unknown");
739 
740  // Prepare the data, which should be sent to the server
741  if ( session->server_password )
742  {
743  snprintf (buf, sizeof(buf), "PASS %s", session->server_password);
744  irc_send_raw (session, buf);
745  }
746 
747  snprintf (buf, sizeof(buf), "NICK %s", session->nick);
748  irc_send_raw (session, buf);
749 
750  /*
751  * RFC 1459 states that "hostname and servername are normally
752  * ignored by the IRC server when the USER command comes from
753  * a directly connected client (for security reasons)", therefore
754  * we don't need them.
755  */
756  snprintf (buf, sizeof(buf), "USER %s unknown unknown :%s",
757  session->username ? session->username : "nobody",
758  session->realname ? session->realname : "noname");
759  irc_send_raw (session, buf);
760 
761  return 0;
762  }
763 
764  if ( session->state != LIBIRC_STATE_CONNECTED )
765  return 1;
766 
767  // Hey, we've got something to read!
768  if ( FD_ISSET (session->sock, in_set) )
769  {
770  int length, offset;
771 
772  unsigned int amount = (sizeof (session->incoming_buf) - 1) - session->incoming_offset;
773  length = socket_recv (&session->sock, session->incoming_buf + session->incoming_offset, amount);
774 
775  if ( length <= 0 )
776  {
777  session->lasterror = (length == 0 ? LIBIRC_ERR_CLOSED : LIBIRC_ERR_TERMINATED);
778  session->state = LIBIRC_STATE_DISCONNECTED;
779  return 1;
780  }
781 
782  session->incoming_offset += length;
783 
784  // process the incoming data
785  while ( (offset = libirc_findcrlf (session->incoming_buf, session->incoming_offset)) > 0 )
786  {
787 #if defined (ENABLE_DEBUG)
788  if ( IS_DEBUG_ENABLED(session) )
789  libirc_dump_data ("RECV", session->incoming_buf, offset);
790 #endif
791  // parse the string
792  libirc_process_incoming_data (session, offset - 2);
793 
794  if ( session->incoming_offset - offset > 0 )
795  memmove (session->incoming_buf, session->incoming_buf + offset, session->incoming_offset - offset);
796 
797  session->incoming_offset -= offset;
798  }
799  }
800 
801  // We can write a stored buffer
802  if ( FD_ISSET (session->sock, out_set) )
803  {
804  int length;
805 
806  // Because outgoing_buf could be changed asynchronously, we should
807  // lock any change
808  libirc_mutex_lock (&session->mutex_session);
809  length = socket_send (&session->sock, session->outgoing_buf, session->outgoing_offset);
810 
811  if ( length <= 0 )
812  {
813  session->lasterror = (length == 0 ? LIBIRC_ERR_CLOSED : LIBIRC_ERR_TERMINATED);
814  session->state = LIBIRC_STATE_DISCONNECTED;
815 
816  libirc_mutex_unlock (&session->mutex_session);
817  return 1;
818  }
819 
820 #if defined (ENABLE_DEBUG)
821  if ( IS_DEBUG_ENABLED(session) )
822  libirc_dump_data ("SEND", session->outgoing_buf, length);
823 #endif
824 
825  if ( session->outgoing_offset - length > 0 )
826  memmove (session->outgoing_buf, session->outgoing_buf + length, session->outgoing_offset - length);
827 
828  session->outgoing_offset -= length;
829  libirc_mutex_unlock (&session->mutex_session);
830  }
831 
832  return 0;
833 }

836 int irc_send_raw (irc_session_t * session, const char * format, ...)
837 {
838  char buf[1024] = {};
839  va_list va_alist;
840 
841  if ( session->state != LIBIRC_STATE_CONNECTED )
842  {
843  session->lasterror = LIBIRC_ERR_STATE;
844  return 1;
845  }
846 
847  va_start (va_alist, format);
848  vsnprintf (buf, sizeof(buf), format, va_alist);
849  va_end (va_alist);
850 
851  libirc_mutex_lock (&session->mutex_session);
852 
853  if ( (strlen(buf) + 2) >= (sizeof(session->outgoing_buf) - session->outgoing_offset) )
854  {
855  libirc_mutex_unlock (&session->mutex_session);
856  session->lasterror = LIBIRC_ERR_NOMEM;
857  return 1;
858  }
859

861  session->outgoing_offset += strlen (buf);
862  session->outgoing_buf[session->outgoing_offset++] = 0x0D;
863  session->outgoing_buf[session->outgoing_offset++] = 0x0A;
864 
865  libirc_mutex_unlock (&session->mutex_session);
866  return 0;
867 }
