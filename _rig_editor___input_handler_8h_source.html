175  inline void ButtonPressInputReceived(bool handled_by_gui)
176  {
177  if (handled_by_gui && BITMASK_IS_0(flags, BUTTON_PRESS_INPUT_RECEIVED)) // First input + handled by GUI
178  {
179  BITMASK_SET_1(flags, ALL_BUTTON_PRESSES_HANDLED_BY_GUI);
180  }
181  else if (!handled_by_gui && BITMASK_IS_1(flags, ALL_BUTTON_PRESSES_HANDLED_BY_GUI)) // Any input + not handled by GUI
182  {
183  BITMASK_SET_0(flags, ALL_BUTTON_PRESSES_HANDLED_BY_GUI);
184  }
185  // else ALL_BUTTON_PRESSES_HANDLED_BY_GUI stays FALSE
186  BITMASK_SET_1(flags, BUTTON_PRESS_INPUT_RECEIVED);
187  }
188 
189  inline void ButtonReleaseInputReceived(bool handled_by_gui)
