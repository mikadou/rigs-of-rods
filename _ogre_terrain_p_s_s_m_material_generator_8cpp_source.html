00774     void TerrainPSSMMaterialGenerator::SM2Profile::ShaderHelperCg::generateVpHeader(
00775         const SM2Profile* prof, const Terrain* terrain, TechniqueType tt, StringUtil::StrStreamType& outStream)
00776     {
00777         outStream <<
00778             "void main_vp(\n";
00779         bool compression = terrain->_getUseVertexCompression() && tt != RENDER_COMPOSITE_MAP;
00780         if (compression)
00781         {
00782             outStream <<
00783                 "float2 posIndex : POSITION,\n"
00784                 "float height  : TEXCOORD0,\n";
00785         }
00786         else
00787         {
00788             outStream <<
00789                 "float4 pos : POSITION,\n"
00790                 "float2 uv  : TEXCOORD0,\n";
00791 
00792         }
00793         if (tt != RENDER_COMPOSITE_MAP)
00794             outStream << "float2 delta  : TEXCOORD1,\n"; // lodDelta, lodThreshold
00795 
00796         outStream <<
00797             "uniform float4x4 worldMatrix,\n"
00798             "uniform float4x4 viewProjMatrix,\n"
00799             "uniform float2   lodMorph,\n"; // morph amount, morph LOD target
00800 
00801         if (compression)
00802         {
00803             outStream <<
00804                 "uniform float4x4   posIndexToObjectSpace,\n"
00805                 "uniform float    baseUVScale,\n";
00806         }
00807         // uv multipliers
00808         uint maxLayers = prof->getMaxLayers(terrain);
00809         uint numLayers = std::min(maxLayers, static_cast<uint>(terrain->getLayerCount()));
00810         uint numUVMultipliers = (numLayers / 4);
00811         if (numLayers % 4)
00812             ++numUVMultipliers;
00813         for (uint i = 0; i < numUVMultipliers; ++i)
00814             outStream << "uniform float4 uvMul_" << i << ", \n";
00815 
00816         outStream <<
00817             "out float4 oPos : POSITION,\n"
00818             "out float4 oPosObj : TEXCOORD0 \n";
00819 
00820         uint texCoordSet = 1;
00821         outStream <<
00822             ", out float4 oUVMisc : TEXCOORD" << texCoordSet++ << " // xy = uv, z = camDepth\n";
00823 
00824         // layer UV's premultiplied, packed as xy/zw
00825         uint numUVSets = numLayers / 2;
00826         if (numLayers % 2)
00827             ++numUVSets;
00828         if (tt != LOW_LOD)
00829         {
00830             for (uint i = 0; i < numUVSets; ++i)
00831             {
00832                 outStream <<
00833                     ", out float4 oUV" << i << " : TEXCOORD" << texCoordSet++ << "\n";
00834             }
00835         }
00836 
00837         if (prof->getParent()->getDebugLevel() && tt != RENDER_COMPOSITE_MAP)
00838         {
00839             outStream << ", out float2 lodInfo : TEXCOORD" << texCoordSet++ << "\n";
00840         }
00841 
00842         bool fog = terrain->getSceneManager()->getFogMode() != FOG_NONE && tt != RENDER_COMPOSITE_MAP;
00843         if (fog)
00844         {
00845             outStream <<
00846                 ", uniform float4 fogParams\n"
00847                 ", out float fogVal : COLOR\n";
00848         }
00849 
00850         if (prof->isShadowingEnabled(tt, terrain))
00851         {
00852             texCoordSet = generateVpDynamicShadowsParams(texCoordSet, prof, terrain, tt, outStream);
00853         }
00854 
00855         // check we haven't exceeded texture coordinates
00856         if (texCoordSet > 8)
00857         {
00858             OGRE_EXCEPT(Exception::ERR_INVALIDPARAMS,
00859                 "Requested options require too many texture coordinate sets! Try reducing the number of layers.",
00860                 __FUNCTION__);
00861         }
00862 
00863         outStream <<
00864             ")\n"
00865             "{\n";
00866         if (compression)
00867         {
00868             outStream <<
00869                 "   float4 pos;\n"
00870                 "   pos = mul(posIndexToObjectSpace, float4(posIndex, height, 1));\n"
00871                 "   float2 uv = float2(posIndex.x * baseUVScale, 1.0 - (posIndex.y * baseUVScale));\n";
00872         }
00873         outStream <<
00874             "   float4 worldPos = mul(worldMatrix, pos);\n"
00875             "   oPosObj = pos;\n";
00876 
00877         if (tt != RENDER_COMPOSITE_MAP)
00878         {
00879             // determine whether to apply the LOD morph to this vertex
00880             // we store the deltas against all vertices so we only want to apply 
00881             // the morph to the ones which would disappear. The target LOD which is
00882             // being morphed to is stored in lodMorph.y, and the LOD at which 
00883             // the vertex should be morphed is stored in uv.w. If we subtract
00884             // the former from the latter, and arrange to only morph if the
00885             // result is negative (it will only be -1 in fact, since after that
00886             // the vertex will never be indexed), we will achieve our aim.
00887             // sign(vertexLOD - targetLOD) == -1 is to morph
00888             outStream <<
00889                 "   float toMorph = -min(0, sign(delta.y - lodMorph.y));\n";
00890             // this will either be 1 (morph) or 0 (don't morph)
00891             if (prof->getParent()->getDebugLevel())
00892             {
00893                 // x == LOD level (-1 since value is target level, we want to display actual)
00894                 outStream << "lodInfo.x = (lodMorph.y - 1) / " << terrain->getNumLodLevels() << ";\n";
00895                 // y == LOD morph
00896                 outStream << "lodInfo.y = toMorph * lodMorph.x;\n";
00897             }
00898 
00899             // morph
00900             switch (terrain->getAlignment())
00901             {
00902             case Terrain::ALIGN_X_Y:
00903                 outStream << "  worldPos.z += delta.x * toMorph * lodMorph.x;\n";
00904                 break;
00905             case Terrain::ALIGN_X_Z:
00906                 outStream << "  worldPos.y += delta.x * toMorph * lodMorph.x;\n";
00907                 break;
00908             case Terrain::ALIGN_Y_Z:
00909                 outStream << "  worldPos.x += delta.x * toMorph * lodMorph.x;\n";
00910                 break;
00911             };
00912         }
00913 
00914 
00915         // generate UVs
00916         if (tt != LOW_LOD)
00917         {
00918             for (uint i = 0; i < numUVSets; ++i)
00919             {
00920                 uint layer = i * 2;
00921                 uint uvMulIdx = layer / 4;
00922 
00923                 outStream <<
00924                     "   oUV" << i << ".xy = " << " uv.xy * uvMul_" << uvMulIdx << "." << getChannel(layer) << ";\n";
00925                 outStream <<
00926                     "   oUV" << i << ".zw = " << " uv.xy * uvMul_" << uvMulIdx << "." << getChannel(layer + 1) << ";\n";
00927 
00928             }
00929 
00930         }
00931 
00932 
00933     }

00953             "float4 vertexPos : POSITION,\n"
00954             "float4 position : TEXCOORD0,\n";
00955 
00956         uint texCoordSet = 1;
00957         outStream <<
00958             "float4 uvMisc : TEXCOORD" << texCoordSet++ << ",\n";
00959 
00960         // UV's premultiplied, packed as xy/zw
00961         uint maxLayers = prof->getMaxLayers(terrain);
00962         uint numBlendTextures = std::min(terrain->getBlendTextureCount(maxLayers), terrain->getBlendTextureCount());
00963         uint numLayers = std::min(maxLayers, static_cast<uint>(terrain->getLayerCount()));

00965         if (numLayers % 2)
00966             ++numUVSets;
00967         if (tt != LOW_LOD)
00968         {
00969             for (uint i = 0; i < numUVSets; ++i)
00970             {
00971                 outStream <<
00972                     "float4 layerUV" << i << " : TEXCOORD" << texCoordSet++ << ", \n";
00973             }
00974 
00975         }
00976         if (prof->getParent()->getDebugLevel() && tt != RENDER_COMPOSITE_MAP)
00977         {
00978             outStream << "float2 lodInfo : TEXCOORD" << texCoordSet++ << ", \n";
00979         }
00980 
00981         bool fog = terrain->getSceneManager()->getFogMode() != FOG_NONE && tt != RENDER_COMPOSITE_MAP;
00982         if (fog)
00983         {
00984             outStream <<
00985                 "uniform float3 fogColour, \n"
00986                 "float fogVal : COLOR,\n";
00987         }
00988 
00989         uint currentSamplerIdx = 0;
00990 
00991         outStream <<
00992             // Only 1 light supported in this version
00993             // deferred shading profile / generator later, ok? :)
00994             "uniform float3 ambient,\n"
00995             "uniform float4 lightPosObjSpace,\n"
00996             "uniform float3 lightDiffuseColour,\n"
00997             "uniform float3 lightSpecularColour,\n"
00998             "uniform float3 eyePosObjSpace,\n"
00999             // pack scale, bias and specular
01000             "uniform float4 scaleBiasSpecular,\n";
01001 
01002         if (tt == LOW_LOD)
01003         {
01004             // single composite map covers all the others below
01005             outStream <<
01006                 "uniform sampler2D compositeMap : register(s" << currentSamplerIdx++ << ")\n";
01007         }
01008         else
01009         {
01010             outStream <<
01011                 "uniform sampler2D globalNormal : register(s" << currentSamplerIdx++ << ")\n";
01012 
01013 
01014             if (terrain->getGlobalColourMapEnabled() && prof->isGlobalColourMapEnabled())
01015             {
01016                 outStream << ", uniform sampler2D globalColourMap : register(s"
01017                     << currentSamplerIdx++ << ")\n";

01020             {
01021                 outStream << ", uniform sampler2D lightMap : register(s"
01022                     << currentSamplerIdx++ << ")\n";
01023             }
01024             // Blend textures - sampler definitions
01025             for (uint i = 0; i < numBlendTextures; ++i)
01026             {
01027                 outStream << ", uniform sampler2D blendTex" << i
01028                     << " : register(s" << currentSamplerIdx++ << ")\n";
01029             }
01030 
01031             // Layer textures - sampler definitions & UV multipliers
01032             for (uint i = 0; i < numLayers; ++i)
01033             {
01034                 outStream << ", uniform sampler2D difftex" << i
01035                     << " : register(s" << currentSamplerIdx++ << ")\n";
01036                 outStream << ", uniform sampler2D normtex" << i
01037                     << " : register(s" << currentSamplerIdx++ << ")\n";
01038             }
01039         }
01040 
01041         if (prof->isShadowingEnabled(tt, terrain))
01042         {
01043             generateFpDynamicShadowsParams(&texCoordSet, &currentSamplerIdx, prof, terrain, tt, outStream);
01044         }
01045 
01046         // check we haven't exceeded samplers
01047         if (currentSamplerIdx > 16)
01048         {
01049             OGRE_EXCEPT(Exception::ERR_INVALIDPARAMS,
01050                 "Requested options require too many texture samplers! Try reducing the number of layers.",
01051                 __FUNCTION__);
01052         }
01053 
01054         outStream <<
01055             ") : COLOR\n"
01056             "{\n"
01057             "   float4 outputCol;\n"
01058             "   float shadow = 1.0;\n"
01059             "   float2 uv = uvMisc.xy;\n"
01060             // base colour
01061             "   outputCol = float4(0,0,0,1);\n";
01062 
01063         if (tt != LOW_LOD)
01064         {
01065             outStream <<
01066                 // global normal
01067                 "   float3 normal = expand(tex2D(globalNormal, uv)).rgb;\n";
01068 
01069         }
01070 
01071         outStream <<
01072             "   float3 lightDir = \n"
01073             "       lightPosObjSpace.xyz -  (position.xyz * lightPosObjSpace.w);\n"
01074             "   float3 eyeDir = eyePosObjSpace - position.xyz;\n"
01075 
01076             // set up accumulation areas
01077             "   float3 diffuse = float3(0,0,0);\n"
01078             "   float specular = 0;\n";
01079 
01080 
01081         if (tt == LOW_LOD)
01082         {
01083             // we just do a single calculation from composite map
01084             outStream <<
01085                 "   float4 composite = tex2D(compositeMap, uv);\n"
01086                 "   diffuse = composite.rgb;\n";
01087             // TODO - specular; we'll need normals for this!
01088         }
01089         else
01090         {
01091             // set up the blend values
01092             for (uint i = 0; i < numBlendTextures; ++i)
01093             {
01094                 outStream << "  float4 blendTexVal" << i << " = tex2D(blendTex" << i << ", uv);\n";
01095             }
01096 
01097             if (prof->isLayerNormalMappingEnabled())
01098             {
01099                 // derive the tangent space basis
01100                 // we do this in the pixel shader because we don't have per-vertex normals
01101                 // because of the LOD, we use a normal map
01102                 // tangent is always +x or -z in object space depending on alignment
01103                 switch (terrain->getAlignment())
01104                 {
01105                 case Terrain::ALIGN_X_Y:
01106                 case Terrain::ALIGN_X_Z:

01128 
01129             }
01130             else
01131             {
01132                 // simple per-pixel lighting with no normal mapping
01133                 outStream << "  lightDir = normalize(lightDir);\n";
01134                 outStream << "  eyeDir = normalize(eyeDir);\n";
01135                 outStream << "  float3 halfAngle = normalize(lightDir + eyeDir);\n";
01136                 outStream << "  float4 litRes = lit(dot(lightDir, normal), dot(halfAngle, normal), scaleBiasSpecular.z);\n";
01137 
01138             }
01139         }

01157         String blendWeightStr = String("blendTexVal") + StringConverter::toString(blendIdx) +
01158             "." + blendChannel;

01164         */

01167         outStream << "  float2 uv" << layer << " = layerUV" << uvIdx << uvChannels << ";\n";
