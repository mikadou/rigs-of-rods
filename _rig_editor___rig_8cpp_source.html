00749 bool Rig::RefreshMouseHoveredNode(Vector2int const & mouse_position)
00750 {
00751     /*
00752     Lookup method: for each node
00753         - Compute mouse offset
00754         - Convert it's components to abs()
00755         - Pick the larger component as 'distance'
00756         - Compare 'distance' to previous node
00757 
00758     Finally, compare result node's 'distance' to configured box.
00759     */
00760 
00761     auto itor = m_nodes.begin();
00762     Node* result = m_mouse_hovered_node;
00763     if (result == nullptr)
00764     {
00765         result = &itor->second;
00766         ++itor;
00767     }
00768     Vector2int node_mouse_offset = result->GetScreenPosition() - mouse_position;
00769     node_mouse_offset.x *= (node_mouse_offset.x < 0) ? -1 : 1; // Absolute value
00770     node_mouse_offset.y *= (node_mouse_offset.y < 0) ? -1 : 1; // Absolute value
00771     int result_mouse_distance = (node_mouse_offset.x > node_mouse_offset.y) ? node_mouse_offset.x : node_mouse_offset.y;
00772     
00773     auto itor_end = m_nodes.end();
00774 
00775     for ( ; itor != itor_end ; ++itor)
00776     {
00777         node_mouse_offset = itor->second.GetScreenPosition() - mouse_position;
00778         node_mouse_offset.x *= (node_mouse_offset.x < 0) ? -1 : 1; // Absolute value
00779         node_mouse_offset.y *= (node_mouse_offset.y < 0) ? -1 : 1; // Absolute value
00780         int current_mouse_distance = (node_mouse_offset.x > node_mouse_offset.y) ? node_mouse_offset.x : node_mouse_offset.y;
00781         if (current_mouse_distance < result_mouse_distance)
00782         {
00783             result_mouse_distance = current_mouse_distance;
00784             result = &itor->second;
00785         }
00786     }
00787 
00788     if (result_mouse_distance > m_config->node_mouse_box_halfsize_px)
00789     {
00790         result = nullptr;
