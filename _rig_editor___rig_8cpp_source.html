57 Ogre::String RigBuildingReport::ToString() const
58 {
59  std::stringstream s;
60  auto end = m_messages.end();
61  auto itor = m_messages.begin();
62  for (; itor != end; ++itor)
63  {
64  switch (itor->level)
65  {
66  case Message::LEVEL_ERROR:
67  s << "ERROR ";
68  break;
69  case Message::LEVEL_INFO:
70  s << "INFO ";
71  break;
72  case Message::LEVEL_WARNING:
73  s << "WARNING ";
74  break;
75  default:
76  assert(false && "Wrong message type");
77  break;
78  }
79  s << "(module: " << itor->module_name << ", section: " << itor->section_name << "): " << itor->text << std::endl;
80  }
81  return s.str();
82 }

749 bool Rig::RefreshMouseHoveredNode(Vector2int const & mouse_position)
750 {
751  /*
752  Lookup method: for each node
753  - Compute mouse offset
754  - Convert it's components to abs()
755  - Pick the larger component as 'distance'
756  - Compare 'distance' to previous node
757 
758  Finally, compare result node's 'distance' to configured box.
759  */
760 
761  auto itor = m_nodes.begin();
762  Node* result = m_mouse_hovered_node;
763  if (result == nullptr)
764  {
765  result = &itor->second;
766  ++itor;
767  }
768  Vector2int node_mouse_offset = result->GetScreenPosition() - mouse_position;
769  node_mouse_offset.x *= (node_mouse_offset.x < 0) ? -1 : 1; // Absolute value
770  node_mouse_offset.y *= (node_mouse_offset.y < 0) ? -1 : 1; // Absolute value
771  int result_mouse_distance = (node_mouse_offset.x > node_mouse_offset.y) ? node_mouse_offset.x : node_mouse_offset.y;
772 
773  auto itor_end = m_nodes.end();
774 
775  for ( ; itor != itor_end ; ++itor)
776  {
777  node_mouse_offset = itor->second.GetScreenPosition() - mouse_position;
778  node_mouse_offset.x *= (node_mouse_offset.x < 0) ? -1 : 1; // Absolute value
779  node_mouse_offset.y *= (node_mouse_offset.y < 0) ? -1 : 1; // Absolute value
780  int current_mouse_distance = (node_mouse_offset.x > node_mouse_offset.y) ? node_mouse_offset.x : node_mouse_offset.y;
781  if (current_mouse_distance < result_mouse_distance)
782  {
783  result_mouse_distance = current_mouse_distance;
784  result = &itor->second;
785  }
786  }
787 
788  if (result_mouse_distance > m_config->node_mouse_box_halfsize_px)
789  {
790  result = nullptr;
791  }
792 
793  bool ret_val = (m_mouse_hovered_node != result);
794  m_mouse_hovered_node = result;
795  return ret_val;
