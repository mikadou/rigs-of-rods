35 void LandVehicleSimulation::UpdateCruiseControl(Beam* curr_truck, float dt)
36 {
37  if (RoR::Application::GetInputEngine()->getEventValue(EV_TRUCK_BRAKE) > 0.05f ||
38  RoR::Application::GetInputEngine()->getEventValue(EV_TRUCK_MANUAL_CLUTCH) > 0.05f ||
39  (curr_truck->cc_target_speed < curr_truck->cc_target_speed_lower_limit) ||
40  (curr_truck->parkingbrake && curr_truck->engine->getGear() > 0) ||
41  !curr_truck->engine->isRunning() ||
42  !curr_truck->engine->hasContact())
43  {
44  curr_truck->cruisecontrolToggle();
45  return;
46  }
47 
48  if (curr_truck->engine->getGear() > 0)
49  {
50  // Try to maintain the target speed
51  if (curr_truck->cc_target_speed > curr_truck->WheelSpeed)
52  {
53  float accl = (curr_truck->cc_target_speed - curr_truck->WheelSpeed) * 2.0f;
54  accl = std::max(curr_truck->engine->getAcc(), accl);
55  accl = std::min(accl, 1.0f);
56  curr_truck->engine->setAcc(accl);
57  }
58  } else if (curr_truck->engine->getGear() == 0) // out of gear
59  {
60  // Try to maintain the target rpm
61  if (curr_truck->cc_target_rpm > curr_truck->engine->getRPM())
62  {
63  float accl = (curr_truck->cc_target_rpm - curr_truck->engine->getRPM()) * 0.01f;
64  accl = std::max(curr_truck->engine->getAcc(), accl);
65  accl = std::min(accl, 1.0f);
66  curr_truck->engine->setAcc(accl);
67  }
68  }
69 
70  if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_ACCL))
71  {
72  if (curr_truck->engine->getGear() > 0)
73  {
74  curr_truck->cc_target_speed += 2.5f * dt;
75  curr_truck->cc_target_speed = std::max(curr_truck->cc_target_speed_lower_limit, curr_truck->cc_target_speed);
76  if (curr_truck->sl_enabled)
77  {
78  curr_truck->cc_target_speed = std::min(curr_truck->cc_target_speed, curr_truck->sl_speed_limit);
79  }
80  } else if (curr_truck->engine->getGear() == 0) // out of gear
81  {
82  curr_truck->cc_target_rpm += 1000.0f * dt;
83  curr_truck->cc_target_rpm = std::min(curr_truck->cc_target_rpm, curr_truck->engine->getMaxRPM());
84  }
85  }
86  if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_DECL))
87  {
88  if (curr_truck->engine->getGear() > 0)
89  {
90  curr_truck->cc_target_speed -= 2.5f * dt;
91  curr_truck->cc_target_speed = std::max(curr_truck->cc_target_speed_lower_limit, curr_truck->cc_target_speed);
92  } else if (curr_truck->engine->getGear() == 0) // out of gear
93  {
94  curr_truck->cc_target_rpm -= 1000.0f * dt;
95  curr_truck->cc_target_rpm = std::max(curr_truck->engine->getMinRPM(), curr_truck->cc_target_rpm);
96  }
97  }
98  if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_READJUST))
99  {
100  curr_truck->cc_target_speed = std::max(curr_truck->WheelSpeed, curr_truck->cc_target_speed);
101  if (curr_truck->sl_enabled)
102  {
103  curr_truck->cc_target_speed = std::min(curr_truck->cc_target_speed, curr_truck->sl_speed_limit);
104  }
105  curr_truck->cc_target_rpm = curr_truck->engine->getRPM();
106  }
107 
108  if (curr_truck->cc_can_brake)
109  {
110  if (curr_truck->WheelSpeed > curr_truck->cc_target_speed + 0.5f && !RoR::Application::GetInputEngine()->getEventValue(EV_TRUCK_ACCELERATE))
111  {
112  float brake = (curr_truck->WheelSpeed - curr_truck->cc_target_speed) * 0.5f;
113  brake = std::min(brake, 1.0f);
114  curr_truck->brake = curr_truck->brakeforce * brake;
115  }
116  }

119 void LandVehicleSimulation::CheckSpeedLimit(Beam* curr_truck, float dt)
120 {
121  if (curr_truck->sl_enabled && curr_truck->engine->getGear() != 0)

124  accl = std::max(0.0f, accl);
