00044         curr_truck->cruisecontrolToggle();

00046     }
00047 
00048     if (curr_truck->engine->getGear() > 0)
00049     {
00050         // Try to maintain the target speed
00051         if (curr_truck->cc_target_speed > curr_truck->WheelSpeed)
00052         {
00053             float accl = (curr_truck->cc_target_speed - curr_truck->WheelSpeed) * 2.0f;
00054             accl = std::max(curr_truck->engine->getAcc(), accl);
00055             accl = std::min(accl, 1.0f);
00056             curr_truck->engine->setAcc(accl);
00057         }
00058     } else if (curr_truck->engine->getGear() == 0) // out of gear
00059     {
00060         // Try to maintain the target rpm
00061         if (curr_truck->cc_target_rpm > curr_truck->engine->getRPM())
00062         {
00063             float accl = (curr_truck->cc_target_rpm - curr_truck->engine->getRPM()) * 0.01f;
00064             accl = std::max(curr_truck->engine->getAcc(), accl);
00065             accl = std::min(accl, 1.0f);
00066             curr_truck->engine->setAcc(accl);
00067         }
00068     }
00069 
00070     if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_ACCL))
00071     {
00072         if (curr_truck->engine->getGear() > 0)
00073         {
00074             curr_truck->cc_target_speed += 2.5f * dt;
00075             curr_truck->cc_target_speed  = std::max(curr_truck->cc_target_speed_lower_limit, curr_truck->cc_target_speed);
00076             if (curr_truck->sl_enabled)
00077             {
00078                 curr_truck->cc_target_speed  = std::min(curr_truck->cc_target_speed, curr_truck->sl_speed_limit);
00079             }
00080         } else if (curr_truck->engine->getGear() == 0) // out of gear
00081         {
00082             curr_truck->cc_target_rpm += 1000.0f * dt;
00083             curr_truck->cc_target_rpm  = std::min(curr_truck->cc_target_rpm, curr_truck->engine->getMaxRPM());
00084         }
00085     }
00086     if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_DECL))
00087     {
00088         if (curr_truck->engine->getGear() > 0)
00089         {
00090             curr_truck->cc_target_speed -= 2.5f * dt;
00091             curr_truck->cc_target_speed  = std::max(curr_truck->cc_target_speed_lower_limit, curr_truck->cc_target_speed);
00092         } else if (curr_truck->engine->getGear() == 0) // out of gear
00093         {
00094             curr_truck->cc_target_rpm -= 1000.0f * dt;
00095             curr_truck->cc_target_rpm  = std::max(curr_truck->engine->getMinRPM(), curr_truck->cc_target_rpm);
00096         }
00097     }
00098     if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_READJUST))
00099     {
00100         curr_truck->cc_target_speed = std::max(curr_truck->WheelSpeed, curr_truck->cc_target_speed);
00101         if (curr_truck->sl_enabled)
00102         {
00103             curr_truck->cc_target_speed = std::min(curr_truck->cc_target_speed, curr_truck->sl_speed_limit);
00104         }
00105         curr_truck->cc_target_rpm   = curr_truck->engine->getRPM();
00106     }
00107 
00108     if (curr_truck->cc_can_brake)
00109     {
00110         if (curr_truck->WheelSpeed > curr_truck->cc_target_speed + 0.5f && !RoR::Application::GetInputEngine()->getEventValue(EV_TRUCK_ACCELERATE))
00111         {
00112             float brake = (curr_truck->WheelSpeed - curr_truck->cc_target_speed) * 0.5f;
00113             brake = std::min(brake, 1.0f);
00114             curr_truck->brake = curr_truck->brakeforce * brake;
00115         }
00116     }
