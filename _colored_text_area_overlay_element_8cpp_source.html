00083 DisplayString ColoredTextAreaOverlayElement::StripColors(const Ogre::String& otext)
00084 {
00085     try
00086     {
00087         DisplayString text = DisplayString(otext.c_str());
00088         DisplayString StrippedText;
00089         int i;
00090         for (i = 0; i < (int)text.size()-1; ++i)
00091         {
00092             if (text[i] == '^' &&
00093                 text[i+1] >= '0' && text[i+1] <= '9')   // This is a color code, ignore it
00094             {
00095                 ++i;
00096             }
00097             else
00098             {
00099                 StrippedText.append(1, text[i]);
00100             }
00101         }
00102         // One last character to add because loop went to size()-1
00103         if (i < (int)text.size())
00104             StrippedText.append(1, text[i]);
00105         return StrippedText;
00106     } catch(...)
00107     {
00108     }
00109     return String("UTF8 error (String cannot be displayed with current font set)");

00112 void ColoredTextAreaOverlayElement::setCaption(const DisplayString& text)
00113 {
00114     m_Colors.clear();
00115     m_Colors.resize(text.size(), 9);
00116     bool noColor = true;
00117     int i, iNumColorCodes = 0, iNumSpaces = 0;
00118     for (i = 0; i < (int)text.size()-1; ++i)
00119     {
00120         if (text[i] == ' ' || text[i] == '\n')
00121         {
00122             // Spaces and newlines are skipped when rendering and as such can't have a color
00123             ++iNumSpaces;
00124         }
00125         else if (text[i] == '^' &&
00126             text[i+1] >= '0' && text[i+1] <= '9')   // This is a color code
00127         {
00128             // Fill the color array starting from this point to the end with the new color code
00129             // adjustments need to made because color codes will be removed and spaces are not counted
00130             fill(m_Colors.begin()+i-(2*iNumColorCodes)-iNumSpaces, m_Colors.end(), text[i+1]-'0');
00131             ++i;
00132             ++iNumColorCodes;
00133             mColoursChanged = true;
00134             noColor = false;
00135         }
00136     }
00137     if (noColor)
00138         mColoursChanged = true;
00139     // Set the caption using the base class, but strip the color codes from it first
00140     TextAreaOverlayElement::setCaption(StripColors(text));
00141 }
