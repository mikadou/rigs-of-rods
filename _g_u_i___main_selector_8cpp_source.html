461 bool CLASS::SearchCompare(Ogre::String searchString, CacheEntry *ce)
462 {
463  if (searchString.find(":") == Ogre::String::npos)
464  {
465  // normal search
466 
467  // the name
468  Ogre::String dname_lower = ce->dname;
469  Ogre::StringUtil::toLowerCase(dname_lower);
470  if (dname_lower.find(searchString) != Ogre::String::npos)
471  return true;
472 
473  // the filename
474  Ogre::String fname_lower = ce->fname;
475  Ogre::StringUtil::toLowerCase(fname_lower);
476  if (fname_lower.find(searchString) != Ogre::String::npos)
477  return true;
478 
479  // the description
480  Ogre::String desc = ce->description;
481  Ogre::StringUtil::toLowerCase(desc);
482  if (desc.find(searchString) != Ogre::String::npos)
483  return true;
484 
485  // the authors
486  if (!ce->authors.empty())
487  {
488  std::vector<AuthorInfo>::const_iterator it;
489  for (it = ce->authors.begin(); it != ce->authors.end(); it++)
490  {
491  // author name
492  Ogre::String aname = it->name;
493  Ogre::StringUtil::toLowerCase(aname);
494  if (aname.find(searchString) != Ogre::String::npos)
495  return true;
496 
497  // author email
498  Ogre::String aemail = it->email;
499  Ogre::StringUtil::toLowerCase(aemail);
500  if (aemail.find(searchString) != Ogre::String::npos)
501  return true;
502  }
503  }
504  return false;
505  }
506  else
507  {
508  Ogre::StringVector v = Ogre::StringUtil::split(searchString, ":");
509  if (v.size() < 2) return false; //invalid syntax
510 
511  if (v[0] == "hash")
512  {
513  Ogre::String hash = ce->hash;
514  Ogre::StringUtil::toLowerCase(hash);
515  return (hash.find(v[1]) != Ogre::String::npos);
516  }
517  else if (v[0] == "guid")
518  {
519  Ogre::String guid = ce->guid;
520  Ogre::StringUtil::toLowerCase(guid);
521  return (guid.find(v[1]) != Ogre::String::npos);
522  }
523  else if (v[0] == "author")
524  {
525  // the authors
526  if (!ce->authors.empty())
527  {
528  std::vector<AuthorInfo>::const_iterator it;
529  for (it = ce->authors.begin(); it != ce->authors.end(); it++)
530  {
531  // author name
532  Ogre::String aname = it->name;
533  Ogre::StringUtil::toLowerCase(aname);
534  if (aname.find(v[1]) != Ogre::String::npos)
535  return true;
536 
537  // author email
538  Ogre::String aemail = it->email;
539  Ogre::StringUtil::toLowerCase(aemail);
540  if (aemail.find(v[1]) != Ogre::String::npos)
541  return true;
542  }
543  }
544  return false;
545  }
546  else if (v[0] == "wheels")
547  {
548  Ogre::String wheelsStr = TOUTFSTRING(ce->wheelcount) + "x" + TOUTFSTRING(ce->propwheelcount);
549  return (wheelsStr == v[1]);
550  }
551  else if (v[0] == "file")
552  {
553  Ogre::String fn = ce->fname;
554  Ogre::StringUtil::toLowerCase(fn);
555  return (fn.find(v[1]) != Ogre::String::npos);
556  }
557 
558 
559  }
560  return false;
561 }

865  float imgRatio = imgSize.width / (float)imgSize.height;

869  MyGUI::IntPoint newPosition(0, 0);

875  newSize.height = maxSize.width / imgRatio;

878  }
879  else

881  newSize.width = maxSize.height * imgRatio;
882  newSize.height = maxSize.height;
883  newPosition.left = maxSize.width - newSize.width;
884  newPosition.top = 0;
885  }

887  m_Preview->setSize(newSize);
