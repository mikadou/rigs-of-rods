233 Vector3 FlexMesh::updateVertices()
234 {
235  Vector3 center = (nodes[nodeIDs[0]].smoothpos + nodes[nodeIDs[1]].smoothpos) / 2.0;
236 
237  //optimization possible here : just copy bands on face
238 
239  covertices[0].vertex=nodes[nodeIDs[0]].smoothpos-center;
240  //normals
241  covertices[0].normal=approx_normalise(nodes[nodeIDs[0]].smoothpos-nodes[nodeIDs[1]].smoothpos);
242 
243  covertices[1].vertex=nodes[nodeIDs[1]].smoothpos-center;
244  //normals
245  covertices[1].normal=-covertices[0].normal;
246 
247  for (int i=0; i<nbrays*2; i++)
248  {
249  covertices[2+i].vertex=nodes[nodeIDs[2+i]].smoothpos-center;
250  //normals
251  if ((i%2)==0)
252  {
253  covertices[2+i].normal=approx_normalise(nodes[nodeIDs[0]].smoothpos-nodes[nodeIDs[1]].smoothpos);
254  } else
255  {
256  covertices[2+i].normal=-covertices[2+i-1].normal;
257  }
258  if (is_rimmed)
259  {
260  covertices[2+4*nbrays+i].vertex=nodes[nodeIDs[2+4*nbrays+i]].smoothpos-center;
261  //normals
262  if ((i%2)==0)
263  {
264  covertices[2+4*nbrays+i].normal=approx_normalise(nodes[nodeIDs[2+4*nbrays+i]].smoothpos-nodes[nodeIDs[2+4*nbrays+i+1]].smoothpos);
265  } else
266  {
267  covertices[2+4*nbrays+i].normal=-covertices[2+4*nbrays+i-1].normal;
268  }
269  //bands
270  covertices[2+2*nbrays+i].vertex=covertices[2+4*nbrays+i].vertex;
271  covertices[2+2*nbrays+i].normal=approx_normalise(covertices[2+4*nbrays+i].vertex);
272  } else
273  {
274  //bands
275  covertices[2+2*nbrays+i].vertex=covertices[2+i].vertex;
276  covertices[2+2*nbrays+i].normal=approx_normalise(covertices[2+i].vertex);
277  }
278  }
279  return center;
