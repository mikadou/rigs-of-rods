205 Vector3 FlexMeshWheel::updateVertices()
206 {
207  Vector3 center = (nodes[id0].smoothpos + nodes[id1].smoothpos) / 2.0;
208  Vector3 ray = nodes[idstart].smoothpos - nodes[id0].smoothpos;
209  Vector3 axis = nodes[id0].smoothpos - nodes[id1].smoothpos;
210 
211  axis.normalise();
212 
213  for (int i=0; i<nbrays; i++)
214  {
215  Plane pl=Plane(axis, nodes[id0].smoothpos);
216  ray=nodes[idstart+i*2].smoothpos-nodes[id0].smoothpos;
217  ray=pl.projectVector(ray);
218  ray.normalise();
219  covertices[i*6 ].vertex=nodes[id0].smoothpos+rim_radius*ray-center;
220 
221  covertices[i*6+1].vertex=nodes[idstart+i*2].smoothpos-0.05*(nodes[idstart+i*2].smoothpos-nodes[id0].smoothpos)-center;
222  covertices[i*6+2].vertex=nodes[idstart+i*2].smoothpos-0.1*(nodes[idstart+i*2].smoothpos-nodes[idstart+i*2+1].smoothpos)-center;
223  covertices[i*6+3].vertex=nodes[idstart+i*2+1].smoothpos-0.1*(nodes[idstart+i*2+1].smoothpos-nodes[idstart+i*2].smoothpos)-center;
224  covertices[i*6+4].vertex=nodes[idstart+i*2+1].smoothpos-0.05*(nodes[idstart+i*2+1].smoothpos-nodes[id1].smoothpos)-center;
225 
226  pl=Plane(-axis, nodes[id1].smoothpos);
227  ray=nodes[idstart+i*2+1].smoothpos-nodes[id1].smoothpos;
228  ray=pl.projectVector(ray);
229  ray.normalise();
230  covertices[i*6+5].vertex=nodes[id1].smoothpos+rim_radius*ray-center;
231 
232  //normals
233  covertices[i*6 ].normal=axis;
234  covertices[i*6+1].normal=(covertices[i*6].vertex-covertices[i*6+1].vertex).crossProduct(covertices[i*6].vertex-covertices[((i+1)%nbrays)*6+1].vertex)/normy;
235  covertices[i*6+2].normal=ray;
236  covertices[i*6+3].normal=ray;
237  covertices[i*6+4].normal=(covertices[i*6+4].vertex-covertices[i*6+5].vertex).crossProduct(covertices[i*6+4].vertex-covertices[((i+1)%nbrays)*6+4].vertex)/normy;
238  covertices[i*6+5].normal=-axis;
239  }
240  for (int i=0; i<6; i++)
241  {
242  covertices[nbrays*6+i].vertex=covertices[i].vertex;
243  covertices[nbrays*6+i].normal=covertices[i].normal;
244  }
245 
246  return center;
