00299     bool Mesh::isPointInGrid(const Ogre::Vector2 &Position)
00300     {
00301         Ogre::AxisAlignedBox WordMeshBox = mEntity->getWorldBoundingBox();
00302 
00303         // Get our mesh grid rectangle:
00304         // c-----------d
00305         // |           |
00306         // |           |
00307         // |           |
00308         // a-----------b
00309         Ogre::Vector3
00310             a = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_LEFT_BOTTOM),
00311             b = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_RIGHT_BOTTOM),
00312             c = WordMeshBox.getCorner(Ogre::AxisAlignedBox::NEAR_RIGHT_BOTTOM),
00313             d = WordMeshBox.getCorner(Ogre::AxisAlignedBox::NEAR_LEFT_BOTTOM);
00314 
00315         // Transform all corners to Ogre::Vector2 array
00316         Ogre::Vector2 Corners2D[4] =
00317            {Ogre::Vector2(a.x, a.z),
00318             Ogre::Vector2(b.x, b.z),
00319             Ogre::Vector2(c.x, c.z),
00320             Ogre::Vector2(d.x, d.z)};
00321 
00322         // Determinate if Position is into our rectangle, we use a line intersection detection
00323         // because our mesh rectangle can be rotated, if the number of collisions with the four
00324         // segments AB, BC, CD, DA is one, the Position point is into the rectangle, else(if number
00325         // of collisions are 0 or 2, the Position point is outside the rectangle.
00326         int NumberOfCollisions = 0;
00327         // Find a point wich isn't be inside the rectangle
00328         Ogre::Vector2 DestPoint = Corners2D[0] + (Corners2D[1]-Corners2D[0])*2;
00329         for (int k = 0; k < 3; k++)
00330         {
00331             if (Math::intersectionOfTwoLines(Corners2D[k], Corners2D[k+1], Position, DestPoint) != Ogre::Vector2::ZERO)
00332             {
00333                 NumberOfCollisions ++;
00334             }
00335 
00336             if (k == 2)
00337             {
00338                 if (Math::intersectionOfTwoLines(Corners2D[3], Corners2D[0], Position, DestPoint) != Ogre::Vector2::ZERO)
00339                 {
00340                     NumberOfCollisions ++;
00341                 }
00342             }
00343         }
00344         if (NumberOfCollisions == 1)
00345         {
00346             return true;
00347         }
00348 
00349         return false;
00350     }

00373             a = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_LEFT_BOTTOM),

00381             Ogre::Vector2(c.x, c.z)};
00382 
00383         // Get segments AB and AC
00384         Ogre::Vector2 AB = Corners2D[1]-Corners2D[0],
00385                       AC = Corners2D[2]-Corners2D[0];
00386 
00387         // Find the X/Y position projecting the Position point to AB and AC segments.
00388         Ogre::Vector2 XProjectedPoint = Position-AC,
00389                       YProjectedPoint = Position-AB;

00391         // Fint the intersections points
00392         Ogre::Vector2 XPoint = Math::intersectionOfTwoLines(Corners2D[0],Corners2D[1],Position,XProjectedPoint),
00393                       YPoint = Math::intersectionOfTwoLines(Corners2D[0],Corners2D[2],Position,YProjectedPoint);
00394 
00395         // Find lengths
00396         Ogre::Real ABLength = AB.length(),
00397                    ACLength = AC.length(),
00398                    XLength  = (XPoint-Corners2D[0]).length(),
00399                    YLength  = (YPoint-Corners2D[0]).length();
00400 
00401         // Find final x/y grid positions in [0,1] range
00402         Ogre::Real XFinal = XLength / ABLength,
00403                    YFinal = YLength / ACLength;
