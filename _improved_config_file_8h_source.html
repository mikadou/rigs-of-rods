76  bool saveAs(Ogre::String fn)
77  {
78  if (!fn.length())
79  {
80  OGRE_EXCEPT(Exception::ERR_INTERNAL_ERROR, "Saving of the configuration File is only allowed when the configuration was not loaded using the resource system!", "ImprovedConfigFile::save");
81  return false;
82  }
83  FILE *f = fopen(fn.c_str(), "w");
84  if (!f)
85  {
86  OGRE_EXCEPT(Exception::ERR_FILE_NOT_FOUND, "Cannot open File '"+fn+"' for writing.", "ImprovedConfigFile::save");
87  return false;
88  }
89 
90  SettingsBySection::iterator secIt;
91  for (secIt = mSettings.begin(); secIt!=mSettings.end(); secIt++)
92  {
93  if (secIt->first.size() > 0)
94  fprintf(f, "[%s]\n", secIt->first.c_str());
95  SettingsMultiMap::iterator setIt;
96  for (setIt = secIt->second->begin(); setIt!=secIt->second->end(); setIt++)
97  {
98  fprintf(f, "%s%c%s\n", setIt->first.c_str(), separators[0], setIt->second.c_str());
99  }
100 
101  }
102  fclose(f);
103  return true;
104  }

106  void setSetting(String key, String value, String section = StringUtil::BLANK)
107  {
108  SettingsMultiMap *set = mSettings[section];
109  if (!set)
110  {
111  // new section
112  set = new SettingsMultiMap();
113  mSettings[section] = set;
114  }
115  if (set->count(key))
116  // known key, delete old first
117  set->erase(key);
118  // add key
119  set->insert(std::multimap<String, String>::value_type(key, value));
120  }
