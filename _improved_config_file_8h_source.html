00077     {
00078         if (!fn.length())
00079         {
00080             OGRE_EXCEPT(Exception::ERR_INTERNAL_ERROR, "Saving of the configuration File is only allowed when the configuration was not loaded using the resource system!", "ImprovedConfigFile::save");
00081             return false;
00082         }
00083         FILE *f = fopen(fn.c_str(), "w");
00084         if (!f)
00085         {
00086             OGRE_EXCEPT(Exception::ERR_FILE_NOT_FOUND, "Cannot open File '"+fn+"' for writing.", "ImprovedConfigFile::save");
00087             return false;
00088         }
00089 
00090         SettingsBySection::iterator secIt;
00091         for (secIt = mSettings.begin(); secIt!=mSettings.end(); secIt++)
00092         {
00093             if (secIt->first.size() > 0)
00094                 fprintf(f, "[%s]\n", secIt->first.c_str());
00095             SettingsMultiMap::iterator setIt;
00096             for (setIt = secIt->second->begin(); setIt!=secIt->second->end(); setIt++)
00097             {
00098                 fprintf(f, "%s%c%s\n", setIt->first.c_str(), separators[0], setIt->second.c_str());
00099             }
00100             
00101         }
00102         fclose(f);
00103         return true;
00104     }
