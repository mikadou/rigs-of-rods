00064 void TextureToolWindow::fillCombo()
00065 {
00066     bool dynamicOnly = mChkDynamic->getStateSelected();
00067     mCBo->removeAllItems();
00068 
00069     ResourceManager::ResourceMapIterator it = TextureManager::getSingleton().getResourceIterator();
00070 
00071     while (it.hasMoreElements())
00072     {
00073         TexturePtr txt(it.getNext().staticCast<Texture>());
00074         
00075         if (dynamicOnly && ((txt->getUsage() & TU_STATIC) != 0)) continue;
00076         
00077         mCBo->addItem(txt->getName());
00078     }
00079 
00080     if (mCBo->getItemCount() > 0)
00081     {
00082         mCBo->setIndexSelected(0);
00083         mCBo->beginToItemSelected();
00084         updateControls(mCBo->getItemNameAt(0));
00085     }

00121 {
00122     try

00124         bool exists = TextureManager::getSingleton().resourceExists(texName);

00150

00152         switch(tex->getTextureType())
00153         {
00154         case TEX_TYPE_1D:  typeStr = "1D"; break;
00155         case TEX_TYPE_2D:  typeStr = "2D"; break;

00157         case TEX_TYPE_CUBE_MAP:  typeStr = "Cube Map"; break;
00158         }
00159         str += "#00aa00type: #000000" + typeStr + "\n";
00160 
00161         String usageStr="";

00163             usageStr += "static,\n";
00164         if (tex->getUsage() & TU_DYNAMIC)

00173             usageStr += "dynamic write only discardable,\n";
