64 void TextureToolWindow::fillCombo()
65 {
66  bool dynamicOnly = mChkDynamic->getStateSelected();
67  mCBo->removeAllItems();
68 
69  ResourceManager::ResourceMapIterator it = TextureManager::getSingleton().getResourceIterator();
70 
71  while (it.hasMoreElements())
72  {
73  TexturePtr txt(it.getNext().staticCast<Texture>());
74 
75  if (dynamicOnly && ((txt->getUsage() & TU_STATIC) != 0)) continue;
76 
77  mCBo->addItem(txt->getName());
78  }
79 
80  if (mCBo->getItemCount() > 0)
81  {
82  mCBo->setIndexSelected(0);
83  mCBo->beginToItemSelected();
84  updateControls(mCBo->getItemNameAt(0));
85  }
86 }

94 void TextureToolWindow::saveTexture( String texName, bool usePNG )
95 {
96  try
97  {
98  TexturePtr tex = TextureManager::getSingleton().getByName(texName);
99  if (tex.isNull()) return;
100 
101  Image img;
102  tex->convertToImage(img);
103 
104  // Save to disk!
105  String outname = SSETTING("User Path", "") + texName;
106  if (usePNG) outname += ".png";
107 
108  img.save(outname);
109 
110  UTFString msg = _L("saved texture as ") + outname;
111  RoR::Application::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_MSGTYPE_INFO, msg, "information.png");
112  }
113  catch(Exception &e)
114  {
115  UTFString str = "Exception while saving image: " + e.getFullDescription();
116  RoR::Application::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_MSGTYPE_INFO, str, "error.png");
117  }
118 }

120 void TextureToolWindow::updateControls( String texName )
121 {
122  try
123  {
124  bool exists = TextureManager::getSingleton().resourceExists(texName);
125  if (!exists)
126  {
127  mTxt->setCaption(convertToMyGUIString("Texture not found:\n"+texName));
128  mBtnSavePNG->setEnabled(false);
129  return;
130  }
131 
132  TexturePtr tex = TextureManager::getSingleton().getByName(texName);
133  if (tex.isNull())
134  {
135  mTxt->setCaption(convertToMyGUIString("Error loading texture:\n"+texName));
136  mBtnSavePNG->setEnabled(false);
137  return;
138  }
139 
140  String str = "#aa0000" + texName + "#000000\n";
141  str += "#00aa00res: #000000" + TOSTRING(tex->getWidth()) + " x " + TOSTRING(tex->getHeight()) + " pixels\n";
142  str += "#00aa00size: #000000" + formatBytes(tex->getSize()) + "\n";
143  str += "#00aa00format: #000000" + PixelUtil::getFormatName(tex->getFormat()) + "\n";
144  if (tex->getNumFaces() > 1)
145  str += "#00aa00faces: #000000" + TOSTRING(tex->getNumFaces()) + "\n";
146  if (tex->getFSAA() > 0)
147  str += "#00aa00FSAA: #000000" + TOSTRING(tex->getFSAA()) + "\n";
148  if (tex->getNumMipmaps() > 0)
149  str += "#00aa00mipmaps: #000000" + TOSTRING(tex->getNumMipmaps()) + "\n";
150 
151  String typeStr="";
152  switch(tex->getTextureType())
153  {
154  case TEX_TYPE_1D: typeStr = "1D"; break;
155  case TEX_TYPE_2D: typeStr = "2D"; break;
156  case TEX_TYPE_3D: typeStr = "3D"; break;
157  case TEX_TYPE_CUBE_MAP: typeStr = "Cube Map"; break;
158  }
159  str += "#00aa00type: #000000" + typeStr + "\n";
160 
161  String usageStr="";
162  if (tex->getUsage() & TU_STATIC)
163  usageStr += "static,\n";
164  if (tex->getUsage() & TU_DYNAMIC)
165  usageStr += "dynamic,\n";
166  if (tex->getUsage() & TU_WRITE_ONLY)
167  usageStr += "write only,\n";
168  if (tex->getUsage() & TU_STATIC_WRITE_ONLY)
169  usageStr += "static write only,\n";
170  if (tex->getUsage() & TU_DYNAMIC_WRITE_ONLY)
171  usageStr += "dynamic write only,\n";
172  if (tex->getUsage() & TU_DYNAMIC_WRITE_ONLY_DISCARDABLE)
173  usageStr += "dynamic write only discardable,\n";
174  if (tex->getUsage() & TU_AUTOMIPMAP)
175  usageStr += "automipmap,\n";
176  if (tex->getUsage() & TU_RENDERTARGET)
177  usageStr += "rendertarget,\n";
178  if (tex->getUsage() & TU_DEFAULT)
179  usageStr += "default\n";
180 
181  str += "#00aa00usage: #000000" + usageStr + "\n";
182  if (tex->getDepth() > 1)
183  str += "#00aa00depth: #000000" + TOSTRING(tex->getDepth()) + "\n";
184 
185 
186  mTxt->setCaption(convertToMyGUIString(str));
187  mImage->setImageTexture(texName);
188  mBtnSavePNG->setEnabled(true);
189  }
190  catch(Exception &e)
191  {
192  UTFString str = "Exception while opening texture:" + e.getFullDescription();
193  RoR::Application::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_MSGTYPE_INFO, str, "error.png");
194  }
195 }
