30 void MaterialFunctionMapper::addMaterial(int flareid, materialmapping_t t)

59  {

78  p->setAmbient(ColourValue::Black);
79  p->setDiffuse(ColourValue::Black);
80  }
81  } else
82  {
83  // normal technique
84  Pass *p = tech->getPass(0);
85  if (!p) continue;
86 
87  TextureUnitState *tus = p->getTextureUnitState(0);
88  if (!tus) continue;
89 
90  if (tus->getNumFrames() < 2)
91  continue;
92 
93  int frame = isvisible?1:0;
94 
95  tus->setCurrentFrame(frame);
96 
97  if (isvisible)
98  p->setSelfIllumination(it->emissiveColour);
99  else
100  p->setSelfIllumination(ColourValue::ZERO);
