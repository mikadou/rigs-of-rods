374 void PointColDetector::build_kdtree(int begin, int end, int axis, int index) {
375  int median;
376  for (;;) {
377  int slice_size = end - begin;
378  if (slice_size != 1) {
379  if (slice_size == 2) {
380  median = begin + 1;
381 
382  if (ref_list[begin].point[axis] > ref_list[median].point[axis]) {
383  std::swap(ref_list[begin],ref_list[median]);
384  }
385 
386  kdtree[index].min = ref_list[begin].point[axis];
387  kdtree[index].max = ref_list[median].point[axis];
388  }
389  else {
390  median = begin + (slice_size / 2);
391  partintwo(begin, median, end, axis, kdtree[index].min, kdtree[index].max);
392  }
393 
394  kdtree[index].middle = ref_list[median].point[axis];
395  kdtree[index].ref = NULL;

397  axis++;
398

402

405 
406  //Tail cutting

409  }
410  else {
411  kdtree[index].middle = ref_list[begin].point[axis];
