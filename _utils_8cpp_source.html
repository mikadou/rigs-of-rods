00358 {
00359     size_t location = ds->tell();
00360     // copy whole file into a buffer
00361     uint8_t *buf = 0;
00362     ds->seek(0); // from start
00363     // alloc buffer
00364     uint32_t bufSize = ds->size();
00365     buf = (uint8_t *)malloc(bufSize+1);
00366     // read into buffer
00367     ds->read(buf, bufSize);
00368 
00369     // and build the hash over it
00370     char hash_result[250];
00371     memset(hash_result, 0, 249);
00372 
00373     {
00374         RoR::CSHA1 sha1;
00375         sha1.UpdateHash(buf, bufSize);
00376         sha1.Final();
00377         sha1.ReportHash(hash_result, RoR::CSHA1::REPORT_HEX_SHORT);
00378     }
00379 
00380     // revert DS to previous position
00381     ds->seek(location);
00382     // release memory
00383     free(buf);
00384     buf = 0;
00385     
00386     hash = String(hash_result);
