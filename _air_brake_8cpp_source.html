00157 void Airbrake::updatePosition(float amount)
00158 {
00159     ratio=amount;
00160     if (!snode) return;
00161     Vector3 normal=(nodey->smoothpos-noderef->smoothpos).crossProduct(nodex->smoothpos-noderef->smoothpos);
00162     normal.normalise();
00163     //position
00164     Vector3 mposition=noderef->smoothpos+offset.x*(nodex->smoothpos-noderef->smoothpos)+offset.y*(nodey->smoothpos-noderef->smoothpos);
00165     snode->setPosition(mposition+normal*offset.z);
00166     //orientation
00167     Vector3 refx=nodex->smoothpos-noderef->smoothpos;
00168     refx.normalise();
00169     Vector3 refy=refx.crossProduct(normal);
00170     Quaternion orientation=Quaternion(Degree(-ratio*maxangle), (nodex->smoothpos-noderef->smoothpos).normalisedCopy())*Quaternion(refx, normal, refy);
00171     snode->setOrientation(orientation);
00172 }

00177     float altitude=noderef->AbsPosition.y;

00183 
00184     Vector3 wind=-noderef->Velocity;

00186 
00187     Vector3 drag=(1.2*area*sin(fabs(ratio*maxangle/57.3))*0.5*airdensity*wspeed/4.0)*wind;
00188     noderef->Forces+=drag;
