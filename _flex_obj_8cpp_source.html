255 Vector3 FlexObj::updateVertices()
256 {
257  unsigned int i;
258  Vector3 center;
259  center=(nodes[nodeIDs[0]].smoothpos+nodes[nodeIDs[1]].smoothpos)/2.0;
260  for (i=0; i<nVertices; i++)
261  {
262  //set position
263  covertices[i].vertex=nodes[nodeIDs[i]].smoothpos-center;
264  //reset normals
265  covertices[i].normal=Vector3::ZERO;
266  }
267  //accumulate normals per triangle
268  for (i=0; i<ibufCount/3; i++)
269  {
270  Vector3 v1, v2;
271  v1=nodes[nodeIDs[faces[i*3+1]]].smoothpos-nodes[nodeIDs[faces[i*3]]].smoothpos;
272  v2=nodes[nodeIDs[faces[i*3+2]]].smoothpos-nodes[nodeIDs[faces[i*3]]].smoothpos;
273  v1=v1.crossProduct(v2);
274  float s=v1.length();
275  //avoid large tris
276  if (s>sref[i])
277  {
278  covertices[faces[i*3+1]].vertex=covertices[faces[i*3]].vertex+Vector3(0.1,0,0);
279  covertices[faces[i*3+2]].vertex=covertices[faces[i*3]].vertex+Vector3(0,0,0.1);
280  }
281  //v1.normalise();
282  if (s == 0)
283  continue;
284  v1=v1/s;
285  covertices[faces[i*3]].normal+=v1;
286  covertices[faces[i*3+1]].normal+=v1;
287  covertices[faces[i*3+2]].normal+=v1;
288  }
289  //normalize

291  {
292  covertices[i].normal = approx_normalise(covertices[i].normal);
