00045 Vector3 Buoyance::computePressureForceSub(Vector3 a, Vector3 b, Vector3 c, Vector3 vel, int type)
00046 {
00047     //compute normal vector
00048     Vector3 normal=(b-a).crossProduct(c-a);
00049     float surf=normal.length();
00050     if (surf<0.00001) return Vector3::ZERO;
00051     normal=normal/surf; //normalize
00052     surf=surf/2.0; //surface
00053     float vol=0.0;
00054     if (type!=BUOY_DRAGONLY)
00055     {
00056         //compute pression prism points
00057         Vector3 ap=a+(gEnv->terrainManager->getWater()->getHeightWaves(a)-a.y)*9810*normal;
00058         Vector3 bp=b+(gEnv->terrainManager->getWater()->getHeightWaves(b)-b.y)*9810*normal;
00059         Vector3 cp=c+(gEnv->terrainManager->getWater()->getHeightWaves(c)-c.y)*9810*normal;
00060         //find centroid
00061         Vector3 ctd=(a+b+c+ap+bp+cp)/6.0;
00062         //compute volume
00063         vol+=computeVolume(ctd, a, b, c);
00064         vol+=computeVolume(ctd, a, ap, bp);
00065         vol+=computeVolume(ctd, a, bp, b);
00066         vol+=computeVolume(ctd, b, bp, cp);
00067         vol+=computeVolume(ctd, b, cp, c);
00068         vol+=computeVolume(ctd, c, cp, ap);
00069         vol+=computeVolume(ctd, c, ap, a);
00070         vol+=computeVolume(ctd, ap, cp, bp);
00071     };
00072     Vector3 drg=Vector3::ZERO;
00073     if (type!=BUOY_DRAGLESS)
00074     {
00075         //now, the drag
00076         //take in account the wave speed
00077         //compute center
00078         Vector3 tc=(a+b+c)/3.0;
00079         vel=vel-gEnv->terrainManager->getWater()->getVelocity(tc);
00080         float vell=vel.length();
00081         if (vell>0.01)
00082         {
00083             float cosaoa=fabs(normal.dotProduct(vel/vell));
00084     //      drg=(-500.0*surf*vell*cosaoa)*vel;
00085             drg=(-500.0*surf*vell*vell*cosaoa)*normal;
00086             if (normal.dotProduct(vel/vell)<0) drg=-drg;
00087             if (update && splashp)
00088             {
00089                 float fxl=vell*cosaoa*surf;
00090                 if (fxl>1.5) //if enough pushing drag
00091                 {
00092                     Vector3 fxdir=fxl*normal;
00093                     if (fxdir.y<0) fxdir.y=-fxdir.y;
00094                     if (gEnv->terrainManager->getWater()->getHeightWaves(a)-a.y<0.1) splashp->malloc(a, fxdir);
00095                     else if (gEnv->terrainManager->getWater()->getHeightWaves(b)-b.y<0.1) splashp->malloc(b, fxdir);
00096                     else if (gEnv->terrainManager->getWater()->getHeightWaves(c)-c.y<0.1) splashp->malloc(c, fxdir);
00097                 }
00098             }
00099         }
00100     }
00101     //okay
00102     if (sink) return drg;
00103     return vol*normal+drg;
00104 }

00107 Vector3 Buoyance::computePressureForce(Vector3 a, Vector3 b, Vector3 c, Vector3 vel, int type)
00108 {
00109     float wha=gEnv->terrainManager->getWater()->getHeightWaves((a+b+c)/3.0);
00110     //check if fully emerged
00111     if (a.y>wha && b.y>wha && c.y>wha) return Vector3::ZERO;
00112     //check if semi emerged
00113     if (a.y>wha || b.y>wha || c.y>wha)
00114     {
00115         //okay, several cases
00116         //one dip
00117         if (a.y<wha && b.y>wha && c.y>wha)
00118         {
00119             return computePressureForceSub(a,a+(wha-a.y)/(b.y-a.y)*(b-a),a+(wha-a.y)/(c.y-a.y)*(c-a), vel, type);
00120         }
00121         if (b.y<wha && c.y>wha && a.y>wha)
00122         {
00123             return computePressureForceSub(b,b+(wha-b.y)/(c.y-b.y)*(c-b),b+(wha-b.y)/(a.y-b.y)*(a-b), vel, type);
00124         }
00125         if (c.y<wha && a.y>wha && b.y>wha)
00126         {
00127             return computePressureForceSub(c,c+(wha-c.y)/(a.y-c.y)*(a-c),c+(wha-c.y)/(b.y-c.y)*(b-c), vel, type);
00128         }
00129         //two dips
00130         if (a.y>wha && b.y<wha && c.y<wha)
00131         {
00132             Vector3 tb=a+(wha-a.y)/(b.y-a.y)*(b-a);
00133             Vector3 tc=a+(wha-a.y)/(c.y-a.y)*(c-a);
00134             Vector3 f=computePressureForceSub(tb, b, tc, vel, type);
00135             return f+computePressureForceSub(tc, b, c, vel, type);
00136         }
00137         if (b.y>wha && c.y<wha && a.y<wha)
00138         {
00139             Vector3 tc=b+(wha-b.y)/(c.y-b.y)*(c-b);
00140             Vector3 ta=b+(wha-b.y)/(a.y-b.y)*(a-b);
00141             Vector3 f=computePressureForceSub(tc, c, ta, vel, type);
00142             return f+computePressureForceSub(ta, c, a, vel, type);
00143         }
00144         if (c.y>wha && a.y<wha && b.y<wha)
00145         {
00146             Vector3 ta=c+(wha-c.y)/(a.y-c.y)*(a-c);
00147             Vector3 tb=c+(wha-c.y)/(b.y-c.y)*(b-c);

00150         }

00152     }
00153     else
00154     {
00155         //fully submerged case
00156         return computePressureForceSub(a,b,c,vel, type);
