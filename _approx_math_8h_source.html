113 // Calculates approximate 1/square_root(x)
114 // it is faster than fast_invSqrt BUT
115 // use it in code not requiring precision
116 inline float approx_invSqrt(float x)
117 {
118  float xhalf = 0.5f*x;
119  int i = *(int*)&x; // get bits for floating value
120  i = 0x5f3759df - (i >> 1); // give initial guess y0
121  x = *(float*)&i; // convert bits back to float
122  x *= 1.5f - xhalf*x*x; // newton step, repeating this step
