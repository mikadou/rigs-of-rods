00101     info.uPriorities[CR_SMAPI] = 1; // Try Simple MAPI the last
00102     info.dwFlags = 0; // Install all available exception handlers

00104 
00105     int nInstResult = crInstall(&info);
00106     if (nInstResult!=0)
00107     {
00108         // Something goes wrong!
00109         TCHAR szErrorMsg[512];

00111

00113         printf("%s\n", szErrorMsg);
00114 
00115 
00116         ErrorUtils::ShowError(_L("Exception handling registration problem"), String(szErrorMsg));
