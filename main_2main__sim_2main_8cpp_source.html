78 void install_crashrpt()
79 {
80  // Install CrashRpt support
81  CR_INSTALL_INFO info;
82  memset(&info, 0, sizeof(CR_INSTALL_INFO));
83  info.cb = sizeof(CR_INSTALL_INFO);
84  info.pszAppName = "Rigs of Rods";
85  info.pszAppVersion = ROR_VERSION_STRING;
86  info.pszEmailSubject = "Error Report for Rigs of Rods";
87  info.pszEmailTo = "error-report@rigsofrods.com";
88 
89  char tmp[512]="";
90  sprintf(tmp, "http://api.rigsofrods.com/crashreport/?version=%s_%s", __DATE__, __TIME__);
91  for (unsigned int i=0;i<strnlen(tmp, 512);i++)
92  {
93  if (tmp[i] == ' ')
94  tmp[i] = '_';
95  }
96 
97  info.pszUrl = tmp;
98  info.pfnCrashCallback = crashCallback;
99  info.uPriorities[CR_HTTP] = 3; // Try HTTP the first
100  info.uPriorities[CR_SMTP] = 2; // Try SMTP the second
101  info.uPriorities[CR_SMAPI] = 1; // Try Simple MAPI the last
102  info.dwFlags = 0; // Install all available exception handlers
103  info.pszPrivacyPolicyURL = "http://wiki.rigsofrods.com/pages/Crash_Report_Privacy_Policy"; // URL for the Privacy Policy link
104 
105  int nInstResult = crInstall(&info);
106  if (nInstResult!=0)
107  {
108  // Something goes wrong!
109  TCHAR szErrorMsg[512];
110  szErrorMsg[0]=0;
111 
112  crGetLastErrorMsg(szErrorMsg, 512);
113  printf("%s\n", szErrorMsg);
114 
115 
116  ErrorUtils::ShowError(_L("Exception handling registration problem"), String(szErrorMsg));
117 
118  assert(nInstResult==0);
119  }
120 }
