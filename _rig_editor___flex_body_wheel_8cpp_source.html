39  Ogre::Vector3 axis_point_outer = m_axis_outer_node->GetPosition();
40 
41  // Prepare buffers
42  m_vertices.clear();
43  m_vertices.reserve(m_definition.num_rays * 4);
44  m_vertices.push_back(axis_point_inner);
45  m_vertices.push_back(axis_point_outer);
46 
47  unsigned int num_edges = (m_definition.num_rays * 25);
48  unsigned int axis_index_inner = 0;
49  unsigned int axis_index_outer = 1;
50  unsigned int rigidity_index = 2; // Always reserve space for rigidity node, even if unused
51  unsigned int base_node_index = 3;
52  bool rigidity_node_use_outer; // Only relevant if rigidity node exists
53  if (m_rigidity_node != nullptr)
54  {
55  Ogre::Vector3 rigidity_pos = m_rigidity_node->GetPosition();
56 
57  unsigned int num_edges = (m_definition.num_rays * 26);
58  m_vertices.push_back(rigidity_pos);
59 
60  // Find out which side of the wheel is closer to the rigidity node
61  float distance_inner = rigidity_pos.squaredDistance(axis_point_inner); // Squared is faster
62  float distance_outer = rigidity_pos.squaredDistance(axis_point_outer);
63  rigidity_node_use_outer = distance_outer < distance_inner;
64  }
65  else
66  {
67  m_vertices.push_back(Ogre::Vector3::ZERO); // Dummy value for rigidity node
68  }
69  m_edges.clear();
70  m_edges.reserve(num_edges);
71 
72  // Init calculation
73  Ogre::Vector3 axis_vector = axis_point_inner - axis_point_outer;
74  axis_vector.normalise();
75  Ogre::Vector3 rim_ray_vector = axis_vector.perpendicular() * m_definition.rim_radius;
76  Ogre::Quaternion rim_ray_rotator = Ogre::Quaternion(Ogre::Degree(-360.f / (m_definition.num_rays * 2)), axis_vector);
77 
78  // Vertices - rim
79  for (unsigned int i = 0; i < m_definition.num_rays; i++)
80  {
81  // Outer ring
82  m_vertices.push_back(axis_point_inner + rim_ray_vector);
83  rim_ray_vector = rim_ray_rotator * rim_ray_vector;
84 
85  // Inner ring
86  m_vertices.push_back(axis_point_outer + rim_ray_vector);
87  rim_ray_vector = rim_ray_rotator * rim_ray_vector;
88  }
89 
90  Ogre::Vector3 tyre_ray_vector = axis_vector.perpendicular() * m_definition.tyre_radius;
91  Ogre::Quaternion& tyre_ray_rotator = rim_ray_rotator;
92  tyre_ray_vector = tyre_ray_rotator * tyre_ray_vector;
93 
94  // Vertices - tyre
95  for (unsigned int i = 0; i < m_definition.num_rays; i++)
96  {
97  // Outer ring
98  m_vertices.push_back(axis_point_inner + tyre_ray_vector);
99  tyre_ray_vector = tyre_ray_rotator * tyre_ray_vector;
100 
101  // Inner ring
102  m_vertices.push_back(axis_point_outer + tyre_ray_vector);
103  tyre_ray_vector = tyre_ray_rotator * tyre_ray_vector;
104  }
105 
106  // Indices - rim
107  for (unsigned int i = 0; i < m_definition.num_rays; i++)
108  {
109  // Rim axis to rim ring
110  unsigned int rim_inner_index = base_node_index + (i * 2);
111  unsigned int rim_outer_index = rim_inner_index + 1;
112 
113  m_edges.push_back(Edge(axis_index_inner, rim_inner_index, Edge::TYPE_FLEXBODYWHEEL_RIM_CONNECTION));
114  m_edges.push_back(Edge(axis_index_outer, rim_outer_index, Edge::TYPE_FLEXBODYWHEEL_RIM_CONNECTION));
115  m_edges.push_back(Edge(axis_index_outer, rim_inner_index, Edge::TYPE_FLEXBODYWHEEL_RIM_CONNECTION));
116  m_edges.push_back(Edge(axis_index_inner, rim_outer_index, Edge::TYPE_FLEXBODYWHEEL_RIM_CONNECTION));
117 
118  // Reinforcement rim ring
119  unsigned int rim_next_inner_index = base_node_index + (((i + 1) % m_definition.num_rays) * 2);
120  unsigned int rim_next_outer_index = rim_next_inner_index + 1;
121 
122  m_edges.push_back(Edge(rim_inner_index, rim_outer_index, Edge::TYPE_FLEXBODYWHEEL_RIM_REINFORCEMENT));
123  m_edges.push_back(Edge(rim_inner_index, rim_next_inner_index, Edge::TYPE_FLEXBODYWHEEL_RIM_REINFORCEMENT));
124  m_edges.push_back(Edge(rim_outer_index, rim_next_outer_index, Edge::TYPE_FLEXBODYWHEEL_RIM_REINFORCEMENT));
125  m_edges.push_back(Edge(rim_outer_index, rim_next_inner_index, Edge::TYPE_FLEXBODYWHEEL_RIM_REINFORCEMENT));
126  }
127 
128  // Indices - tyre
129  for (unsigned int i = 0; i < m_definition.num_rays; i++)
130  {
131 
132  int rim_inner_index = base_node_index + (2 * i);
133  int rim_outer_index = rim_inner_index + 1;
134  int tire_inner_index = rim_inner_index + (2 * m_definition.num_rays);
135  int tire_previous_inner_index = (i == 0) ? (base_node_index + (4 * m_definition.num_rays) - 2) : tire_inner_index - 2;
136  int tire_outer_index = tire_inner_index + 1;
137  int tire_previous_outer_index = tire_previous_inner_index + 1;
138  int tire_next_inner_index = base_node_index + (2 * m_definition.num_rays) + (2 * (i % m_definition.num_rays));
139  int tire_next_outer_index = tire_next_inner_index + 1;
140 
141  // Rim to tire tread
142  m_edges.push_back(Edge(rim_inner_index, tire_inner_index , Edge::TYPE_FLEXBODYWHEEL_TYRE_CONNECTION));
143  m_edges.push_back(Edge(rim_inner_index, tire_previous_outer_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_CONNECTION));
144  m_edges.push_back(Edge(rim_inner_index, tire_previous_inner_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_CONNECTION));
145  m_edges.push_back(Edge(rim_outer_index, tire_inner_index , Edge::TYPE_FLEXBODYWHEEL_TYRE_CONNECTION));
146  m_edges.push_back(Edge(rim_outer_index, tire_outer_index , Edge::TYPE_FLEXBODYWHEEL_TYRE_CONNECTION));
147  m_edges.push_back(Edge(rim_outer_index, tire_previous_outer_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_CONNECTION));
148 
149  // Reinforcement tire tread
150  m_edges.push_back(Edge(tire_inner_index, tire_outer_index , Edge::TYPE_FLEXBODYWHEEL_TYRE_REINFORCEMENT));
151  m_edges.push_back(Edge(tire_inner_index, tire_next_inner_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_REINFORCEMENT));
152  m_edges.push_back(Edge(tire_outer_index, tire_next_outer_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_REINFORCEMENT));
153  m_edges.push_back(Edge(tire_outer_index, tire_next_inner_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_REINFORCEMENT));
154 
155  // Rigidity edge
156  int rigidity_target_index = rigidity_node_use_outer ? tire_outer_index : tire_inner_index;
157  m_edges.push_back(Edge(rigidity_index, rigidity_target_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_RIGIDITY));
158  }
159 
160  //tiretread anti collapse reinforcements, using precalced support beams
161  for (unsigned int i=0; i<m_definition.num_rays; i++)
162  {
163  int tire_inner_index = base_node_index + i*2 + m_definition.num_rays*2;
164  int tire_outer_index = tire_inner_index + 1;
165 
166  m_edges.push_back(Edge(axis_index_inner, tire_inner_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_REINFORCEMENT));
167  m_edges.push_back(Edge(axis_index_outer, tire_outer_index, Edge::TYPE_FLEXBODYWHEEL_TYRE_REINFORCEMENT));
168  }

171 
172  this->SetGeometryIsDirty(false);
173  this->SetHasRayCountChanged(false);

188 void FlexBodyWheel::Update(AllWheelsAggregateData *all_data)
189 {
190  assert(all_data != nullptr);
191  FlexBodyWheelAggregateData& data = all_data->flexbodywheels_data;
192  if (data.num_elements == 0)
193  {
194  return;
195  }
196 
197  // Ray count
198  if (data.IsRayCountUniform() && data.num_rays != m_definition.num_rays)
199  {
200  m_definition.num_rays = data.num_rays;
201  this->SetGeometryIsDirty(true);
202  this->SetHasRayCountChanged(true);
203  }
204  // Radius
205  if (data.IsTyreRadiusUniform())
206  {
207  this->SetGeometryIsDirty(true);
208  m_definition.tyre_radius = data.tyre_radius;
209  }
210  if (data.IsRimRadiusUniform())
211  {
212  this->SetGeometryIsDirty(true);
213  m_definition.rim_radius = data.rim_radius;
214  }
215  // Nodes
216  if (data.num_elements == 1)
217  {
218  if (data.axis_nodes[0] != m_definition.nodes[0]) {
219  m_definition.nodes[0] = data.axis_nodes[0];
220  this->SetGeometryIsDirty(true);
221  }
222  if (data.axis_nodes[1] != m_definition.nodes[1]) {
223  m_definition.nodes[1] = data.axis_nodes[1];
224  this->SetGeometryIsDirty(true);
225  }
226  if (data.rigidity_node != m_definition.rigidity_node) {
227  m_definition.rigidity_node = data.rigidity_node;
228  this->SetGeometryIsDirty(true);
229  }
230  if (data.reference_arm_node != m_definition.reference_arm_node) {
231  m_definition.reference_arm_node = data.reference_arm_node;
232  }
233  }
234  // Other
235  if (data.IsBrakingModeUniform()) { m_definition.braking = data.braking; }
236  if (data.IsMassUniform()) { m_definition.mass = data.mass; }
237  if (data.IsPropulsionModeUniform()) { m_definition.propulsion = data.propulsion; }
238  if (data.IsRimDampingUniform()) { m_definition.beam_defaults->damping_constant = data.rim_damping; }
239  if (data.IsRimMeshNameUniform()) { m_definition.rim_mesh_name = data.rim_mesh_name; }
240  if (data.IsRimSpringUniform()) { m_definition.beam_defaults->springiness = data.rim_spring; }
241  if (data.IsSideUniform()) { m_definition.side = data.is_right_side ? RigDef::MeshWheel::SIDE_RIGHT : RigDef::MeshWheel::SIDE_LEFT; }
242  if (data.IsTyreDampingUniform()) { m_definition.tyre_damping = data.tyre_damping; }
243  if (data.IsTyreMeshNameUniform()) { m_definition.tyre_mesh_name = data.tyre_mesh_name; }
244  if (data.IsTyreSpringUniform()) { m_definition.tyre_springiness = data.tyre_spring; }
245 }
