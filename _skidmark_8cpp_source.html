43 int SkidmarkManager::loadDefaultModels()
44 {
45  LOG("SkidmarkManager loading default models");
46  // check if we have a config file
47  String group = "";
48  try
49  {
50  group = ResourceGroupManager::getSingleton().findGroupContainingResource("skidmarks.cfg");
51  } catch(...)
52  {
53  }
54  // emit a warning if we did not found the file
55  if (group.empty())
56  {
57  LOG("skidmarks| skidmarks.cfg not found");
58  return 1;
59  }
60 
61  // open the file for reading
62  DataStreamPtr ds = ResourceGroupManager::getSingleton().openResource("skidmarks.cfg", group);
63  String line = "";
64  String currentModel = "";
65 
66  while (!ds->eof())
67  {
68  line=ds->getLine();
69  StringUtil::trim(line);
70 
71  if (line.empty() || line[0]==';')
72  continue;
73 
74  StringVector args = StringUtil::split(line, ",");
75 
76  if (args.size() == 1)
77  {
78  currentModel = line;
79  continue;
80  }
81 
82  // process the line if we got a model
83  if (!currentModel.empty())
84  processLine(args, currentModel);
85  }
86  return 0;
87 }

231 void Skidmark::updatePoint()
232 {
233  Vector3 thisPoint = wheel->lastContactType?wheel->lastContactOuter:wheel->lastContactInner;
234  Vector3 axis = wheel->lastContactType?(wheel->refnode1->RelPosition - wheel->refnode0->RelPosition):(wheel->refnode0->RelPosition - wheel->refnode1->RelPosition);
235  Vector3 thisPointAV = thisPoint + axis * 0.5f;
236  Real distance = 0;
237  Real maxDist = maxDistance;
238  String texture = "none";
239  SkidmarkManager::getSingleton().getTexture("default", wheel->lastGroundModel->name, wheel->lastSlip, texture);
240 
241  // dont add points with no texture
242  if (texture == "none") return;
243 
244  if (wheel->speed > 1) maxDist *= wheel->speed;
245 
246  if (!objects.size())
247  {
248  // add first bucket
249  addObject(thisPoint, texture);
250  } else
251  {
252  // check existing buckets
253  skidmark_t skid = objects.back();
254 
255  distance = skid.lastPointAv.distance(thisPointAV);
256  // too near to update?
257  if (distance < minDistance)
258  {
259  //LOG("E: too near for update");
260  return;
261  }
262 
263  // change ground texture if required
264  if (skid.pos > 0 && skid.groundTexture[0] != texture)
265  {
266  // new object with new texture
267  if (distance > maxDist)
268  {
269  // to far away for connection
270  addObject(thisPoint, texture);
271  } else
272  {
273  // add new bucket with connection to last bucket
274  Vector3 lp1 = objects.back().points[objects.back().pos-1];
275  Vector3 lp2 = objects.back().points[objects.back().pos-2];
276  addObject(lp1, texture);
277  addPoint(lp2, distance, texture);
278  addPoint(lp1, distance, texture);
279  }
280  } else
281  {
282  // no texture change required :D
283 
284  // far enough for new bucket?
285  if (skid.pos >= (int)skid.points.size())
286  {
287  if (distance > maxDist)
288  {
289  // to far away for connection
290  addObject(thisPoint, texture);
291  } else
292  {
293  // add new bucket with connection to last bucket
294  Vector3 lp1 = objects.back().points[objects.back().pos-1];
295  Vector3 lp2 = objects.back().points[objects.back().pos-2];
296  addObject(lp1, texture);
297  addPoint(lp2, distance, texture);
298  addPoint(lp1, distance, texture);
299  }
300  }
301  else if (distance > maxDistance)
302  {
303  // just new bucket, no connection to last bucket
304  addObject(thisPoint, texture);
305  }
306  }
307  }
308 
309  skidmark_t skid = objects.back();
310 
311  float overaxis = 0.2f;
312  // tactics: we always choose the latest oint and then create two points
313 
314  // XXX: TO BE IMPROVED
315  //Vector3 groundNormal = Vector3::ZERO;
316  //hfinder->getNormalAt(thisPoint.x, thisPoint.y, thisPoint.z, &groundNormal);
317 
318  //LOG("ground normal: "+TOSTRING(wheel->refnode1->RelPosition.dotProduct(groundNormal)));
319 
320  // choose node wheel by the latest added point
321  if (!wheel->lastContactType)
322  {
323  // choose inner
324  //LOG("inner");
325  addPoint(wheel->lastContactInner - (axis * overaxis), distance, texture);
326  addPoint(wheel->lastContactInner + axis + (axis * overaxis), distance, texture);
327  } else
328  {
329  // choose outer
330  //LOG("outer");
331  addPoint(wheel->lastContactOuter + axis + (axis * overaxis), distance, texture);
332  addPoint(wheel->lastContactOuter - (axis * overaxis), distance, texture);
333  }
334 
335  // save as last point (in the middle of the wheel)
336  objects.back().lastPointAv = thisPointAV;
337 
338  /*
339  // debug code: adds boxes to the average point
340  SceneNode *sn = mNode->getParentSceneNode()->createChildSceneNode();
341  sn->attachObject(gEnv->ogreSceneManager->createEntity("addPointTRACK"+TOSTRING(objects.back().lastPointAv) +TOSTRING(axis), "beam.mesh"));
342  sn->setPosition(thisPointAV);
343  sn->setScale(0.1f, 0.01f, 0.1f);
344  */
345 }

358 void Skidmark::update()
359 {
360  if (!objects.size()) return;
361  skidmark_t skid = objects.back();
362  Vector3 vaabMin = skid.points[0];
363  Vector3 vaabMax = skid.points[0];
364  skid.obj->beginUpdate(0);
365  bool behindEnd = false;
366  Vector3 lastValid = Vector3::ZERO;
367  int to_counter=0;
368  float tcox_counter=0;
369 
370  for (int i = 0; i < lenght; i++, to_counter++)
371  {
372  if (i>=skid.pos) behindEnd=true;
373 
374  if (to_counter>3)
375  to_counter=0;
376 
377  if (!behindEnd)
378  tcox_counter += skid.faceSizes[i] / minDistance;
379 
380  while(tcox_counter>1)
381  tcox_counter--;
382 
383  if (behindEnd)
384  {
385  skid.obj->position(lastValid);
386  skid.obj->textureCoord(0,0);
387  } else
388  {
389  skid.obj->position(skid.points[i]);
390 
391  Vector2 tco = tex_coords[to_counter];
392  tco.x *= skid.faceSizes[i] / minDistance; // scale texture according face size
393  skid.obj->textureCoord(tco);
394 
395  lastValid = skid.points[i];
396  }
397 
398  if (skid.points[i].x < vaabMin.x) vaabMin.x = skid.points[i].x;
399  if (skid.points[i].y < vaabMin.y) vaabMin.y = skid.points[i].y;
400  if (skid.points[i].z < vaabMin.z) vaabMin.z = skid.points[i].z;
401  if (skid.points[i].x > vaabMax.x) vaabMax.x = skid.points[i].x;
402  if (skid.points[i].y > vaabMax.y) vaabMax.y = skid.points[i].y;
403  if (skid.points[i].z > vaabMax.z) vaabMax.z = skid.points[i].z;
404  }
405  skid.obj->end();
406 
407  skid.obj->setBoundingBox(AxisAlignedBox(vaabMin, vaabMax));
408 
409  // Use infinite AAB to always stay visible
410  /*
411  // for debugging only
412  AxisAlignedBox aabInf;
413  aabInf.setInfinite();
414  skid.obj->setBoundingBox(aabInf);
415  */
416 
417  mDirty = false;
418 }
