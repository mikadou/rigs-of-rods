100 inline void replaceString(std::string &str, std::string searchString, std::string replaceString)
101 {
102  std::string::size_type pos = 0;
103  while ( (pos = str.find(searchString, pos)) != std::string::npos )
104  {
105  str.replace( pos, searchString.size(), replaceString );
106  pos++;
107  }
108 }
