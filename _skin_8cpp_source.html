103 void Skin::replaceMaterialTextures(Ogre::String materialName)
104 {
105  MaterialPtr mat = MaterialManager::getSingleton().getByName(materialName);
106  if (!mat.isNull())
107  {
108  for (int t = 0; t < mat->getNumTechniques(); t++)
109  {
110  Technique *tech = mat->getTechnique(0);
111  if (!tech) continue;
112  for (int p=0; p < tech->getNumPasses(); p++)
113  {
114  Pass *pass = tech->getPass(p);
115  if (!pass) continue;
116  for (int tu = 0; tu < pass->getNumTextureUnitStates(); tu++)
117  {
118  TextureUnitState *tus = pass->getTextureUnitState(tu);
119  if (!tus) continue;
120 
121  //if (tus->getTextureType() != TEX_TYPE_2D) continue; // only replace 2d images
122  // walk the frames, usually there is only one
123  for (unsigned int fr=0; fr<tus->getNumFrames(); fr++)
124  {
125  String textureName = tus->getFrameTextureName(fr);
126  std::map<Ogre::String, Ogre::String>::iterator it = replaceTextures.find(textureName);
127  if (it != replaceTextures.end())
128  {
129  textureName = it->second; //getReplacementForTexture(textureName);
130  tus->setFrameTextureName(textureName, fr);
131  }
132  }
133  }
134  }
135  }
136  }

139 void Skin::replaceMeshMaterials(Ogre::Entity *e)
140 {
141  if (!e) return;
142 
143  // make it unique FIRST, otherwise we change the base material ...
144  uniquifyMeshMaterials(e);
145 
146  // then walk the entity and look for replacements
147  for (int n=0; n<(int)e->getNumSubEntities();n++)
148  {
149  SubEntity *subent = e->getSubEntity(n);
150  String materialName = subent->getMaterialName();
151  std::map<Ogre::String, Ogre::String>::iterator it = replaceMaterials.find(stripMaterialNameUniqueNess(materialName));
152  if (it != replaceMaterials.end())
153  {
154  materialName = it->second;
155  subent->setMaterialName(materialName);
156  } else
157  {
158  // look for texture replacements
159  replaceMaterialTextures(subent->getMaterialName());
160  }
161  }

164 void Skin::uniquifyMeshMaterials(Ogre::Entity *e)

172  // MAGGIICCCC
