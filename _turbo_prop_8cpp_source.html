00227 void Turboprop::updateForces(float dt, int doUpdate)

00232         float altitude=nodes[noderef].AbsPosition.y;
00233         //float sea_level_temperature=273.15+15.0; //in Kelvin
00234         float sea_level_pressure=101325; //in Pa
00235         //float airtemperature=sea_level_temperature-altitude*0.0065; //in Kelvin

00240         SoundScriptManager::getSingleton().modulate(trucknum, mod_id, rpm);

00245     //evaluate the rotation speed
00246     float velacc=0;
00247     for (int i=0; i<numblades; i++) velacc+=(nodes[nodep[i]].Velocity-nodes[noderef].Velocity).length();
00248     rpm=(velacc/numblades) * RAD_PER_SEC_TO_RPM / radius;
00249     //check for broken prop
00250     Vector3 avg=Vector3::ZERO;
00251     for (int i=0; i<numblades; i++) avg+=nodes[nodep[i]].RelPosition;
00252     avg=avg/numblades;
00253     if ((avg-nodes[noderef].RelPosition).length()>0.4)
00254     {
00255         failed=true;
00256     }
00257     //evaluate engine power
00258     float enginepower=0; //in kilo-Watt
00259     float warmupfactor=1.0;
00260     if (warmup)
00261     {
00262         warmupfactor=(timer-warmupstart)/warmuptime;
00263         if (warmupfactor>=1.0) warmup=false;
00264     }
00265     float revpenalty=1.0;
00266     if (reverse) revpenalty=0.5;
00267     if (!failed && ignition) enginepower=(0.0575+throtle*revpenalty*0.9425)*fullpower*warmupfactor;
00268     //the magic formula
00269     float enginecouple=0.0; //in N.m
00270     if (rpm>10.0) enginecouple=9549.3*enginepower/rpm;
00271     else enginecouple=9549.3*enginepower/10.0;
00272     indicated_torque=enginecouple;
00273     
00274     if (torquenode!=-1)
00275     {
00276         Vector3 along=nodes[noderef].RelPosition-nodes[nodeback].RelPosition;
00277         Plane ppl=Plane(along, 0);
00278         Vector3 orth=ppl.projectVector(nodes[noderef].RelPosition)-ppl.projectVector(nodes[torquenode].RelPosition);
00279         Vector3 cdir=orth.crossProduct(along);
00280         cdir.normalise();
00281         nodes[torquenode].Forces+=(enginecouple/torquedist)*cdir;
00282     }
00283     
00284     float tipforce=(enginecouple/radius)/numblades;
00285     //okay, now we know the contribution from the engine
00286 
00287     //pitch
00288     if (fixed_pitch>0) pitch=fixed_pitch; else
00289     {
00290         if (!reverse)
00291         {
00292             if (throtle<0.01)
00293             {
00294                 //beta range
00295                 if (pitch>0 && rpm<regspeed*1.4) pitch-=pitchspeed*dt;
00296                 if (rpm>regspeed*1.4) pitch+=pitchspeed*dt;
00297             }
00298             else
00299             {
00300                 float dpitch=rpm-regspeed;
00301                 if (dpitch>pitchspeed) dpitch=pitchspeed;

00304             }
00305         } else
00306         {
00307             if (rpm<regspeed*1.1)
00308             {
00309                 if (pitch<-4.0) pitch+=pitchspeed*dt;
00310                 else pitch-=pitchspeed*dt;
00311             }
00312             if (rpm>regspeed*1.11)
00313             {
00314                 pitch-=pitchspeed*dt;
00315             }
00316         }
00317     }
00318     if (!failed)
00319     {
00320         axis=nodes[noderef].RelPosition-nodes[nodeback].RelPosition;
00321         axis.normalise();
00322     }
00323     //estimate amount of energy
00324     float estrotenergy=0.5*numblades*nodes[nodep[0]].mass*radius*radius*(rpm/RAD_PER_SEC_TO_RPM)*(rpm/RAD_PER_SEC_TO_RPM);

00343             {
00344                 //proportion
00345                 float proport=((float)j+0.5)/6.0;
00346                 //evaluate wind direction

00373                 //lift

00375                 eforce+=lift*liftv;

00377

00386             //correct amount of energy

00389             if (correctfactor>1000.0) correctfactor=1000.0;

00396             //add drag
00397             Vector3 wind=-nodes[nodep[i]].Velocity;
