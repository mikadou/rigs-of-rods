00188 #ifdef USE_CAELUM
00189     // caelum needs to know that we changed the cameras
00190     if (gEnv->sky && gEnv->frameListener->loading_state == TERRAIN_LOADED)
00191         gEnv->sky->notifyCameraChanged(mVidCam);

00193 #endif // USE_CAELUM
00194

00196     if (rttTex) rttTex->update();
00197

00199 
00200     // get the normal of the camera plane now
00201     Vector3 normal=(-(truck->nodes[nref].smoothpos - truck->nodes[nz].smoothpos)).crossProduct(-(truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos));
00202     normal.normalise();
00203 
00204     // add user set offset
00205     Vector3 pos = truck->nodes[camNode].smoothpos +
00206         (offset.x * normal) +
00207         (offset.y * (truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos)) +
00208         (offset.z * (truck->nodes[nref].smoothpos - truck->nodes[nz].smoothpos));
00209 
00210     //avoid the camera roll
00211     // camup orientates to frustrum of world by default -> rotating the cam related to trucks yaw, lets bind cam rotation videocamera base (nref,ny,nz) as frustum
00212     // could this be done faster&better with a plane setFrustumExtents ?
00213     Vector3 frustumUP = truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos;
00214     frustumUP.normalise();
00215     mVidCam->setFixedYawAxis(true, frustumUP);
00216 
00217     // camRole 1 = mirror
00218     if (camRole == 1)
00219     {
00220         //rotate the normal of the mirror by user rotation setting so it reflects correct
00221         normal = rotation * normal;
00222         // merge camera direction and reflect it on our plane
00223         mVidCam->setDirection((pos - gEnv->mainCamera->getPosition()).reflect(normal));
00224     } else
00225     {
00226         // this is a videocamera
00227         if (camRole == -1)
00228         {
00229             // rotate the camera according to the nodes orientation and user rotation
00230             Vector3 refx = truck->nodes[nz].smoothpos - truck->nodes[nref].smoothpos;
00231             refx.normalise();
00232             Vector3 refy = truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos;
00233             refy.normalise();
00234             Quaternion rot = Quaternion(-refx, -refy, -normal);
00235             mVidCam->setOrientation(rot * rotation); // rotate the camera orientation towards the calculated cam direction plus user rotation
00236         } else
00237         {
00238             // we assume this is a tracking videocamera
00239             normal = truck->nodes[lookat].smoothpos - pos;
00240             normal.normalise();
00241             Vector3 refx = truck->nodes[nz].smoothpos - truck->nodes[nref].smoothpos;
00242             refx.normalise();
00243             // why does this flip ~2-3° around zero orientation and only with trackercam. back to slower crossproduct calc, a bit slower but better .. sigh
00244             // Vector3 refy = truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos;
00245             Vector3 refy = refx.crossProduct(normal);
00246             refy.normalise();
00247             Quaternion rot = Quaternion(-refx, -refy, -normal);
00248             mVidCam->setOrientation(rot*rotation); // rotate the camera orientation towards the calculated cam direction plus user rotation
00249         }
00250     }
00251 
00252     if (debugMode)
00253     {
00254         debugNode->setPosition(pos);
