184 void VideoCamera::update(float dt)
185 {
186  if (RoR::Application::GetGuiManager()->GetPauseMenuVisible()) return; //Stop everything when pause menu is visible
187 
188 #ifdef USE_CAELUM
189  // caelum needs to know that we changed the cameras
190  if (gEnv->sky && gEnv->frameListener->loading_state == TERRAIN_LOADED)
191  gEnv->sky->notifyCameraChanged(mVidCam);
192 
193 #endif // USE_CAELUM
194 
195  // update the texture now, otherwise shuttering
196  if (rttTex) rttTex->update();
197 
198  if (rwMirror) rwMirror->update();
199 
200  // get the normal of the camera plane now
201  Vector3 normal=(-(truck->nodes[nref].smoothpos - truck->nodes[nz].smoothpos)).crossProduct(-(truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos));
202  normal.normalise();
203 
204  // add user set offset
205  Vector3 pos = truck->nodes[camNode].smoothpos +
206  (offset.x * normal) +
207  (offset.y * (truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos)) +
208  (offset.z * (truck->nodes[nref].smoothpos - truck->nodes[nz].smoothpos));
209 
210  //avoid the camera roll
211  // camup orientates to frustrum of world by default -> rotating the cam related to trucks yaw, lets bind cam rotation videocamera base (nref,ny,nz) as frustum
212  // could this be done faster&better with a plane setFrustumExtents ?
213  Vector3 frustumUP = truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos;
214  frustumUP.normalise();
215  mVidCam->setFixedYawAxis(true, frustumUP);
216 
217  // camRole 1 = mirror
218  if (camRole == 1)
219  {
220  //rotate the normal of the mirror by user rotation setting so it reflects correct
221  normal = rotation * normal;
222  // merge camera direction and reflect it on our plane
223  mVidCam->setDirection((pos - gEnv->mainCamera->getPosition()).reflect(normal));
224  } else
225  {
226  // this is a videocamera
227  if (camRole == -1)
228  {
229  // rotate the camera according to the nodes orientation and user rotation
230  Vector3 refx = truck->nodes[nz].smoothpos - truck->nodes[nref].smoothpos;
231  refx.normalise();
232  Vector3 refy = truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos;
233  refy.normalise();
234  Quaternion rot = Quaternion(-refx, -refy, -normal);
235  mVidCam->setOrientation(rot * rotation); // rotate the camera orientation towards the calculated cam direction plus user rotation
236  } else
237  {
238  // we assume this is a tracking videocamera
239  normal = truck->nodes[lookat].smoothpos - pos;
240  normal.normalise();
241  Vector3 refx = truck->nodes[nz].smoothpos - truck->nodes[nref].smoothpos;
242  refx.normalise();
243  // why does this flip ~2-3 around zero orientation and only with trackercam. back to slower crossproduct calc, a bit slower but better .. sigh
244  // Vector3 refy = truck->nodes[nref].smoothpos - truck->nodes[ny].smoothpos;
245  Vector3 refy = refx.crossProduct(normal);
246  refy.normalise();
247  Quaternion rot = Quaternion(-refx, -refy, -normal);
248  mVidCam->setOrientation(rot*rotation); // rotate the camera orientation towards the calculated cam direction plus user rotation
249  }
250  }
251 
252  if (debugMode)
253  {
254  debugNode->setPosition(pos);
255  debugNode->setOrientation(mVidCam->getOrientation());
256  }
257 
258  // set the new position
259  mVidCam->setPosition(pos);
260 }

262 VideoCamera *VideoCamera::Setup(RigSpawner *rig_spawner, RigDef::VideoCamera & def)
263 {
264  try
265  {
266  Ogre::MaterialPtr mat = Ogre::MaterialManager::getSingleton().getByName(def.material_name);
267  if (mat.isNull())
268  {
269  std::stringstream msg;
270  msg << "Unknown material: '" << def.material_name << "', trying to continue...";
271  rig_spawner->AddMessage(RigSpawner::Message::TYPE_ERROR, msg.str());
272  return nullptr;
273  }
274 
275  Beam *rig = rig_spawner->GetRig();
276 
277  // clone the material to stay unique
278  std::stringstream mat_clone_name;
279  mat_clone_name << rig->truckname << def.material_name << "_" << counter;
280  counter++;
281  MaterialPtr mat_clone = rig_spawner->CloneMaterial(def.material_name, mat_clone_name.str());
282 
283  /* we need to find and replace any materials that could come afterwards */
284  if (rig->materialReplacer != nullptr)
285  {
286  rig->materialReplacer->addMaterialReplace(mat->getName(), mat_clone_name.str());
287  }
288 
289  VideoCamera *v = new VideoCamera(rig);
290  v->fov = def.field_of_view;
291  v->minclip = def.min_clip_distance;
292  v->maxclip = def.max_clip_distance;
293  v->nz = rig_spawner->GetNodeIndexOrThrow(def.left_node);
294  v->ny = rig_spawner->GetNodeIndexOrThrow(def.bottom_node);
295  v->nref = rig_spawner->GetNodeIndexOrThrow(def.reference_node);
296  v->offset = def.offset;
297  v->switchoff = def.camera_mode; // add performance switch off ->meeds fix, only "always on" supported yet
298  v->materialName = mat_clone_name.str();
299  v->mirrorSize = Vector2(def.texture_width, def.texture_height);
300 
301  /* camera name */
302  if (! def.camera_name.empty())
303  {
304  v->vidCamName = def.camera_name;
305  }
306  else
307  {
308  v->vidCamName = def.material_name; /* Fallback */
309  }
310 
311  //rotate camera picture 180, skip for mirrors
312  float rotation_z = (def.camera_role != 1) ? def.rotation.z + 180 : def.rotation.z;
313  v->rotation
314  = Ogre::Quaternion(Ogre::Degree(rotation_z), Ogre::Vector3::UNIT_Z)
315  * Ogre::Quaternion(Ogre::Degree(def.rotation.y), Ogre::Vector3::UNIT_Y)
316  * Ogre::Quaternion(Ogre::Degree(def.rotation.x), Ogre::Vector3::UNIT_X);
317 
318  // set alternative camposition (optional)
319  v->camNode = rig_spawner->GetNodeIndexOrThrow(def.alt_reference_node.IsValidAnyState() ? def.alt_reference_node : def.reference_node);
320 
321  // set alternative lookat position (optional)
322  int camera_role = def.camera_role;
323  if (def.alt_orientation_node.IsValidAnyState())
324  {
325  v->lookat = rig_spawner->GetNodeIndexOrThrow(def.alt_orientation_node);
326  camera_role = 0; // this is a tracecam, overwrite mode setting
327  }
328  else
329  {
330  v->lookat = -1;
331  }
332  v->camRole = camera_role; // -1= camera, 0 = trackcam, 1 = mirror
333 
334  v->init();
335 
336  return v;
337  }
338  catch(...)
339  {
340  return nullptr;
341  }
342 }
