01010     void dumpThreads( Dumper dumper ) {
01011         u64 rawDuration = ( Timer::getticks() - globalStart );
01012 
01013         Caller *accumulate = new Caller( "/Top Callers" ), *packer = new Caller( "/Thread Packer" );
01014         Buffer<Caller *> packedThreads;
01015 
01016         dumper.Init();
01017         dumper.GlobalInfo( rawDuration );
01018 
01019         threads.AcquireGlobalLock();    
01020 
01021         // crawl the list of theads and store their data in to packer
01022         Buffer<Root> &threadsref = *threads.list;
01023         for ( u32 i = 0; i < threadsref.Size(); i++ ) {
01024             Root &thread = threadsref[i];
01025 
01026             // if the thread is no longer active, the lock won't be valid
01027             bool active = ( thread.root->IsActive() );
01028             if ( active ) {
01029                 thread.threadState->threadLock.Acquire();
01030                 // disable requiring our local lock in case the caller is in our thread, accumulate will try to set it otherwise
01031                 Caller::thisThread.requireThreadLock = false;
01032                 for ( Caller *walk = thread.threadState->activeCaller; walk; walk = walk->GetParent() )
01033                     walk->GetTimer().SoftStop();
01034             }
01035 
01036 #if defined(__PROFILER_CONSOLIDATE_THREADS__)
01037             // merge the thread into the packer object, will result in 1 caller per thread name, not 1 caller per thread instance
01038             Caller::foreach::Merger( packer ).addFrom( thread.root );
01039             Caller *child = packer->FindOrCreate( thread.root->GetName() );
01040 
01041             // add the child to the list of threads to dump (use the active flag to indicate if it's been added)
01042             if ( !child->IsActive() ) {
01043                 packedThreads.Push( child );
01044                 child->SetActive( true );
01045             }
01046 #else
01047             // create a dummy entry for each thread (fake a name with the address of the thread root)
01048             Caller *stub = packer->FindOrCreate( (const char *)thread.root );
01049             Caller::foreach::Merger( stub ).addFrom( thread.root );
01050             Caller *stubroot = stub->FindOrCreate( thread.root->GetName() );
01051             stubroot->SetParent( NULL ); // for proper crawling
01052             packedThreads.Push( stubroot );
01053 #endif
01054 
01055             if ( active ) {
01056                 Caller::thisThread.requireThreadLock = true;
01057                 thread.threadState->threadLock.Release();
01058             }
01059         }
01060 
01061         // working on local data now, don't need the threads lock any more
01062         threads.ReleaseGlobalLock();    
01063 
01064         // do the pre-computations on the gathered threads
01065         Caller::ComputeChildTicks preprocessor( *accumulate );
01066         for ( u32 i = 0; i < packedThreads.Size(); i++ )
01067             preprocessor( packedThreads[i] );
01068 
01069         dumper.ThreadsInfo( Caller::maxStats( Caller::Max::TotalCalls ), Caller::mTimerOverhead, Caller::mRdtscOverhead );
01070 
01071         // print the gathered threads
01072         u64 sumTicks = 0;
01073         for ( u32 i = 0; i < packedThreads.Size(); i++ ) {
01074             Caller *root = packedThreads[i];
01075             u64 threadTicks = root->GetTimer().ticks;
01076             sumTicks += threadTicks;
01077             Caller::mGlobalDuration = threadTicks;
01078             dumper.PrintThread( root );
01079         }
01080 
01081         // print the totals, use the summed total of ticks to adjust percentages
01082         Caller::mGlobalDuration = sumTicks;
01083         dumper.PrintAccumulated( accumulate );      
01084         dumper.Finish();
01085 
01086         delete accumulate;
01087         delete packer;
01088     }
