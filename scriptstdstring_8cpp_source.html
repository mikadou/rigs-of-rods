624  sign = true;
625  *end++;
626  }
627  else if( *end == '+' )
628  *end++;
629 
630  asINT64 res = 0;
631  if( base == 10 )
632  {
633  while( *end >= '0' && *end <= '9' )
634  {
635  res *= 10;
636  res += *end++ - '0';
637  }
638  }
639  else if( base == 16 )
640  {
641  while( (*end >= '0' && *end <= '9') ||
642  (*end >= 'a' && *end <= 'f') ||
643  (*end >= 'A' && *end <= 'F') )
644  {
645  res *= 16;
646  if( *end >= '0' && *end <= '9' )
647  res += *end++ - '0';
648  else if( *end >= 'a' && *end <= 'f' )
649  res += *end++ - 'a' + 10;
650  else if( *end >= 'A' && *end <= 'F' )
651  res += *end++ - 'A' + 10;
652  }
653  }
654 
655  if( byteCount )
656  *byteCount = end - val.c_str();
657 
658  if( sign )
659  res = -res;
660 
661  return res;
662 }

666 double parseFloat(const string &val, asUINT *byteCount)
667 {
668  char *end;
669 
670  // WinCE doesn't have setlocale. Some quick testing on my current platform
671  // still manages to parse the numbers such as "3.14" even if the decimal for the
672  // locale is ",".
673 #if !defined(_WIN32_WCE) && !defined(ANDROID)
674  // Set the locale to C so that we are guaranteed to parse the float value correctly
675  char *orig = setlocale(LC_NUMERIC, 0);
676  setlocale(LC_NUMERIC, "C");
677 #endif
678 
679  double res = strtod(val.c_str(), &end);
680 
681 #if !defined(_WIN32_WCE) && !defined(ANDROID)
682  // Restore the locale
683  setlocale(LC_NUMERIC, orig);
684 #endif
685 
686  if( byteCount )
687  *byteCount = end - val.c_str();
688 
689  return res;
