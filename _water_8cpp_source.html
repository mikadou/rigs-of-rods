00580 {

00583     {
00584         // constant height, sea is flat as pancake

00593     // we will store the result in this variable, init it with the default height
00594     float result = wHeight;
00595     // now walk through all the wave trains. One 'train' is one sin/cos set that will generate once wave. All the trains together will sum up, so that they generate a 'rough' sea

00602             amp = wavetrains[i].maxheight;

00604         // calculate the sinus with the values of the config file and add it to the result
00605         result += amp * sin(Math::TWO_PI * ( \
