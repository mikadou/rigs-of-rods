256 void SurveyMapManager::update(Ogre::Real dt)
257 {
258  if ( dt == 0 ) return;
259 
260  Beam *curr_truck = BeamFactory::getSingleton().getCurrentTruck();
261  mVelocity = 0.0f;
262 
263  if (curr_truck)
264  {
265  mVelocity = curr_truck->nodes[0].Velocity.length();
266  }
267 
268  if (RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_TOGGLE_VIEW))
269  {
270  toggleMapView();
271  }
272 
273  if (RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_TOGGLE_ICONS))
274  {
275  mMapEntitiesVisible = !mMapEntitiesVisible;
276  }
277 
278  if (mMapMode == SURVEY_MAP_NONE) return;
279 
280  static bool needsUpdate = true;
281  switch (mMapMode)
282  {
283  case SURVEY_MAP_SMALL:
284  if (RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_ALPHA))
285  {
286  toggleMapAlpha();
287  }
288  if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_SURVEY_MAP_ZOOM_IN))
289  {
290  setMapZoomRelative(1.0f);
291  }
292  if (RoR::Application::GetInputEngine()->getEventBoolValue(EV_SURVEY_MAP_ZOOM_OUT))
293  {
294  setMapZoomRelative(-1.0f);
295  }
296  if (getMapZoom() > 0.0f)
297  {
298  if (curr_truck)
299  {
300  setMapCenter(curr_truck->getPosition(), mMapCenterThreshold * (1 - mMapZoom));
301  } else
302  {
303  setMapCenter(gEnv->player->getPosition(), mMapCenterThreshold * (1 - mMapZoom));
304  }
305  } else
306  {
307  setMapCenter(Vector3(mMapSize.x / 2.0f, 0.0f, mMapSize.z / 2.0f));
308  }
309  break;
310 
311  case SURVEY_MAP_BIG:
312  setMapCenter(Vector3(mMapSize.x / 2.0f, 0.0f, mMapSize.z / 2.0f));
313 
314  if (getMapZoom() > 0.0f)
315  {
316  setMapZoom(0.0f, mMapTextureNeedsUpdate, false);
317  mMapTextureNeedsUpdate = false;
318  }
319 
320  if (curr_truck &&
321  gEnv->cameraManager &&
322  gEnv->cameraManager->hasActiveBehavior() &&
323  !gEnv->cameraManager->gameControlsLocked())
324  {
325  if (mVelocity > 5.0f || gEnv->cameraManager->getCurrentBehavior() == CameraManager::CAMERA_BEHAVIOR_VEHICLE_CINECAM)
326  {
327  setWindowPosition(1, -1, 0.3f);
328  //setAlpha(mAlpha);
329  mMapMode = SURVEY_MAP_SMALL;
330  } else
331  {
332  // Soft cross-fade into small map
333  //setAlpha(1.0f / sqrt(std::max(1.0f, mVelocity - 1.0f)), false);
334  }
335  } else
336  {
337  if (getAlpha() != mAlpha)
338  {
339  setAlpha(mAlpha);
340  }
341 
342  if (RoR::Application::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_ALPHA))
343  {
344  toggleMapAlpha();
345  }
346  }
347  break;
348 
349  default:
350  break;
351  }
352 
353  updateMapEntityPositions();
354 }

356 void SurveyMapManager::toggleMapView()
357 {
358  mMapMode = (mMapMode + 1) % SURVEY_MAP_END;
359 
360  if (mMapMode == SURVEY_MAP_BIG && (mVelocity > 5.0f ||
361  (gEnv->cameraManager &&
362  gEnv->cameraManager->hasActiveBehavior() &&
363  gEnv->cameraManager->getCurrentBehavior() == CameraManager::CAMERA_BEHAVIOR_VEHICLE_CINECAM)))
364  {
365  mMapMode = (mMapMode + 1) % SURVEY_MAP_END;
366  }
367 
368  if (mMapMode == SURVEY_MAP_NONE)
369  {
370  setVisibility(false);
371  } else
372  {
373  if (mMapMode == SURVEY_MAP_SMALL)
374  {
375  setWindowPosition(1, -1, 0.3f);
376  } else if (mMapMode == SURVEY_MAP_BIG)
377  {
378  setWindowPosition(0, 0, 0.98f);
379  }
380  setAlpha(mAlpha);
381  setVisibility(true);
382  }
