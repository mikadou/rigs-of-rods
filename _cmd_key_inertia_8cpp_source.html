00031 Real CmdKeyInertia::calcCmdKeyDelay(Real cmdInput,int cmdKey,Real dt)
00032 {
00033     if (!cmdKeyInertia[cmdKey].startSpline || !cmdKeyInertia[cmdKey].stopSpline)
00034     {
00035         return cmdInput;
00036     }
00037 
00038     Real calculatedOutput=cmdKeyInertia[cmdKey].lastOutput;
00039     Real lastOutput=cmdKeyInertia[cmdKey].lastOutput;
00040     // rel difference to calculate if we have to use start values(accelerating) or stop values
00041     Real relDiff=fabs(cmdInput)-fabs(lastOutput);
00042     // difference to calculate if were are on the negative side
00043     Real absDiff=cmdInput-lastOutput;
00044     // if the value is close to our input, reset the timer
00045     if (fabs(absDiff)<0.002)
00046         cmdKeyInertia[cmdKey].time=0;
00047     // +dt after the timer had been set to zero prevents the motion to stop at 0.002
00048     cmdKeyInertia[cmdKey].time+=dt;
00049 
00050     Real startFactor=cmdKeyInertia[cmdKey].startDelay*cmdKeyInertia[cmdKey].time;
00051     Real stopFactor=cmdKeyInertia[cmdKey].stopDelay*cmdKeyInertia[cmdKey].time;
00052     // positive values between 0 and 1
00053     if (absDiff>0)
00054     {   // we have to accelerate our last outout to the new commanded input
00055         if (relDiff>0)
00056             calculatedOutput=lastOutput+calculateCmdOutput(startFactor,cmdKeyInertia[cmdKey].startSpline);
00057         if (relDiff<0)
00058             // we have to deccelerate our last outout to the new commanded input
00059             calculatedOutput=lastOutput+calculateCmdOutput(stopFactor,cmdKeyInertia[cmdKey].stopSpline);
00060         if (calculatedOutput>cmdInput)
00061             // if the calculated value is bigger than input set to input to avoid overshooting
00062             calculatedOutput=cmdInput;
00063     }
00064     // negative values, mainly needed for hydros, between 0 and -1
00065     if (absDiff<0)
00066     {
00067         if (relDiff>0)
00068             calculatedOutput=lastOutput-calculateCmdOutput(startFactor,cmdKeyInertia[cmdKey].startSpline);
00069         if (relDiff<0)
00070             calculatedOutput=lastOutput-calculateCmdOutput(stopFactor,cmdKeyInertia[cmdKey].stopSpline);
00071         if (calculatedOutput<cmdInput)
00072             calculatedOutput=cmdInput;
00073     }
00074     cmdKeyInertia[cmdKey].lastOutput=calculatedOutput;
00075     return calculatedOutput;
00076 }
