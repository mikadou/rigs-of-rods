168 void PreviewRenderer::render2dviews(Beam *truck, Camera *cam, float minCameraRadius)
169 {
170  float ominCameraRadius = minCameraRadius;
171  Vector3 truckPos = truck->getPosition();
172 
173  for (int o=0; o<2; o++)
174  {
175  String oext = "ortho.";
176  if (o == 0)
177  {
178  cam->setProjectionType(PT_ORTHOGRAPHIC);
179  minCameraRadius = ominCameraRadius * 2.1f;
180  }
181  else if (o == 1)
182  {
183  oext = "3d.";
184  cam->setProjectionType(PT_PERSPECTIVE);
185  minCameraRadius = ominCameraRadius * 2.4f;
186  }
187 
188  for (int i=0; i<2; i++)
189  {
190  String ext = "normal.";
191  if (i == 0)
192  {
193  truck->hideSkeleton(true);
194  } else if (i == 1)
195  {
196  ext = "skeleton.";
197  // now show the skeleton
198  truck->showSkeleton(true, true);
199  truck->updateSimpleSkeleton();
200  }
201  ext = oext + ext;
202 
203  // 3d first :)
204  cam->setPosition(Vector3(truckPos.x - minCameraRadius, truckPos.y * 1.5f, truckPos.z - minCameraRadius));
205  cam->lookAt(truckPos);
206  render(ext + "perspective");
207 
208  // right
209  cam->setPosition(Vector3(truckPos.x, truckPos.y, truckPos.z - minCameraRadius));
210  cam->lookAt(truckPos);
211  render(ext + "right");
212 
213  // left
214  cam->setPosition(Vector3(truckPos.x, truckPos.y, truckPos.z + minCameraRadius));
215  cam->lookAt(truckPos);
216  render(ext + "left");
217 
218  // front
219  cam->setPosition(Vector3(truckPos.x - minCameraRadius, truckPos.y, truckPos.z));
220  cam->lookAt(truckPos);
221  render(ext + "front");
222 
223  // back
224  cam->setPosition(Vector3(truckPos.x + minCameraRadius, truckPos.y, truckPos.z));
225  cam->lookAt(truckPos);
226  render(ext + "back");
227 
228  // top
229  cam->setPosition(Vector3(truckPos.x, truckPos.y + minCameraRadius, truckPos.z + 0.01f));
230  cam->lookAt(truckPos);
231  render(ext + "top");
232 
233  // bottom
234  cam->setPosition(Vector3(truckPos.x, truckPos.y - minCameraRadius, truckPos.z + 0.01f));
235  cam->lookAt(truckPos);
236  render(ext + "bottom");
237 
238  }
239  }
