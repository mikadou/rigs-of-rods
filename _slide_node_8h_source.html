69 static inline Ogre::Vector3 nearestPoint(const Ogre::Vector3& pt1,
71  const Ogre::Vector3& pt2,
72  const Ogre::Vector3& tp)
73 {
74  const Ogre::Vector3 a = tp - pt1;
75  const Ogre::Vector3 b = fast_normalise(pt2-pt1);
76 
77  return pt1 + (a.dotProduct(b)) * b;
78 }

80 static inline Ogre::Vector3 nearestPointOnLine(const Ogre::Vector3& pt1,
81  const Ogre::Vector3& pt2,
82  const Ogre::Vector3& tp)
83 {
84  Ogre::Vector3 a = tp - pt1;
85  Ogre::Vector3 b = pt2 - pt1;
86  Ogre::Real len = fast_length(b);
87 
88  b = fast_normalise(b);
89  len = std::max(0.0f, std::min(a.dotProduct(b), len));
90  b *= len;
91 
92  a = pt1;
93  a += b;
94  return a;
95 }
