00071         const Ogre::Vector3& pt2,
00072         const Ogre::Vector3& tp)
00073 {   
00074     const Ogre::Vector3 a = tp - pt1;
00075     const Ogre::Vector3 b = fast_normalise(pt2-pt1);

00080 static inline Ogre::Vector3 nearestPointOnLine(const Ogre::Vector3& pt1,
00081         const Ogre::Vector3& pt2,
00082         const Ogre::Vector3& tp)
00083 {   
00084     Ogre::Vector3 a = tp - pt1;
00085     Ogre::Vector3 b = pt2 - pt1;
00086     Ogre::Real len = fast_length(b);
00087 
00088     b = fast_normalise(b);
00089     len = std::max(0.0f, std::min(a.dotProduct(b), len));   
00090     b *= len;
00091     
00092     a = pt1;
00093     a += b;
00094     return a;
00095 }
