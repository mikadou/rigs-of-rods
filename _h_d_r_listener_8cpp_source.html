00033 {

00035     Ogre::CompositionTechnique::TextureDefinitionIterator defIter =
00036         instance->getTechnique()->getTextureDefinitionIterator();

00038     {
00039         Ogre::CompositionTechnique::TextureDefinition* def =
00040             defIter.getNext();
00041         if (def->name == "rt_bloom0")
00042         {
00043             mBloomSize = (int)def->width; // should be square
00044             // Calculate Gaussian texture offsets & weights
00045             float deviation = 3.0f;
00046             float texelSize = 1.0f / (float)mBloomSize;
00047 
00048             // central sample, no offset
00049             mBloomTexOffsetsHorz[0][0] = 0.0f;
00050             mBloomTexOffsetsHorz[0][1] = 0.0f;
00051             mBloomTexOffsetsVert[0][0] = 0.0f;
00052             mBloomTexOffsetsVert[0][1] = 0.0f;
00053             mBloomTexWeights[0][0] = mBloomTexWeights[0][1] =
00054                 mBloomTexWeights[0][2] = Ogre::Math::gaussianDistribution(0, 0, deviation);
00055             mBloomTexWeights[0][3] = 1.0f;
00056 
00057             // 'pre' samples
00058             for (int i = 1; i < 8; ++i)
00059             {
00060                 mBloomTexWeights[i][0] = mBloomTexWeights[i][1] =
00061                     mBloomTexWeights[i][2] = 1.25f * Ogre::Math::gaussianDistribution(i, 0, deviation);
00062                 mBloomTexWeights[i][3] = 1.0f;
00063                 mBloomTexOffsetsHorz[i][0] = i * texelSize;
00064                 mBloomTexOffsetsHorz[i][1] = 0.0f;
00065                 mBloomTexOffsetsVert[i][0] = 0.0f;
00066                 mBloomTexOffsetsVert[i][1] = i * texelSize;
00067             }
00068             // 'post' samples
00069             for (int i = 8; i < 15; ++i)
00070             {
00071                 mBloomTexWeights[i][0] = mBloomTexWeights[i][1] =
00072                     mBloomTexWeights[i][2] = mBloomTexWeights[i - 7][0];
00073                 mBloomTexWeights[i][3] = 1.0f;
00074 
00075                 mBloomTexOffsetsHorz[i][0] = -mBloomTexOffsetsHorz[i - 7][0];
00076                 mBloomTexOffsetsHorz[i][1] = 0.0f;
00077                 mBloomTexOffsetsVert[i][0] = 0.0f;
00078                 mBloomTexOffsetsVert[i][1] = -mBloomTexOffsetsVert[i - 7][1];

00081         }
