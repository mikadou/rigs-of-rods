65 static unsigned hardware_concurrency()
66 {
67  #if defined(PTW32_VERSION) || defined(__hpux)
68  return pthread_num_processors_np();
69  #elif defined(_GNU_SOURCE)
70  return get_nprocs();
71  #elif defined(__APPLE__) || defined(__FreeBSD__)
72  int count;
73  size_t size = sizeof(count);
74  return sysctlbyname("hw.ncpu", &count, &size, NULL, 0) ? 0 : count;
75  #elif defined(BOOST_HAS_UNISTD_H) && defined(_SC_NPROCESSORS_ONLN)
76  int const count = sysconf(_SC_NPROCESSORS_ONLN);
77  return (count > 0) ? count : 0;
78  #else
79  return 0;
80  #endif
81 }

438 bool BeamFactory::truckIntersectionCollAABB(int a, int b)
439 {
440  if (trucks[a]->collisionBoundingBoxes.empty() && trucks[b]->collisionBoundingBoxes.empty())
441  {
442  return truckIntersectionAABB(a, b);
443  } else if (trucks[a]->collisionBoundingBoxes.empty())
444  {
445  for (std::vector<AxisAlignedBox>::iterator it = trucks[b]->collisionBoundingBoxes.begin(); it != trucks[b]->collisionBoundingBoxes.end(); ++it)
446  if (it->intersects(trucks[a]->boundingBox))
447  return true;
448  } else if (trucks[b]->collisionBoundingBoxes.empty())
449  {
450  for (std::vector<AxisAlignedBox>::iterator it = trucks[a]->collisionBoundingBoxes.begin(); it != trucks[a]->collisionBoundingBoxes.end(); ++it)
451  if (it->intersects(trucks[b]->boundingBox))
452  return true;
453  } else
454  {
455  for (std::vector<AxisAlignedBox>::iterator it_a = trucks[a]->collisionBoundingBoxes.begin(); it_a != trucks[a]->collisionBoundingBoxes.end(); ++it_a)
456  for (std::vector<AxisAlignedBox>::iterator it_b = trucks[b]->collisionBoundingBoxes.begin(); it_b != trucks[b]->collisionBoundingBoxes.end(); ++it_b)
457  if (it_a->intersects(*it_b))
458  return true;
459  }
460 
461  return false;
462 }

507 void BeamFactory::recursiveActivation(int j)
508 {
509  if (!trucks[j] || trucks[j]->state > DESACTIVATED) return;
510 
511  for (int t=0; t < free_truck; t++)
512  {
513  if (t == j || !trucks[t]) continue;
514  if ((trucks[t]->state == SLEEPING || trucks[t]->state == MAYSLEEP || trucks[t]->state == GOSLEEP || (trucks[t]->state == DESACTIVATED && trucks[t]->sleepcount >= 5)) &&
515  predictTruckIntersectionCollAABB(t, j))
516  {
517  trucks[t]->desactivate(); // make the truck not leading but active

847

866 
867  // update 2D replay if activated
868  if (tdr) tdr->update(dt);

871  for (int t=0; t < free_truck; t++)
872  {
873  if (!trucks[t]) continue;

884

886  break;

890  trucks[t]->engine->update(dt, 1);

892  trucks[t]->sendStreamData();
