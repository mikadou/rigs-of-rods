135 void CLASS::Import(
136  boost::shared_ptr<RigDef::Engine> engine,
137  boost::shared_ptr<RigDef::Engoption> engoption
138  )
139 {
140  if (!engine || !engoption)
141  {
142  this->SetVehicleHasEngine(false);
143  return;
144  }
145  this->SetVehicleHasEngine(true);
146 
147  // SECTION "ENGINE"
148 
149  m_editbox_shift_down_rpm->setCaption(TOSTRING(engine->shift_down_rpm));
150  m_editbox_shift_up_rpm->setCaption(TOSTRING(engine->shift_up_rpm));
151 
152  m_editbox_torque->setCaption(TOSTRING(engine->torque));
153 
154  // Gearbox
155  m_editbox_global_gear_ratio ->setCaption(TOSTRING(engine->global_gear_ratio));
156  m_editbox_reverse_gear_ratio->setCaption(TOSTRING(engine->reverse_gear_ratio));
157  m_editbox_neutral_gear_ratio->setCaption(TOSTRING(engine->neutral_gear_ratio));
158  m_editbox_first_gear_ratio ->setCaption(TOSTRING(engine->gear_ratios[0]));
159 
160  // Gearbox - forward gear list
161  auto itor = engine->gear_ratios.begin() + 1;
162  auto itor_end = engine->gear_ratios.end();
163  std::ostringstream ratios_line;
164  for (; itor != itor_end; ++itor)
165  {
166  m_forward_gears_textbox_empty = false;
167  ratios_line << *itor << " ";
168  }
169  m_editbox_more_gear_ratios->setCaption(ratios_line.str());
170 
171  // SECTION "ENGOPTION"
172  m_editbox_engoption_engine_inertia ->setCaption(TOSTRING(engoption->inertia));
173  m_editbox_engoption_clutch_force ->setCaption(TOSTRING(engoption->clutch_force));
174  m_editbox_engoption_shift_time ->setCaption(TOSTRING(engoption->shift_time));
175  m_editbox_engoption_clutch_time ->setCaption(TOSTRING(engoption->clutch_time));
176  m_editbox_engoption_post_shift_time ->setCaption(TOSTRING(engoption->post_shift_time));
177  m_editbox_engoption_stall_rpm ->setCaption(TOSTRING(engoption->stall_rpm));
178  m_editbox_engoption_idle_rpm ->setCaption(TOSTRING(engoption->idle_rpm));
179  m_editbox_engoption_max_idle_mixture->setCaption(TOSTRING(engoption->max_idle_mixture));
180  m_editbox_engoption_min_idle_mixture->setCaption(TOSTRING(engoption->min_idle_mixture));
181 
182  // Engine type
183  assert(engoption->type == Engoption::ENGINE_TYPE_c_CAR || engoption->type == Engoption::ENGINE_TYPE_t_TRUCK);
184  this->SetEngineType(engoption->type);
185 }

187 boost::shared_ptr<RigDef::Engine> CLASS::ExportEngine()
188 {
189  if (!m_vehicle_has_engine)
190  {
191  return boost::shared_ptr<RigDef::Engine>();
192  }
193  auto engine = boost::shared_ptr<RigDef::Engine>(new RigDef::Engine());
194 
195  engine->shift_down_rpm = PARSEREAL(m_editbox_shift_down_rpm->getCaption());
196  engine->shift_up_rpm = PARSEREAL(m_editbox_shift_up_rpm->getCaption());
197  engine->torque = PARSEREAL(m_editbox_torque->getCaption());
198 
199  // Gearbox
200  engine->global_gear_ratio = PARSEREAL(m_editbox_global_gear_ratio->getCaption());
201  engine->reverse_gear_ratio = PARSEREAL(m_editbox_reverse_gear_ratio->getCaption());
202  engine->neutral_gear_ratio = PARSEREAL(m_editbox_neutral_gear_ratio->getCaption());
203  engine->gear_ratios.push_back(PARSEREAL(m_editbox_first_gear_ratio->getCaption()));
204 
205  if (! m_forward_gears_textbox_empty)
206  {
207  // NOTE: Whitespace-only validation is done by onFocus* events.
208 
209  Ogre::String gears_str = m_editbox_more_gear_ratios->getCaption();
210  Ogre::StringUtil::trim(gears_str);
211  Ogre::StringVector tokens = Ogre::StringUtil::split(gears_str, " ");
212  for (auto itor = tokens.begin(); itor != tokens.end(); ++itor)
213  {
214  engine->gear_ratios.push_back(PARSEREAL(*itor));
215  }
216  }
217  return engine;
218 }
