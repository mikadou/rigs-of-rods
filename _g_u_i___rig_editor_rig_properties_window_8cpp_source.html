210 void CLASS::Export(RigEditor::RigProperties* data)
211 {
212  data->m_title = m_editbox_title->getCaption();
213  data->m_guid = m_editbox_guid->getCaption();
214 
215  bool has_uid = ! m_editbox_uid->getCaption().empty();
216  if (has_uid)
217  {
218  data->m_fileinfo.unique_id = m_editbox_uid->getCaption();
219  }
220  data->m_fileinfo._has_unique_id = has_uid;
221 
222  bool has_category = ! m_textbox_category_id->getCaption().empty();
223  if (has_category)
224  {
225  data->m_fileinfo.category_id = PARSEUINT(m_textbox_category_id->getCaption());
226  }
227  data->m_fileinfo._has_category_id = has_category;
228 
229  bool has_version = ! m_editbox_version->getCaption().empty();
230  if (has_version)
231  {
232  data->m_fileinfo.file_version = PARSEUINT(m_editbox_version->getCaption());
233  }
234  data->m_fileinfo._has_file_version_set = has_version;
235 
236  // Description
237  data->m_description.clear();
238  Ogre::String description_str = m_editbox_description->getCaption();
239  Ogre::StringUtil::trim(description_str);
240  if (! description_str.empty())
241  {
242  Ogre::StringVector lines = Ogre::StringUtil::split(description_str, "\r\n"); // Split over CR or LF
243  for (auto itor = lines.begin(); itor != lines.end(); ++itor)
244  {
245  if (! itor->empty()) // Empty line?
246  {
247  std::string line = RoR::Utils::TrimBlanksAndLinebreaks(*itor);
248  if (! line.empty())
249  {
250  data->m_description.push_back(line);
251  }
252  }
253  }
254  }
255 
256  // Authors
257  data->m_authors.clear();
258  Ogre::String authors_str = m_editbox_authors->getCaption();
259  Ogre::StringUtil::trim(authors_str);
260  if (! authors_str.empty())
261  {
262  Ogre::StringVector lines = Ogre::StringUtil::split(authors_str, "\n");
263  for (auto itor = lines.begin(); itor != lines.end(); ++itor)
264  {
265  Ogre::String line = *itor;
266  Ogre::StringUtil::trim(line);
267  if (line.empty())
268  {
269  continue;
270  }
271  Ogre::StringVector tokens = Ogre::StringUtil::split(line, " \t");
272  RigDef::Author author;
273  author.type = tokens[0];
274  if (tokens.size() > 1)
275  {
276  int account_id = PARSEINT(tokens[1]);
277  if (account_id > -1)
278  {
279  author.forum_account_id = static_cast<unsigned int>(account_id);
280  author._has_forum_account = true;
281  }
282 
283  if (tokens.size() > 2)
284  {
285  author.name = tokens[2];
286 
287  if (tokens.size() > 3)
288  {
289  author.email = tokens[3];
290  }
291  }
292  }
293 
294  data->m_authors.push_back(author);
295  }
296  }
297 
298  // Checkboxes
299  data->m_hide_in_chooser = m_checkbox_hide_in_chooser->getStateSelected();
300  data->m_forward_commands = m_checkbox_forwardcommands->getStateSelected();
301  data->m_import_commands = m_checkbox_importcommands->getStateSelected();
302  data->m_is_rescuer = m_checkbox_rescuer->getStateSelected();
303  data->m_disable_default_sounds = m_checkbox_disable_default_sounds->getStateSelected();
304  data->m_enable_advanced_deformation = m_checkbox_use_advanced_deform->getStateSelected();
305  data->m_rollon = m_checkbox_rollon->getStateSelected();
306 
307  // Section 'extcamera'
308  RigDef::ExtCamera::Mode extcamera_mode = RigDef::ExtCamera::MODE_CLASSIC;
309  extcamera_mode = m_radio_camera_behaviour_classic->getStateSelected() ? RigDef::ExtCamera::MODE_CLASSIC : extcamera_mode;
310  extcamera_mode = m_radio_camera_behaviour_cinecam->getStateSelected() ? RigDef::ExtCamera::MODE_CINECAM : extcamera_mode;
311  extcamera_mode = m_radio_camera_behaviour_node->getStateSelected() ? RigDef::ExtCamera::MODE_NODE : extcamera_mode;
312  data->m_extcamera.mode = extcamera_mode;
313  if (extcamera_mode == RigDef::ExtCamera::MODE_NODE)
314  {
315  std::string node_ref_str = m_editbox_extcamera_node->getCaption();
316  unsigned flags = Node::Ref::REGULAR_STATE_IS_VALID | Node::Ref::REGULAR_STATE_IS_NAMED; // Fileformatversion>=450 ~ Use named-only nodes
317  data->m_extcamera.node = Node::Ref(node_ref_str, 0, flags, 0);
318  }
319 
320  // Section 'set_skeleton_settings'
321  data->m_skeleton_settings.beam_thickness_meters = PARSEREAL(m_editbox_beam_thickness_meters->getCaption());
322  data->m_skeleton_settings.visibility_range_meters = PARSEREAL(m_editbox_beam_visibility_range_meters->getCaption());
323 
324  data->m_help_panel_material_name = m_editbox_help_panel_mat_name->getCaption();
325  data->m_globals_cab_material_name = m_editbox_cab_material_name->getCaption();
326  data->m_globals_dry_mass = PARSEREAL(m_editbox_dry_mass->getCaption());
327  data->m_globals_load_mass = PARSEREAL(m_editbox_load_mass->getCaption());
328  data->m_minimass = PARSEREAL(m_editbox_minimass->getCaption());
329
