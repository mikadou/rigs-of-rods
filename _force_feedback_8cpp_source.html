52 void ForceFeedback::setForces(Real roll, Real pitch, Real wspeed, Real dircommand, Real stress)
53 {
54  //LOG(String("ForceFeedback: R=")+TOSTRING(roll)+" D="+TOSTRING(dir)+" S="+TOSTRING(wspeed)+" H="+TOSTRING(stress));
55  if (!hydroEffect)
56  {
57  //we create effect at the last moment, because it does not works otherwise
58  hydroEffect=new OIS::Effect(OIS::Effect::ConstantForce, OIS::Effect::Constant);
59  hydroEffect->direction = OIS::Effect::North;
60  hydroEffect->trigger_button = 0;
61  hydroEffect->trigger_interval = 0;
62  hydroEffect->replay_length = OIS::Effect::OIS_INFINITE; // Linux/Win32: Same behaviour as 0.
63  hydroEffect->replay_delay = 0;
64  hydroEffect->setNumAxes(1);
65  OIS::ConstantEffect* hydroConstForce = dynamic_cast<OIS::ConstantEffect*>(hydroEffect->getForceEffect());
66  hydroConstForce->level = 0; //-10K to +10k
67  hydroConstForce->envelope.attackLength = 0;
68  hydroConstForce->envelope.attackLevel = (unsigned short)hydroConstForce->level;
69  hydroConstForce->envelope.fadeLength = 0;
70  hydroConstForce->envelope.fadeLevel = (unsigned short)hydroConstForce->level;
71 
72  ffdevice->upload(hydroEffect);
73  }
74 
75  OIS::ConstantEffect* hydroConstForce = dynamic_cast<OIS::ConstantEffect*>(hydroEffect->getForceEffect());
76  Real ff=-stress*stress_gain+dircommand*100.0*centering_gain*wspeed*wspeed;
77  if (ff>10000) ff=10000;
78  if (ff<-10000) ff=-10000;
79  hydroConstForce->level = ff; //-10K to +10k
80  ffdevice->modify(hydroEffect);
81 
82 }
