349  SWBaseSocket::SWBaseError error;
350  header_t head;

388 int Network::receivemessage(SWInetSocket *socket, header_t *head, char* content, unsigned int bufferlen)
389 {
390  SWBaseSocket::SWBaseError error;
391 
392  char buffer[MAX_MESSAGE_LENGTH];
393  //ensure that the buffer is clean after each received message!
394  memset(buffer, 0, MAX_MESSAGE_LENGTH);
395 
396  int hlen=0;
397  while (hlen<(int)sizeof(header_t))
398  {
399  int recvnum=socket->recv(buffer+hlen, sizeof(header_t)-hlen,&error);
400  if (recvnum<0)
401  {
402  LOG("NET receive error 1: " + TOSTRING(recvnum));
403  return -1;
404  }
405  hlen+=recvnum;
406  }
407 
408  memcpy(head, buffer, sizeof(header_t));
409 
410  if (head->size >= MAX_MESSAGE_LENGTH)
411  {
412  return -3;
413  }
414 
415  if (head->size>0)
416  {
417  if ((int)sizeof(header_t) > 0)
418  {
419  //read the rest
420  while (hlen<(int)sizeof(header_t)+(int)head->size)
421  {
422  int recvnum=socket->recv(buffer+hlen, (head->size+sizeof(header_t))-hlen,&error);
423  if (recvnum<0)
424  {
425  LOG("NET receive error 2: "+ TOSTRING(recvnum));
426  return -1;
427  }
428  hlen+=recvnum;
429  }
430  }
431  }
432  speed_bytes_recv_tmp += head->size + sizeof(header_t);
433 
434  memcpy(content, buffer+sizeof(header_t), bufferlen);
435  calcSpeed();
436  return 0;
437 }

497 void Network::receivethreadstart()
498 {
499  header_t header;
500 
501  char *buffer=(char*)malloc(MAX_MESSAGE_LENGTH);
502  //bool autoDl = (BSETTING("AutoDownload", false));
503  std::deque < stream_reg_t > streamCreationResults;
504  LOG("Receivethread starting");
505  // unlimited timeout, important!
506 
507  // wait for beamfactory to be existant before doing anything
508  // otherwise you can get runtime conditions
509  while(!BeamFactory::getSingletonPtr())
510  {
511  sleepMilliSeconds(1000);
512  };
513 
514 
515  socket.set_timeout(0,0);
516  while (!shutdown)
517  {
518  //get one message
519  int err=receivemessage(&socket, &header, buffer, MAX_MESSAGE_LENGTH);
520  //LOG("received data: " + TOSTRING(header.command) + ", source: "+TOSTRING(header.source) + ":"+TOSTRING(header.streamid) + ", size: "+TOSTRING(header.size));
521  if (err)
522  {
523  //this is an error!
524  char errmsg[256];
525  sprintf(errmsg, "Error %i while receiving data", err);
526  netFatalError(errmsg);
527  return;
528  }
529 
530  // check for stream registration errors and notify the remote client
531  if (BeamFactory::getSingletonPtr() && BeamFactory::getSingletonPtr()->getStreamRegistrationResults(&streamCreationResults))
532  {
533  while (!streamCreationResults.empty())
534  {
535  stream_reg_t r = streamCreationResults.front();
536  stream_register_t reg = r.reg;
537  sendmessage(&socket, MSG2_STREAM_REGISTER_RESULT, 0, sizeof(stream_register_t), (char *)&reg);
538  streamCreationResults.pop_front();
539  }
540  }
541 
542  // TODO: produce new streamable classes when required
543  if (header.command == MSG2_STREAM_REGISTER)
544  {
545  if (header.source == (int)myuid)
546  // our own stream, ignore
547  continue;
548  stream_register_t *reg = (stream_register_t *)buffer;
549  client_t *client = getClientInfo(header.source);
550  int playerColour = 0;
551  if (client) playerColour = client->user.colournum;
552 
553  String typeStr = "unknown";
554  switch(reg->type)
555  {
556  case 0: typeStr="truck"; break;
557  case 1: typeStr="character"; break;
558  case 3: typeStr="chat"; break;
559  };
560  LOG(" * received stream registration: " + TOSTRING(header.source) + ": "+TOSTRING(header.streamid) + ", type: "+typeStr);
561 
562  if (reg->type == 0)
563  {
564  // truck
565  BeamFactory::getSingleton().createRemote(header.source, header.streamid, reg, playerColour);
566  } else if (reg->type == 1)
567  {
568  // person
569  CharacterFactory::getSingleton().createRemote(header.source, header.streamid, reg, playerColour);
570  } else if (reg->type == 2)
571  {
572  // previously AITRAFFIC, unused for now
573  } else if (reg->type == 3)
574  {
575  // chat stream
576  ChatSystemFactory::getSingleton().createRemote(header.source, header.streamid, reg, playerColour);
577  }
578  continue;
579  }
580  else if (header.command == MSG2_STREAM_REGISTER_RESULT)
581  {
582  stream_register_t *reg = (stream_register_t *)buffer;
583  BeamFactory::getSingleton().addStreamRegistrationResults(header.source, reg);
584  LOG(" * received stream registration result: " + TOSTRING(header.source) + ": "+TOSTRING(header.streamid));
585  }
586  else if (header.source == -1 && (header.command == MSG2_UTF_CHAT || header.command == MSG2_UTF_PRIVCHAT))
587  {
588  // NOTE: this is only a shortcut for server messages, other UIDs propagate over the standard way
589  ChatSystem *cs = ChatSystemFactory::getSingleton().getFirstChatSystem();
590  if (cs) cs->addReceivedPacket(header, buffer);
591  continue;
592  }
593  else if (header.command == MSG2_NETQUALITY && header.source == -1)
594  {
595  if (header.size != sizeof(int))
596  continue;
597  int quality = *(int *)buffer;
598  setNetQuality(quality);
599  continue;
600  }
601  else if (header.command == MSG2_USER_LEAVE)
602  {
603  if (header.source == (int)myuid)
604  {
605  netFatalError(_L("disconnected: remote side closed the connection"), false);
606  return;
607  }
608 
609  // remove all things that belong to that user
610  client_t *client = getClientInfo(header.source);
611  if (client)
612  client->used = false;
613 
614  // now remove all possible streams
615  NetworkStreamManager::getSingleton().removeUser(header.source);
616 
617 #ifdef USE_MYGUI
618  if (GUI_MainMenu::getSingletonPtr() != nullptr) // Does menubar exist yet?
619  // Executed in main menu, but menubar isn't created until simulation starts.
620  {
621  // we can trigger this in the network thread as the function is thread safe.
622  GUI_MainMenu::getSingleton().triggerUpdateVehicleList();
623  }
624 #endif // USE_MYGUI
625  continue;
626  }
627  else if (header.command == MSG2_USER_INFO || header.command == MSG2_USER_JOIN)
628  {
629  if (header.source == (int)myuid)
630  {
631  // we got data about ourself!
632  memcpy(&userdata, buffer, sizeof(user_info_t));
633  CharacterFactory::getSingleton().localUserAttributesChanged(myuid);
634  // update our nickname
635  nickname = UTFString(userdata.username);
636  // save it in the Settings as well
637  SETTINGS.setUTFSetting(L"Nickname", nickname);
638  // update auth status
639  myauthlevel = userdata.authstatus;
640  } else
641  {
642  user_info_t *cinfo = (user_info_t*) buffer;
643  // data about someone else, try to update the array
644  bool found = false; // whether to add a new client
645  client_t *client = getClientInfo(header.source);
646  if (client)
647  {
648  memcpy(&client->user, cinfo, sizeof(user_info_t));
649 
650  // inform the streamfactories of a attribute change
651  CharacterFactory::getSingleton().netUserAttributesChanged(header.source, -1);
652  BeamFactory::getSingleton().netUserAttributesChanged(header.source, -1);
653  found = true;
654  } else
655  {
656  // find a free entry
657  MUTEX_LOCK(&clients_mutex);
658  for (int i=0; i<MAX_PEERS; i++)
659  {
660  if (clients[i].used)
661  continue;
662  clients[i].used = true;
663  memcpy(&clients[i].user, cinfo, sizeof(user_info_t));
664 
665  // inform the streamfactories of a attribute change
666  CharacterFactory::getSingleton().netUserAttributesChanged(header.source, -1);
667  BeamFactory::getSingleton().netUserAttributesChanged(header.source, -1);
668  break;
669  }
670  MUTEX_UNLOCK(&clients_mutex);
671  }
672  }
673 #ifdef USE_MYGUI
674  if (GUI_MainMenu::getSingletonPtr() != nullptr) // Does menubar exist yet?
675  // This code is executed in selector already, but Menubar isn't created until simulation launches.
676  {
677  // we can trigger this in the network thread as the function is thread safe.
678  GUI_MainMenu::getSingleton().triggerUpdateVehicleList();
679  }
680 #endif // USE_MYGUI
681 
682  continue;
683  }
684  else if (header.command == MSG2_GAME_CMD)
