77 void MeshWheel2::Update(AllWheelsAggregateData *all_data)
78 {
79  assert(all_data != nullptr);
80  MeshWheel2AggregateData& data = all_data->meshwheels2_data;
81  if (data.num_elements == 0)
82  {
83  return;
84  }
85 
86  // Ray count
87  if (data.IsRayCountUniform() && data.num_rays != m_definition.num_rays)
88  {
89  m_definition.num_rays = data.num_rays;
90  this->SetGeometryIsDirty(true);
91  this->SetHasRayCountChanged(true);
92  }
93  // Radius
94  if (data.IsTyreRadiusUniform())
95  {
96  this->SetGeometryIsDirty(true);
97  m_definition.tyre_radius = data.tyre_radius;
98  }
99  // Nodes
100  if (data.num_elements == 1)
101  {
102  if (data.axis_nodes[0] != m_definition.nodes[0]) {
103  m_definition.nodes[0] = data.axis_nodes[0];
104  this->SetGeometryIsDirty(true);
105  }
106  if (data.axis_nodes[1] != m_definition.nodes[1]) {
107  m_definition.nodes[1] = data.axis_nodes[1];
108  this->SetGeometryIsDirty(true);
109  }
110  if (data.rigidity_node != m_definition.rigidity_node) {
111  m_definition.rigidity_node = data.rigidity_node;
112  this->SetGeometryIsDirty(true);
113  }
114  if (data.reference_arm_node != m_definition.reference_arm_node) {
115  m_definition.reference_arm_node = data.reference_arm_node;
116  }
117  }
118  // Other
119  if (data.IsBrakingModeUniform()) { m_definition.braking = data.braking; }
120  if (data.IsBreakThresholdUniform()) { m_definition.beam_defaults->breaking_threshold_constant = data.rim_breaking_threshold; }
121  if (data.IsDeformThresholdUniform()) { m_definition.beam_defaults->deformation_threshold_constant = data.rim_deform_threshold; }
122  if (data.IsMassUniform()) { m_definition.mass = data.mass; }
123  if (data.IsPropulsionModeUniform()) { m_definition.propulsion = data.propulsion; }
124  if (data.IsRimDampingUniform()) { m_definition.beam_defaults->damping_constant = data.rim_damping; }
125  if (data.IsRimMeshNameUniform()) { m_definition.mesh_name = data.rim_mesh_name; }
126  if (data.IsRimRadiusUniform()) { m_definition.rim_radius = data.rim_radius; }
127  if (data.IsRimSpringUniform()) { m_definition.beam_defaults->springiness = data.rim_spring; }
128  if (data.IsSideUniform()) { m_definition.side = data.is_right_side ? RigDef::MeshWheel::SIDE_RIGHT : RigDef::MeshWheel::SIDE_LEFT; }
129  if (data.IsTyreDampingUniform()) { m_definition.tyre_damping = data.tyre_damping; }
130  if (data.IsTyreMaterialNameUniform()) { m_definition.material_name = data.tyre_material_name; }
131  if (data.IsTyreSpringUniform()) { m_definition.tyre_springiness = data.tyre_spring; }
132 }
