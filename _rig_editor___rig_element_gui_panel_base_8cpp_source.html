145 void RigElementGuiPanelBase::EditboxCommitValue(EditboxFieldSpec* spec)
146 {
147  MyGUI::InputManager::getInstance().setKeyFocusWidget(nullptr); // Remove focus
148  if (spec->editbox->getCaption().empty())
149  {
150  return;
151  }
152  if (spec->label != nullptr)
153  {
154  spec->label->setTextColour(m_text_color_default); // Mark "not uniform"
155  }
156  spec->SetSourceIsUniform();
157  bool value_updated = false;
158  if (spec->IsSourceFloat())
159  {
160  *spec->GetSourceFloat() = Ogre::StringConverter::parseReal(spec->editbox->getCaption()); // Save value
161  value_updated = true;
162  }
163  else if (spec->IsSourceInt())
164  {
165  *spec->GetSourceInt() = Ogre::StringConverter::parseInt(spec->editbox->getCaption()); // Save value
166  value_updated = true;
167  }
168  else if (spec->IsSourceString())
169  {
170  *spec->GetSourceString() = spec->editbox->getCaption(); // Save value
171  value_updated = true;
172  }
173  else if (spec->IsSourceNode())
174  {
175  RigDef::Node::Id* node_ptr = static_cast<RigDef::Node::Id*>(spec->GetSourceRawPtr());
176  Ogre::String value = spec->editbox->getCaption();
177  Ogre::StringUtil::trim(value);
178  if (value.empty())
179  {
180  node_ptr->Invalidate();
181  }
182  else
183  {
184  // Determine if numbered or named node is entered
185  auto itor = value.begin();
186  auto end = value.end();
187  bool pure_number = true;
188  for (; itor != end; ++itor)
189  {
190  if (!std::isdigit(*itor))
191  {
192  pure_number = false;
193  break;
194  }
195  }
196  if (pure_number)
197  {
198  int num_value = Ogre::StringConverter::parseInt(value);
199  node_ptr->SetNum(num_value);
200  }
201  else
202  {
203  node_ptr->SetStr(value);
204  }
205  }
206  value_updated = true;
207  }
208 
209  if (value_updated)
210  {
211  this->OnFieldValueChanged(spec);
212  }
213 }

403 void RigElementGuiPanelBase::ComboboxCommitValue(GenericFieldSpec* spec)
404 {
405  MyGUI::InputManager::getInstance().setKeyFocusWidget(nullptr); // Remove focus
406  assert(spec->field_widget);
407  MyGUI::ComboBox* combox = static_cast<MyGUI::ComboBox*>(spec->field_widget);
408  if (combox->getCaption().empty())
409  {
410  return;
411  }
412  if (spec->label != nullptr)
413  {
414  spec->label->setTextColour(m_text_color_default); // Mark "not uniform"
415  }
416  spec->SetSourceIsUniform();
417  // This code relies on combobox values being in format "%N - Text", where %N is a number
418  // Ogre::StringConverter::parse[Int/Real]() parses everything until whitespace, so it nicely extracts the number
419  if (spec->IsSourceFloat())
420  {
421  *spec->GetSourceFloat() = Ogre::StringConverter::parseReal(combox->getCaption());
422  }
423  else if (spec->IsSourceInt())
424  {
425  *spec->GetSourceInt() = Ogre::StringConverter::parseInt(combox->getCaption());
426  }
427  else if (spec->IsSourceBool())
428  {
429  int value = Ogre::StringConverter::parseInt(combox->getCaption());
430  *spec->GetSourceBool() = (value != 0);
431  }
432  else if (spec->IsSourceString())
433  {
434  *spec->GetSourceString() = combox->getCaption(); // Save value
435  }
436 }
