135 void HighlightBoxesDynamicMesh::AddBox(
136  Ogre::Vector3 const & extent_max,
137  Ogre::Vector3 const & extent_min,
138  Ogre::ColourValue const & color)
139 {
140  using namespace Ogre;
141 
142  Ogre::Vector3 line_lengths;
143  line_lengths.x = (extent_max.x - extent_min.x) / 4;
144  line_lengths.y = (extent_max.y - extent_min.y) / 4;
145  line_lengths.z = (extent_max.z - extent_min.z) / 4;
146 
147  // XYZ min
148  this->AddCorner(extent_min, line_lengths, Vector3(1,1,1), color);
149  // XY min, Z max
150  this->AddCorner(Vector3(extent_min.x, extent_min.y, extent_max.z), line_lengths, Vector3(1,1,-1), color);
151  // XZ min, Y max
152  this->AddCorner(Vector3(extent_min.x, extent_max.y, extent_min.z), line_lengths, Vector3(1,-1,1), color);
153  // YZ min, X max
154  this->AddCorner(Vector3(extent_max.x, extent_min.y, extent_min.z), line_lengths, Vector3(-1,1,1), color);
155  // XYZ max
156  this->AddCorner(extent_max, line_lengths, Vector3(-1, -1, -1), color);
157  // XY max, Z min
158  this->AddCorner(Vector3(extent_max.x, extent_max.y, extent_min.z), line_lengths, Vector3(-1,-1,1), color);
159  // XZ max, Y min
160  this->AddCorner(Vector3(extent_max.x, extent_min.y, extent_max.z), line_lengths, Vector3(-1,1,-1), color);
161  // YZ max, X min
162  this->AddCorner(Vector3(extent_min.x, extent_max.y, extent_max.z), line_lengths, Vector3(1,-1,-1), color);
163 }
