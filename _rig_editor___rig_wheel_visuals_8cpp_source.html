55 void RigWheelVisuals::BuildWheelsGeometryDynamicMesh(
56  RigEditor::Main* rig_editor,
57  std::vector<LandVehicleWheel*> & wheels
58  )
59 {
60  assert(rig_editor != nullptr);
61 
62  // Prepare material
63  if (! Ogre::MaterialManager::getSingleton().resourceExists("rig-editor-skeleton-wheels-material"))
64  {
65  Ogre::MaterialPtr node_mat = static_cast<Ogre::MaterialPtr>(
66  Ogre::MaterialManager::getSingleton().create("rig-editor-skeleton-wheels-material",
67  Ogre::ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME)
68  );
69 
70  node_mat->getTechnique(0)->getPass(0)->createTextureUnitState();
71  node_mat->getTechnique(0)->getPass(0)->getTextureUnitState(0)->setTextureFiltering(Ogre::TFO_ANISOTROPIC);
72  node_mat->getTechnique(0)->getPass(0)->getTextureUnitState(0)->setTextureAnisotropy(3);
73  node_mat->setLightingEnabled(false);
74  node_mat->setReceiveShadows(false);
75  }
76 
77  // Analyze
78  int wheels_vertex_count = 0;
79  int wheels_index_count = 0;
80  auto wheels_end = wheels.end();
81  for (auto wheels_itor = wheels.begin(); wheels_itor != wheels_end; ++wheels_itor)
82  {
83  wheels_index_count += (*wheels_itor)->GetEdges().size();
84  wheels_vertex_count = wheels_index_count * 2; // Vertices are generated per edge
85  }
86 
87  // Create mesh
88  assert(m_wheels_dynamic_mesh.get() == nullptr);
89  m_wheels_dynamic_mesh = std::unique_ptr<Ogre::ManualObject>(
90  rig_editor->GetOgreSceneManager()->createManualObject()
91  );
92  m_wheels_dynamic_mesh->estimateVertexCount(wheels_vertex_count);
93  m_wheels_dynamic_mesh->estimateIndexCount(wheels_index_count);
94  m_wheels_dynamic_mesh->setCastShadows(false);
95  m_wheels_dynamic_mesh->setDynamic(true);
96  m_wheels_dynamic_mesh->setRenderingDistance(300);
97 
98  // Init
99  m_wheels_dynamic_mesh->begin("rig-editor-skeleton-wheels-material", Ogre::RenderOperation::OT_LINE_LIST);
100 
101  // Process
102  for (auto wheels_itor = wheels.begin(); wheels_itor != wheels_end; ++wheels_itor)
103  {
104  auto & vertices = (*wheels_itor)->GetVertices();
105  auto & edges = (*wheels_itor)->GetEdges();
106 
107  auto edge_end = edges.end();
108  for (auto edge_itor = edges.begin(); edge_itor != edge_end; ++edge_itor)
109  {
110  LandVehicleWheel::Edge edge = *edge_itor;
111  Ogre::ColourValue color = LandVehicleWheel::Edge::GetColorValue(edge.type, rig_editor->GetConfig());
112 
113  m_wheels_dynamic_mesh->position(vertices[edge.indices[0]]);
114  m_wheels_dynamic_mesh->colour(color);
115  m_wheels_dynamic_mesh->position(vertices[edge.indices[1]]);
116  m_wheels_dynamic_mesh->colour(color);
117  }
118  }
119 
120  // Finalize
121  m_wheels_dynamic_mesh->end();
122 }

124 void RigWheelVisuals::RefreshWheelsDynamicMeshes(
125  Ogre::SceneNode* parent_scene_node,
126  RigEditor::Main* rig_editor,
127  std::vector<LandVehicleWheel*> & wheels
128  )
129 {
130  assert(parent_scene_node != nullptr);
131  assert(rig_editor != nullptr);
132 
133  bool is_any_selected = false;
134  bool is_any_hovered = false;
135  bool is_any_geometry_dirty = false;
136  bool is_any_state_dirty = false;
137  bool has_ray_count_changed = false;
138 
139  if (m_wheels_dynamic_mesh.get() == nullptr)
140  {
141  is_any_state_dirty = true;
142  has_ray_count_changed = true;
143  }
144 
145  auto itor_end = wheels.end();
146  for (auto itor = wheels.begin(); itor != itor_end; ++itor)
147  {
148  LandVehicleWheel* wheel = *itor;
149  is_any_selected = is_any_selected || wheel->IsSelected();
150  is_any_hovered = is_any_hovered || wheel->IsHovered();
151  is_any_state_dirty = is_any_state_dirty || wheel->IsSelectionStateDirty();
152  has_ray_count_changed = has_ray_count_changed || wheel->HasRayCountChanged();
153 
154  if (wheel->IsGeometryDirty())
155  {
156  is_any_geometry_dirty = true;
157  wheel->ReGenerateMeshData();
158  }
159  }
160 
161  if (is_any_geometry_dirty)
162  {
163  if (has_ray_count_changed)
164  {
165  // Rebuild the wheels dynamic mesh
166  bool attach_to_scene = false;
167  if (m_wheels_dynamic_mesh.get() != nullptr)
168  {
169  attach_to_scene = m_wheels_dynamic_mesh->isInScene();
170  m_wheels_dynamic_mesh->detachFromParent();
171  m_wheels_dynamic_mesh.reset();
172  }
173  this->BuildWheelsGeometryDynamicMesh(rig_editor, wheels);
174  if (attach_to_scene)
175  {
176  parent_scene_node->attachObject(m_wheels_dynamic_mesh.get());
177  }
178  }
179  else
180  {
181  // Update existing mesh
182  this->UpdateWheelsGeometryDynamicMesh(rig_editor, wheels);
183  }
184  }
185 }

237 void RigWheelVisuals::UpdateWheelsSelectionHighlightBoxes(
238  std::vector<LandVehicleWheel*> & wheels,
239  RigEditor::Main* rig_editor,
240  Ogre::SceneNode* parent_scene_node
241  )
242 {
243  // Check if any wheel is selected
244  auto end = wheels.end();
245  bool selected_found = false;
246  for (auto itor = wheels.begin(); itor != end; ++itor)
247  {
248  LandVehicleWheel* wheel = *itor;
249  if (wheel->IsSelected())
250  {
251  selected_found = true;
252  break;
253  }
254  }
255  if (!selected_found)
256  {
257  m_wheels_selected_dynamic_mesh->DetachFromScene();
258  this->SetIsSelectionDirty(false);
259  return;
260  }
261 
262  // Update selection highlight
263  float box_padding = rig_editor->GetConfig()->wheels_selection_highlight_boxes_padding;
264  m_wheels_selected_dynamic_mesh->BeginUpdate();
265  for (auto itor = wheels.begin(); itor != end; ++itor)
266  {
267  LandVehicleWheel* wheel = *itor;
268  if (wheel->IsSelected())
269  {
270  m_wheels_selected_dynamic_mesh->AddBox(
271  wheel->GetAabbEdgesMax(box_padding),
272  wheel->GetAabbEdgesMin(box_padding),
273  rig_editor->GetConfig()->wheels_selection_highlight_boxes_color);
274  }
275  }
276  m_wheels_selected_dynamic_mesh->EndUpdate();
277  m_wheels_selected_dynamic_mesh->AttachToScene(parent_scene_node);
278  this->SetIsSelectionDirty(false);
279 }
