00059 {

00065         Ogre::MaterialPtr node_mat = static_cast<Ogre::MaterialPtr>(
00066             Ogre::MaterialManager::getSingleton().create("rig-editor-skeleton-wheels-material",

00071         node_mat->getTechnique(0)->getPass(0)->getTextureUnitState(0)->setTextureFiltering(Ogre::TFO_ANISOTROPIC);

00082     {
00083         wheels_index_count += (*wheels_itor)->GetEdges().size();
00084         wheels_vertex_count = wheels_index_count * 2; // Vertices are generated per edge
00085     }
00086 
00087     // Create mesh

00089     m_wheels_dynamic_mesh = std::unique_ptr<Ogre::ManualObject>(
00090             rig_editor->GetOgreSceneManager()->createManualObject()
00091         );
00092     m_wheels_dynamic_mesh->estimateVertexCount(wheels_vertex_count);
00093     m_wheels_dynamic_mesh->estimateIndexCount(wheels_index_count);
00094     m_wheels_dynamic_mesh->setCastShadows(false);
00095     m_wheels_dynamic_mesh->setDynamic(true);
00096     m_wheels_dynamic_mesh->setRenderingDistance(300);
00097 
00098     // Init
00099     m_wheels_dynamic_mesh->begin("rig-editor-skeleton-wheels-material", Ogre::RenderOperation::OT_LINE_LIST);
00100 
00101     // Process
00102     for (auto wheels_itor = wheels.begin(); wheels_itor != wheels_end; ++wheels_itor)
00103     {
00104         auto & vertices = (*wheels_itor)->GetVertices();

00111             Ogre::ColourValue color = LandVehicleWheel::Edge::GetColorValue(edge.type, rig_editor->GetConfig());

00113             m_wheels_dynamic_mesh->position(vertices[edge.indices[0]]);

00124 void RigWheelVisuals::RefreshWheelsDynamicMeshes(

00126     RigEditor::Main* rig_editor,
00127     std::vector<LandVehicleWheel*> & wheels

00138 
00139     if (m_wheels_dynamic_mesh.get() == nullptr)

00141         is_any_state_dirty = true;
00142         has_ray_count_changed = true;

00144 
00145     auto itor_end = wheels.end();
00146     for (auto itor = wheels.begin(); itor != itor_end; ++itor)
00147     {
00148         LandVehicleWheel* wheel = *itor;
00149         is_any_selected       = is_any_selected || wheel->IsSelected();
00150         is_any_hovered        = is_any_hovered || wheel->IsHovered();
00151         is_any_state_dirty    = is_any_state_dirty || wheel->IsSelectionStateDirty();
00152         has_ray_count_changed = has_ray_count_changed || wheel->HasRayCountChanged();
00153 
00154         if (wheel->IsGeometryDirty())
00155         {
00156             is_any_geometry_dirty = true;
00157             wheel->ReGenerateMeshData();
00158         }
00159     }
00160 
00161     if (is_any_geometry_dirty)
00162     {
00163         if (has_ray_count_changed)
00164         {
00165             // Rebuild the wheels dynamic mesh
00166             bool attach_to_scene = false;
00167             if (m_wheels_dynamic_mesh.get() != nullptr)
00168             {
00169                 attach_to_scene = m_wheels_dynamic_mesh->isInScene();
00170                 m_wheels_dynamic_mesh->detachFromParent();
00171                 m_wheels_dynamic_mesh.reset();
00172             }
00173             this->BuildWheelsGeometryDynamicMesh(rig_editor, wheels);
00174             if (attach_to_scene)
00175             {
00176                 parent_scene_node->attachObject(m_wheels_dynamic_mesh.get());
00177             }
00178         }
00179         else
00180         {
00181             // Update existing mesh
00182             this->UpdateWheelsGeometryDynamicMesh(rig_editor, wheels);
00183         }
00184     }
